27 void cvDecimateFilter::apply() {
28  if (m_keepMode) {
29  return;
30  }
31 
32  if (!m_dataObject) {
33  CVLog::Error(QString("cvDecimateFilter::apply: null data object."));
34  return;
35  }
36  if (!m_meshMode) {
37  CVLog::Error(QString("cvDecimateFilter::apply: mesh supported only!"));
38  return;
39  }
40 
41  VtkUtils::vtkInitOnce(m_decimate);
42  m_decimate->SetInputData(m_dataObject);
43  m_decimate->SetTargetReduction(m_targetReduction);
44  m_decimate->SetFeatureAngle(m_featureAngle);
45  m_decimate->SetSplitAngle(m_splitAngle);
46  m_decimate->SetInflectionPointRatio(m_inflectionPointRatio);
47  m_decimate->SetPreserveTopology(m_preserveTopology);
48  m_decimate->SetSplitting(m_splitting);
49  m_decimate->SetPreSplitMesh(m_presplitMesh);
50  m_decimate->SetAccumulateError(m_accumulateError);
51  m_decimate->SetOutputPointsPrecision(m_outputPointsPrecision);
52  m_decimate->SetDegree(m_degree);
53  m_decimate->SetBoundaryVertexDeletion(m_boundaryVertexDeletion);
54 
55  m_decimate->Update();
56 
57  if (!m_filterActor) {
58  VTK_CREATE(vtkPolyDataMapper, mapper);
59  mapper->SetInputConnection(m_decimate->GetOutputPort());
60  VtkUtils::vtkInitOnce(m_filterActor);
61  m_filterActor->SetMapper(mapper);
62  addActor(m_filterActor);
63  }
64 
65  applyDisplayEffect();
66 }
