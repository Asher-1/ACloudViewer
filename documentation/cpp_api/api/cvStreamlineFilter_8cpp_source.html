56 void cvStreamlineFilter::apply() {
57  if (!m_dataObject) return;
58 
59  VtkUtils::vtkInitOnce(m_streamline);
60  m_streamline->SetInputData(m_dataObject);
61  m_streamline->SetIntegrationDirection(m_integrationDirection);
62 
63  if (m_source == SourceType::Point) {
64  VTK_CREATE(vtkPointSource, ps);
65  ps->SetCenter(m_sphereCenter);
66  ps->SetNumberOfPoints(m_numberOfPoints);
67  ps->SetRadius(m_radius);
68  ps->Update();
69  m_streamline->SetSourceConnection(ps->GetOutputPort());
70  } else if (m_source == SourceType::Line) {
71  VTK_CREATE(vtkLineSource, ls);
72  ls->SetResolution(30);
73  ls->SetPoint1(m_linePoint1);
74  ls->SetPoint2(m_linePoint2);
75  ls->Update();
76  m_streamline->SetSourceConnection(ls->GetOutputPort());
77  }
78 
79  m_streamline->SetMaximumPropagation(100);
80  m_streamline->SetInitialIntegrationStep(0.1);
81 
82  switch (m_integratorType) {
83  case 0: {
84  VTK_CREATE(vtkRungeKutta2, integ);
85  m_streamline->SetIntegrator(integ);
86  break;
87  }
88 
89  case 1: {
90  VTK_CREATE(vtkRungeKutta4, integ);
91  m_streamline->SetIntegrator(integ);
92  break;
93  }
94 
95  case 2: {
96  VTK_CREATE(vtkRungeKutta45, integ);
97  m_streamline->SetIntegrator(integ);
98  break;
99  }
100 
101  } // switch
102 
103  m_streamline->Update();
104 
105  VTK_CREATE(vtkPolyDataMapper, streamLineMapper);
106 
107  switch (m_displayMode) {
108  case None:
109  streamLineMapper->SetInputConnection(m_streamline->GetOutputPort());
110  break;
111 
112  case Surface: {
113  VTK_CREATE(vtkRuledSurfaceFilter, surfaceFilter);
114  surfaceFilter->SetOffset(0);
115  surfaceFilter->SetOnRatio(.2);
116  surfaceFilter->PassLinesOn();
117  surfaceFilter->SetRuledModeToPointWalk();
118  surfaceFilter->SetDistanceFactor(30);
119 #if 0
120  surfaceFilter->SetOffset(m_ruledSurfaceParams.offset);
121  surfaceFilter->SetOnRatio(m_ruledSurfaceParams.onRatio);
122  surfaceFilter->SetPassLines(m_ruledSurfaceParams.passLines);
123 // surfaceFilter->SetRuledMode(m_ruledSurfaceParams.ruledMode);
124  surfaceFilter->SetDistanceFactor(m_ruledSurfaceParams.distanceFactor);
125  surfaceFilter->SetResolution(m_ruledSurfaceParams.resolution);
126  surfaceFilter->SetOrientLoops(m_ruledSurfaceParams.orientLoops);
127  surfaceFilter->SetCloseSurface(m_ruledSurfaceParams.closeSurface);
128  surfaceFilter->SetInputConnection(m_streamline->GetOutputPort());
129 #endif
130  streamLineMapper->SetInputConnection(
131  surfaceFilter->GetOutputPort());
132  } break;
133 
134  case Tube: {
135  VTK_CREATE(vtkTubeFilter, tubeFilter);
136  tubeFilter->SetNumberOfSides(m_tubeParams.numberOfSides);
137  tubeFilter->SetRadiusFactor(m_tubeParams.radiusFactor);
138  tubeFilter->SetCapping(m_tubeParams.capping);
139  tubeFilter->SetInputConnection(m_streamline->GetOutputPort());
140  streamLineMapper->SetInputConnection(tubeFilter->GetOutputPort());
141  } break;
142  }
143 
144  if (!m_filterActor) {
145  VtkUtils::vtkInitOnce(m_filterActor);
146  m_filterActor->SetMapper(streamLineMapper);
147  addActor(m_filterActor);
148  }
149 
150  m_filterActor->SetMapper(streamLineMapper);
151 
152  applyDisplayEffect();
153 }

170 void cvStreamlineFilter::dataChanged() {

172 
173  vtkPolyData* polydata = vtkPolyData::SafeDownCast(m_dataObject);
174  vtkDataSet* dataset = vtkDataSet::SafeDownCast(m_dataObject);
175  double bounds[6];

178  if (polydata) {

182  dataset->GetBounds(bounds);
183  dataset->GetCenter(center);
184  }
185 
186  double point1[3];
