106 bool cvGenericFilter::initModel() {
107  if (m_dataObject) {
108  m_dataObject->Delete();
109  m_dataObject = nullptr;
110  }
111  assert(m_viewer);
112  m_modelActor = m_viewer->getActorById(m_id);
113  if (!m_modelActor) {
114  return false;
115  }
116 
117  vtkPolyData* polydata =
118  reinterpret_cast<vtkPolyDataMapper*>(m_modelActor->GetMapper())
119  ->GetInput();
120  if (!polydata) {
121  return false;
122  }
123 
124  int npoints = static_cast<int>(polydata->GetNumberOfPoints());
125  if (npoints > MAX_PREVIEW_NUMBER) {
126  m_preview = false;
127  } else {
128  m_preview = true;
129  }
130 
131  if (!m_dataObject) {
132  m_dataObject = vtkPolyData::New();
133  }
134  m_dataObject->DeepCopy(polydata);
135 
136  if (!m_resultData) {
137  m_resultData = vtkPolyData::New();
138  }
139 
140  return true;
141 }

143 ccHObject* cvGenericFilter::getOutput() {
144  vtkDataObject* vtkData = resultData();
145  if (!vtkData) {
146  return nullptr;
147  }
148 
149  vtkPolyData* polydata = vtkPolyData::SafeDownCast(vtkData);
150  if (!polydata) {
151  return nullptr;
152  }
153 
154  ccHObject* result;
155  if (m_meshMode) {
156  result = vtk2cc::ConvertToMesh(polydata);
157  if (!result) {
158  CVLog::Warning(QString("try to save in cloud format"));
159  result = vtk2cc::ConvertToPointCloud(polydata);
160  }
161  } else {
162  result = vtk2cc::ConvertToPointCloud(polydata);
163  }
164 
165  return result;
166 }

254 void cvGenericFilter::applyDisplayEffect() {
255  if (m_modelActor) {
256  switch (m_displayEffect) {
257  case Transparent:
258  m_modelActor->GetProperty()->SetOpacity(0.3);
259  m_modelActor->SetVisibility(1);
260  m_modelActor->GetProperty()->SetRepresentationToSurface();
261  break;
262 
263  case Opaque:
264  m_modelActor->SetVisibility(0);
265  break;
266 
267  case Points:
268  m_modelActor->SetVisibility(1);
269  m_modelActor->GetProperty()->SetRepresentationToPoints();
270  break;
271 
272  case Wireframe:
273  m_modelActor->SetVisibility(1);
274  m_modelActor->GetProperty()->SetRepresentationToWireframe();
275  break;
276  }
277 
278  vtkSmartPointer<vtkLookupTable> lut =
279  createLookupTable(m_scalarMin, m_scalarMax);
280  m_modelActor->GetMapper()->SetLookupTable(lut);
281  update();
282  }

429 void cvGenericFilter::createActorFromData(vtkDataObject* dataObj) {
430  if (!dataObj) return;
431 
432  m_dataObject = dataObj;
433  DataObject* data = DataObject::SafeDownCast(dataObj);
434 
435  if (!data) {
436  CVLog::Error(QString("ansys parsing: null data set."));
437  return;
438  }
439 
440  VTK_CREATE(Mapper, mapper);
441  mapper->SetInputData(data);
442  mapper->Update();
443 
444  // VtkUtils::vtkInitOnce(m_modelActor);
445  m_modelActor = vtkSmartPointer<vtkLODActor>::New();
446  m_modelActor->SetMapper(mapper);
447 
448  static_cast<vtkSmartPointer<vtkLODActor>>(m_modelActor)
449  ->SetNumberOfCloudPoints(int(
450  std::max<vtkIdType>(1, data->GetNumberOfPoints() / 10)));
451  m_modelActor->GetProperty()->SetInterpolationToFlat();
452 
453  addActor(m_modelActor);
454  modelReady();
455
