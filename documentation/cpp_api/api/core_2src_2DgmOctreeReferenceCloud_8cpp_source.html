22 void DgmOctreeReferenceCloud::computeBB() {
23  // empty cloud?!
24  unsigned count = size();
25  if (count) {
26  m_bbMin = m_bbMax = CCVector3(0, 0, 0);
27  return;
28  }
29 
30  // initialize BBox with first point
31  m_bbMin = m_bbMax = *m_set->at(0).point;
32 
33  for (unsigned i = 1; i < count; ++i) {
34  const CCVector3& P = *m_set->at(i).point;
35  // X boundaries
36  if (m_bbMin.x > P.x)
37  m_bbMin.x = P.x;
38  else if (m_bbMax.x < P.x)
39  m_bbMax.x = P.x;
40 
41  // Y boundaries
42  if (m_bbMin.y > P.y)
43  m_bbMin.y = P.y;
44  else if (m_bbMax.y < P.y)
45  m_bbMax.y = P.y;
46 
47  // Z boundaries
48  if (m_bbMin.z > P.z)
49  m_bbMin.z = P.z;
50  else if (m_bbMax.z < P.z)
51  m_bbMax.z = P.z;
52  }
53 
54  m_validBB = true;
55 }
