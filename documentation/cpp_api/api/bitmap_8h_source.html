199 template <typename T1, typename T2>
200 T2 BitmapColorCast(const T1 value) {
201  return std::min(static_cast<T1>(std::numeric_limits<T2>::max()),
202  std::max(static_cast<T1>(std::numeric_limits<T2>::min()),
203  std::round(value)));
204 }
