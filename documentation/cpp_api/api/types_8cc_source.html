43 FeatureKeypoint::FeatureKeypoint(const float x_, const float y_,
44  const float scale, const float orientation)
45  : x(x_), y(y_) {
46  CHECK_GE(scale, 0.0);
47  const float scale_cos_orientation = scale * std::cos(orientation);
48  const float scale_sin_orientation = scale * std::sin(orientation);
49  a11 = scale_cos_orientation;
50  a12 = -scale_sin_orientation;
51  a21 = scale_sin_orientation;
52  a22 = scale_cos_orientation;
53 }

73 void FeatureKeypoint::Rescale(const float scale_x, const float scale_y) {
74  CHECK_GT(scale_x, 0);
75  CHECK_GT(scale_y, 0);
76  x *= scale_x;
77  y *= scale_y;
78  a11 *= scale_x;
79  a12 *= scale_y;
80  a21 *= scale_x;
81  a22 *= scale_y;
82 }
