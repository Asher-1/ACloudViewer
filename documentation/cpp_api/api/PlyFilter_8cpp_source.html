98 CC_FILE_ERROR PlyFilter::saveToFile(ccHObject* entity,
99  QString filename,
100  e_ply_storage_mode storageType) {
101  if (!entity || filename.isEmpty()) {
102  return CC_FERR_BAD_ARGUMENT;
103  }
104 
105  ccGenericPointCloud* vertices = nullptr;
106  ccGenericMesh* mesh = nullptr;
107  if (entity->isKindOf(CV_TYPES::MESH)) {
108  mesh = ccHObjectCaster::ToGenericMesh(entity);
109  vertices = mesh->getAssociatedCloud();
110  } else if (entity->isKindOf(CV_TYPES::POINT_CLOUD)) {
111  vertices = ccHObjectCaster::ToGenericPointCloud(entity);
112  }
113 
114  if (!vertices) {
115  return CC_FERR_BAD_ENTITY_TYPE;
116  }
117 
118 #ifdef _WIN32
119  // Convert QString to UTF-8 encoded std::string for Windows
120  // ply_create expects const char*, not const wchar_t*
121  p_ply ply = ply_create(filename.toUtf8().constData(), storageType,
122  errorCallback, 0, nullptr);
123 #else
124  p_ply ply = ply_create(qPrintable(filename), storageType, errorCallback, 0,
125  nullptr);
126 #endif
127  if (!ply) {
128  return CC_FERR_THIRD_PARTY_LIB_FAILURE;
129  }
130 
131  // Has the cloud been recentered?
132  e_ply_type coordType =
133  vertices->isShifted() || sizeof(PointCoordinateType) > 4
134  ? PLY_DOUBLE
135  : PLY_FLOAT; // we use double coordinates for shifted
136  // vertices (i.e. >1e6)
137 
138  int result = 1;
139  unsigned vertCount = vertices->size();
140 
141  // 3D points (x,y,z)
142  if (ply_add_element(ply, "vertex", vertCount)) {
143  result = ply_add_scalar_property(ply, "x", coordType);
144  result = ply_add_scalar_property(ply, "y", coordType);
145  result = ply_add_scalar_property(ply, "z", coordType);
146  } else
147  result = 0;
148 
149  ccMaterial::CShared material(nullptr);
150  if (mesh && mesh->hasMaterials()) {
151  // look for textures/materials in case there's no color
152  // if (!mesh->hasColors())
153  {
154  unsigned textureCount = 0;
155  const ccMaterialSet* materials = mesh->getMaterialSet();
156  assert(materials);
157  if (materials) {
158  for (size_t i = 0; i < materials->size(); ++i) {
159  // texture?
160  if (!materials->at(i)->getTexture().isNull()) {
161  // save first encountered texture
162  if (!material) material = materials->at(i);
163  ++textureCount;
164  }
165  }
166  }
167 
168  // one texture: we can save it with the PLY file
169  if (textureCount == 1) {
170  if (materials->size() == 1) {
171  // just one texture/material --> we can handle it
172  } else if (materials->size() > 1) {
173  if (mesh->hasColors()) {
174  CVLog::Error(
175  "PLY files can't handle multiple "
176  "materials/textures!\nOnly the RGB field will "
177  "be saved.");
178  } else {
179  if (mesh->isA(CV_TYPES::MESH)) {
180  if (QMessageBox::question(
181  nullptr,
182  "Multiple materials, one texture",
183  "This mesh has only one texture but "
184  "multiple materials. PLY files can "
185  "only handle one texture.\nShall we "
186  "drop the materials (yes) or convert "
187  "all materials and texture to "
188  "per-vertex RGB colors? (no)",
189  QMessageBox::Yes | QMessageBox::No,
190  QMessageBox::Yes) == QMessageBox::No) {
191  // we can forget the texture
192  material = ccMaterial::CShared(nullptr);
193  // try to convert materials to RGB
194  if (!static_cast<ccMesh*>(mesh)
195  ->convertMaterialsToVertexColors()) {
196  CVLog::Error(
197  "Conversion failed! (not enough "
198  "memory?)");
199  }
200  }
201  } else if (mesh->isA(CV_TYPES::SUB_MESH)) {
202  // we can forget the texture
203  material = ccMaterial::CShared(nullptr);
204  CVLog::Warning(
205  "This sub-mesh has one texture and "
206  "multiple materials. As this is a "
207  "sub-mesh, we will ignore them... you "
208  "should convert the parent mesh "
209  "textures/materials to RGB colors first");
210  } else {
211  assert(false);
212  }
213  }
214  } else {
215  assert(false);
216  }
217  } else if (textureCount > 1) // multiple materials
218  {
219  assert(materials->size() != 0);
220  // we can forget the (first) texture (if any)
221  material = ccMaterial::CShared(nullptr);
222 
223  if (mesh->hasColors()) {
224  CVLog::Error(
225  "PLY files can't handle multiple "
226  "materials/textures!\nOnly the RGB field will be "
227  "saved.");
228  } else {
229  if (mesh->isA(CV_TYPES::MESH)) {
230  // we ask the user if he wants to convert them to RGB
231  if (QMessageBox::question(
232  nullptr, "Multiple textures/materials",
233  "PLY files can't handle multiple "
234  "textures/materials!\nDo you want to "
235  "convert them to per-vertex RGB colors?",
236  QMessageBox::Yes | QMessageBox::No,
237  QMessageBox::No) == QMessageBox::Yes) {
238  if (!static_cast<ccMesh*>(mesh)
239  ->convertMaterialsToVertexColors()) {
240  CVLog::Error(
241  "Conversion failed! (not enough "
242  "memory?)");
243  }
244  }
245  } else if (mesh->isA(CV_TYPES::SUB_MESH)) {
246  CVLog::Warning(
247  "This sub-mesh has multiple "
248  "textures/materials. PLY files can't handle "
249  "them.\nAs this is a sub-mesh, we will have to "
250  "ignore them... you should convert the parent "
251  "mesh textures/materials to RGB colors first");
252  } else {
253  assert(false);
254  }
255  }
256  }
257  }
258  // else
259  //{
260  // CVLog::Warning("[PLY] PLY files can't handle materials/textures!
261  // RGB field will be saved instead"); CVLog::Warning("[PLY] Note: you
262  // can convert materials/textures to RGB if necessary (see 'Edit > Mesh'
263  // menu)");
264  // }
265  }
266 
267  bool hasUniqueColor = false;
268  ColorCompType uniqueColor[3]{0, 0, 0};
269  if (material) {
270  // Material without texture?
271  if (!material->hasTexture()) {
272  const ecvColor::Rgbaf& diffuse = material->getDiffuseFront();
273  uniqueColor[0] =
274  static_cast<ColorCompType>(diffuse.r * ecvColor::MAX);
275  uniqueColor[1] =
276  static_cast<ColorCompType>(diffuse.g * ecvColor::MAX);
277  uniqueColor[2] =
278  static_cast<ColorCompType>(diffuse.b * ecvColor::MAX);
279  hasUniqueColor = true;
280  material = ccMaterial::CShared(nullptr); // we can forget it!
281  }
282  }
283 
284  // RGB colors
285  bool hasColors = vertices->hasColors();
286  if (hasColors) {
287  hasColors = true;
288  // if (ply_add_element(ply, "color", vertCount))
289  //{
290  result = ply_add_scalar_property(ply, "red", PLY_UCHAR);
291  result = ply_add_scalar_property(ply, "green", PLY_UCHAR);
292  result = ply_add_scalar_property(ply, "blue", PLY_UCHAR);
293  //}
294  // else result = 0;
295  }
296 
297  // Normals (nx,ny,nz)
298  bool hasNormals = vertices->hasNormals();
299  if (hasNormals) {
300  // if (ply_add_element(ply, "normal", vertCount))
301  //{
302  e_ply_type normType =
303  (sizeof(PointCoordinateType) > 4 ? PLY_DOUBLE : PLY_FLOAT);
304  result = ply_add_scalar_property(ply, "nx", normType);
305  result = ply_add_scalar_property(ply, "ny", normType);
306  result = ply_add_scalar_property(ply, "nz", normType);
307  //}
308  // else result = 0;
309  }
310 
311  // Scalar fields
312  std::vector<ccScalarField*> scalarFields;
313  if (vertices->isA(CV_TYPES::POINT_CLOUD)) {
314  QStringList originalStdPropsNames;
315  if (vertices->hasMetaData("ply.stdPropNames")) {
316  originalStdPropsNames =
317  vertices->getMetaData("ply.stdPropNames").toStringList();
318  }
319 
320  ccPointCloud* ccCloud = static_cast<ccPointCloud*>(vertices);
321  unsigned sfCount = ccCloud->getNumberOfScalarFields();
322  if (sfCount) {
323  scalarFields.resize(sfCount);
324  unsigned unnamedSFCount = 0;
325  for (unsigned i = 0; i < sfCount; ++i) {
326  scalarFields[i] =
327  static_cast<ccScalarField*>(ccCloud->getScalarField(i));
328  QString sfName =
329  QString::fromStdString(scalarFields[i]->getName());
330  QString propName;
331  if (sfName.isEmpty()) {
332  if (unnamedSFCount == 0) {
333  propName = "scalar";
334  } else {
335  propName = QString("scalar_%1").arg(unnamedSFCount);
336  }
337  ++unnamedSFCount;
338  } else {
339  if (originalStdPropsNames.contains(sfName)) {
340  // don't change the name of a field that was already in
341  // the original file
342  propName = sfName;
343  } else {
344  // append the SF name with 'scalar' for easier detection
345  // at loading time
346  propName = QString("scalar_%1").arg(sfName);
347  propName.replace(' ', '_');
348  }
349  }
350 
351  scalarFields[i]->computeMinAndMax();
352  ScalarType maxValue =
353  std::max(std::abs(scalarFields[i]->getMin()),
354  std::abs(scalarFields[i]->getMax()));
355 
356  e_ply_type scalarType =
357  (maxValue < ecvGlobalShiftManager::
358  MaxBoundgBoxDiagonal()
359  ? PLY_FLOAT
360  : PLY_DOUBLE);
361  if (scalarType == PLY_DOUBLE) {
362  CVLog::Warning(
363  QString("[PLY] Scalar field '%1' has large values "
364  "and will be saved as double values "
365  "instead of float values")
366  .arg(QString::fromStdString(
367  scalarFields[i]->getName())));
368  }
369 
370  result = ply_add_scalar_property(ply, qPrintable(propName),
371  scalarType);
372  }
373  }
374  }
375 
376  // Mesh
377  unsigned triNum = 0;
378  if (mesh) {
379  triNum = mesh->size();
380  if (triNum > 0 && ply_add_element(ply, "face", triNum)) {
381  // DGM: don't change the field name (vertex_indices) as Meshlab
382  // only support this one! (grrrrrrrrr)
383  result = ply_add_list_property(ply, "vertex_indices", PLY_UCHAR,
384  PLY_INT);
385 
386  // texture & texture coordinates?
387  if (material) {
388  assert(material->hasTexture() &&
389  mesh->getTexCoordinatesTable());
390  QFileInfo fileInfo(material->getTextureFilename());
391  QString defaultTextureName = fileInfo.fileName();
392  if (fileInfo.suffix().isNull())
393  defaultTextureName += QString(".png");
394  // try to save the texture!
395  QString textureFilePath = QFileInfo(filename).absolutePath() +
396  QString('/') + defaultTextureName;
397  if (!material->getTexture().save(
398  textureFilePath)) // mirrored --> see ccMaterial
399  {
400  CVLog::Warning(
401  QString("[PLY] Failed to save texture in '%1'!")
402  .arg(textureFilePath));
403  material = ccMaterial::CShared(nullptr);
404  } else {
405  // save texture filename as a comment! Note MeshLab only
406  // supports CamelCase as of writing.
407  result = ply_add_comment(
408  ply, qPrintable(QString("TextureFile %1")
409  .arg(defaultTextureName)));
410  // DGM FIXME: is this the right name?
411  result = ply_add_list_property(
412  ply, "texcoord", PLY_UCHAR,
413  PLY_FLOAT); //'texcoord' to mimick Photoscan
414 
415  CVLog::Print(QString("[PLY] Texture file: %1")
416  .arg(textureFilePath));
417  }
418  }
419  }
420  }
421 
422  if (vertices->hasMetaData("ply.comments")) {
423  QStringList comments =
424  vertices->getMetaData("ply.comments").toStringList();
425  for (const QString& comment : comments) {
426  ply_add_comment(ply, qPrintable(comment));
427  }
428  } else {
429  ply_add_comment(ply, qPrintable(FileIO::createdBy()));
430  ply_add_comment(ply, qPrintable(FileIO::createdDateTime()));
431  }
432 
433  ply_add_obj_info(ply, "Generated by CloudViewer!");
434 
435  // try to write header
436  result = ply_write_header(ply);
437  if (!result) {
438  ply_close(ply);
439  return CC_FERR_THIRD_PARTY_LIB_FAILURE;
440  }
441 
442  // save the point cloud (=vertices)
443  for (unsigned i = 0; i < vertCount; ++i) {
444  const CCVector3* P = vertices->getPoint(i);
445  CCVector3d Pglobal = vertices->toGlobal3d<PointCoordinateType>(*P);
446  ply_write(ply, Pglobal.x);
447  ply_write(ply, Pglobal.y);
448  ply_write(ply, Pglobal.z);
449 
450  if (hasColors) {
451  const ecvColor::Rgb& col = vertices->getPointColor(i);
452  ply_write(ply, static_cast<double>(col.r));
453  ply_write(ply, static_cast<double>(col.g));
454  ply_write(ply, static_cast<double>(col.b));
455  } else if (hasUniqueColor) {
456  ply_write(ply, static_cast<double>(uniqueColor[0]));
457  ply_write(ply, static_cast<double>(uniqueColor[1]));
458  ply_write(ply, static_cast<double>(uniqueColor[2]));
459  }
460 
461  if (hasNormals) {
462  const CCVector3& N = vertices->getPointNormal(i);
463  ply_write(ply, static_cast<double>(N.x));
464  ply_write(ply, static_cast<double>(N.y));
465  ply_write(ply, static_cast<double>(N.z));
466  }
467 
468  for (std::vector<ccScalarField*>::const_iterator sf =
469  scalarFields.begin();
470  sf != scalarFields.end(); ++sf) {
471  ply_write(ply, (*sf)->getValue(i));
472  }
473  }
474 
475  // and the mesh structure
476  if (mesh) {
477  mesh->placeIteratorAtBeginning();
478  for (unsigned i = 0; i < triNum; ++i) {
479  const cloudViewer::VerticesIndexes* tsi =
480  mesh->getNextTriangleVertIndexes(); // DGM:
481  // getNextTriangleVertIndexes
482  // is faster for mesh
483  // groups!
484  ply_write(ply, double(3));
485  assert(tsi->i1 < vertCount);
486  assert(tsi->i2 < vertCount);
487  assert(tsi->i3 < vertCount);
488  ply_write(ply, double(tsi->i1));
489  ply_write(ply, double(tsi->i2));
490  ply_write(ply, double(tsi->i3));
491 
492  if (material) // texture coordinates
493  {
494  ply_write(ply, 6.0);
495  TexCoords2D* tx1 = nullptr;
496  TexCoords2D* tx2 = nullptr;
497  TexCoords2D* tx3 = nullptr;
498  mesh->getTriangleTexCoordinates(i, tx1, tx2, tx3);
499  ply_write(ply, tx1 ? tx1->tx : -1.0);
500  ply_write(ply, tx1 ? tx1->ty : -1.0);
501  ply_write(ply, tx2 ? tx2->tx : -1.0);
502  ply_write(ply, tx2 ? tx2->ty : -1.0);
503  ply_write(ply, tx3 ? tx3->tx : -1.0);
504  ply_write(ply, tx3 ? tx3->ty : -1.0);
505  }
506  }
507  }
508 
509  ply_close(ply);
510 
511  return CC_FERR_NO_ERROR;
512 }

538 static int vertex_cb(p_ply_argument argument) {
539  if (s_NotEnoughMemory) {
540  // skip the next pieces of data
541  return 1;
542  }
543  long flags = 0;
544  ccPointCloud* cloud = nullptr;
545  ply_get_argument_user_data(argument, (void**)(&cloud), &flags);
546 
547  double val = ply_get_argument_value(argument);
548 
549  static CCVector3d s_Point(0, 0, 0);
550 
551  // This looks like it should always be true,
552  // but it's false if x is NaN.
553  if (val == val) {
554  s_Point.u[flags & POS_MASK] = val;
555  } else {
556  // warning: corrupted data!
557  s_PointDataCorrupted = true;
558  s_Point.u[flags & POS_MASK] = 0;
559  // return 0;
560  }
561 
562  if (flags & ELEM_EOL) {
563  // first point: check for 'big' coordinates
564  if (s_PointCount == 0) {
565  bool preserveCoordinateShift = true;
566  if (FileIOFilter::HandleGlobalShift(s_Point, s_Pshift,
567  preserveCoordinateShift,
568  s_loadParameters)) {
569  if (preserveCoordinateShift) {
570  cloud->setGlobalShift(s_Pshift);
571  }
572  CVLog::Warning(
573  "[PLYFilter::loadFile] Cloud (vertices) has been "
574  "recentered! Translation: (%.2f ; %.2f ; %.2f)",
575  s_Pshift.x, s_Pshift.y, s_Pshift.z);
576  }
577  }
578 
579  cloud->addPoint((s_Point + s_Pshift).toPC());
580  ++s_PointCount;
581 
582  s_PointDataCorrupted = false;
583  if ((s_PointCount % PROCESS_EVENTS_FREQ) == 0) {
584  QCoreApplication::processEvents();
585  }
586  }
587 
588  return 1;
589 }
