167 void FileIOFilter::Register(Shared filter) {
168  if (!filter) {
169  assert(false);
170  return;
171  }
172 
173  // check for an existing copy of this filter or one with the same ID
174  const QString id = filter->m_filterInfo.id;
175 
176  auto compareFilters = [filter, id](const Shared& filter2) {
177  return (filter == filter2) || (filter2->m_filterInfo.id == id);
178  };
179 
180  if (std::any_of(s_ioFilters.cbegin(), s_ioFilters.cend(), compareFilters)) {
181  CVLog::Warning(QStringLiteral("[FileIOFilter] I/O filter already "
182  "registered with id '%1'")
183  .arg(id));
184 
185  return;
186  }
187 
188  // insert into the list, sorted by priority first, id second
189  auto comparePriorities = [](const Shared& filter1,
190  const Shared& filter2) -> bool {
191  if (filter1->m_filterInfo.priority == filter2->m_filterInfo.priority) {
192  return filter1->m_filterInfo.id < filter2->m_filterInfo.id;
193  }
194 
195  return filter1->m_filterInfo.priority < filter2->m_filterInfo.priority;
196  };
197 
198  auto pos = std::upper_bound(s_ioFilters.begin(), s_ioFilters.end(), filter,
199  comparePriorities);
200 
201  s_ioFilters.insert(pos, filter);
202 }

253 ccHObject* FileIOFilter::LoadFromFile(const QString& filename,
254  LoadParameters& loadParameters,
255  Shared filter,
256  CC_FILE_ERROR& result) {
257  if (!filter) {
258  CVLog::Error(QString("[Load] Internal error (invalid input filter)[%1]")
259  .arg(filename));
260  result = CC_FERR_CONSOLE_ERROR;
261  assert(false);
262  return nullptr;
263  }
264 
265  // check file existence
266  QFileInfo fi(filename);
267  if (!fi.exists()) {
268  CVLog::Error(QString("[Load] File '%1' doesn't exist!").arg(filename));
269  result = CC_FERR_CONSOLE_ERROR;
270  return nullptr;
271  }
272 
273  // load file
274  ccHObject* container = new ccHObject();
275  result = CC_FERR_NO_ERROR;
276 
277  // we start a new 'action' inside the current sessions
278  unsigned sessionCounter = IncreaseSesionCounter();
279  loadParameters.sessionStart = (sessionCounter == 1);
280 
281  try {
282  result = filter->loadFile(filename, *container, loadParameters);
283  } catch (const std::exception& e) {
284  CVLog::Warning(QString("[I/O] CC has caught an exception while loading "
285  "file '%1'")
286  .arg(filename));
287  CVLog::Warning(QString("[I/O] Exception: %1").arg(e.what()));
288  if (container) {
289  container->removeAllChildren();
290  }
291  result = CC_FERR_CONSOLE_ERROR;
292  } catch (...) {
293  CVLog::Warning(QString("[I/O] CC has caught an unhandled exception "
294  "while loading file '%1'")
295  .arg(filename));
296  if (container) {
297  container->removeAllChildren();
298  }
299  result = CC_FERR_CONSOLE_ERROR;
300  }
301 
302  if (result == CC_FERR_NO_ERROR) {
303  CVLog::Print(
304  QString("[I/O] File '%1' loaded successfully").arg(filename));
305  } else {
306  DisplayErrorMessage(result, "loading", fi.baseName());
307  }
308 
309  unsigned childCount = container->getChildrenNumber();
310  if (childCount != 0) {
311  // we set the main container name as the full filename (with path)
312  container->setName(
313  QString("%1 (%2)").arg(fi.fileName(), fi.absolutePath()));
314  for (unsigned i = 0; i < childCount; ++i) {
315  ccHObject* child = container->getChild(i);
316  child->setBaseName(fi.baseName());
317  child->setFullPath(filename);
318  QString newName = child->getName();
319  if (newName.startsWith("unnamed")) {
320  // we automatically replace occurrences of 'unnamed' in entities
321  // names by the base filename (no path, no extension)
322  newName.replace(QString("unnamed"), fi.baseName());
323  child->setName(newName);
324  }
325  }
326  } else {
327  delete container;
328  container = nullptr;
329  }
330 
331  return container;
332 }

334 ccHObject* FileIOFilter::LoadFromFile(
335  const QString& filename,
336  LoadParameters& loadParameters,
337  CC_FILE_ERROR& result,
338  const QString& fileFilter /*=QString()*/) {
339  Shared filter(nullptr);
340 
341  // if the right filter is specified by the caller
342  if (!fileFilter.isEmpty()) {
343  filter = GetFilter(fileFilter, true);
344  if (!filter) {
345  CVLog::Error(QString("[Load] Internal error: no I/O filter "
346  "corresponds to filter '%1'")
347  .arg(fileFilter));
348  result = CC_FERR_CONSOLE_ERROR;
349  return nullptr;
350  }
351  } else // we need to guess the I/O filter based on the file format
352  {
353  // look for file extension (we trust Qt on this task)
354  QString extension = QFileInfo(filename).suffix();
355  if (extension.isEmpty()) {
356  CVLog::Error("[Load] Can't guess file format: no file extension");
357  result = CC_FERR_CONSOLE_ERROR;
358  return nullptr;
359  }
360 
361  // convert extension to file format
362  filter = FindBestFilterForExtension(extension);
363 
364  // unknown extension?
365  if (!filter) {
366  CVLog::Error(QString("[Load] Can't guess file format: unhandled "
367  "file extension '%1'")
368  .arg(extension));
369  result = CC_FERR_CONSOLE_ERROR;
370  return nullptr;
371  }
372  }
373 
374  return LoadFromFile(filename, loadParameters, filter, result);
375 }

377 CC_FILE_ERROR FileIOFilter::SaveToFile(ccHObject* entities,
378  const QString& filename,
379  const SaveParameters& parameters,
380  Shared filter) {
381  if (!entities || filename.isEmpty() || !filter) return CC_FERR_BAD_ARGUMENT;
382 
383  // if the file name has no extension, we had a default one!
384  QString completeFileName(filename);
385  if (QFileInfo(filename).suffix().isEmpty())
386  completeFileName += QString(".%1").arg(filter->getDefaultExtension());
387 
388  CC_FILE_ERROR result = CC_FERR_NO_ERROR;
389  try {
390  result = filter->saveToFile(entities, completeFileName, parameters);
391  } catch (...) {
392  CVLog::Warning(QString("[I/O] CC has caught an unhandled exception "
393  "while saving file '%1'")
394  .arg(filename));
395  result = CC_FERR_CONSOLE_ERROR;
396  }
397 
398  if (result == CC_FERR_NO_ERROR) {
399  CVLog::Print(
400  QString("[I/O] File '%1' saved successfully").arg(filename));
401  } else {
402  DisplayErrorMessage(result, "saving", filename);
403  }
404 
405  return result;

425 void FileIOFilter::DisplayErrorMessage(CC_FILE_ERROR err,
426  const QString& action,
427  const QString& filename) {
428  QString errorStr;
429 
430  bool warning = false;
431  switch (err) {

434  case CC_FERR_BAD_ARGUMENT:
435  errorStr = "bad argument (internal)";
436  break;
437  case CC_FERR_UNKNOWN_FILE:
438  errorStr = "unknown file";
439  break;
440  case CC_FERR_WRONG_FILE_TYPE:
441  errorStr = "wrong file type (check header)";
442  break;
443  case CC_FERR_WRITING:
444  errorStr = "writing error (disk full/no access right?)";
445  break;
446  case CC_FERR_READING:
447  errorStr = "reading error (no access right?)";
448  break;
449  case CC_FERR_NO_SAVE:
450  errorStr = "nothing to save";
451  break;
452  case CC_FERR_NO_LOAD:
453  errorStr = "nothing to load";
454  break;
455  case CC_FERR_BAD_ENTITY_TYPE:
456  errorStr = "incompatible entity/file types";
457  break;
458  case CC_FERR_CANCELED_BY_USER:
459  errorStr = "process canceled by user";
460  warning = true;
461  break;
462  case CC_FERR_NOT_ENOUGH_MEMORY:
463  errorStr = "not enough memory";
464  break;
465  case CC_FERR_MALFORMED_FILE:
466  errorStr = "malformed file";
467  break;
468  case CC_FERR_BROKEN_DEPENDENCY_ERROR:
469  errorStr = "dependent entities missing (see Console)";
470  break;
471  case CC_FERR_FILE_WAS_WRITTEN_BY_UNKNOWN_PLUGIN:
472  errorStr =
473  "the file was written by a plugin but none of the loaded "
474  "plugins can deserialize it";
475  break;
476  case CC_FERR_THIRD_PARTY_LIB_FAILURE:
477  errorStr =
478  "the third-party library in charge of saving/loading the "
479  "file has failed to perform the operation";
480  break;
481  case CC_FERR_THIRD_PARTY_LIB_EXCEPTION:
482  errorStr =
483  "the third-party library in charge of saving/loading the "
484  "file has thrown an exception";
485  break;
486  case CC_FERR_NOT_IMPLEMENTED:
487  errorStr = "this function is not implemented yet!";
488  break;
489  case CC_FERR_CONSOLE_ERROR:
490  errorStr = "see console";
491  break;
492  default:
493  return; // no message will be displayed!
494  }
495 
496  QString outputString =
497  QString("An error occurred while %1 '%2': ").arg(action, filename) +
498  errorStr;
499  if (warning)
500  CVLog::Warning(outputString);
501  else
502  CVLog::Error(outputString);

509 bool FileIOFilter::HandleGlobalShift(
510  const CCVector3d& P,
511  CCVector3d& Pshift,
512  bool& preserveCoordinateShift,
513  LoadParameters& loadParameters,
514  bool useInputCoordinatesShiftIfPossible /*=false*/) {

516  *loadParameters.coordinatesShiftEnabled &&
517  loadParameters.coordinatesShift);
518  if (shiftAlreadyEnabled) {
519  Pshift = *loadParameters.coordinatesShift;
520  preserveCoordinateShift = loadParameters.preserveShiftOnSave;
521  }
522 
523  bool applyAll = false;

525  ecvGlobalShiftManager::Handle(

529  // we save coordinates shift information

541  *loadParameters.coordinatesShift = Pshift;

546  }
