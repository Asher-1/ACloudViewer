91 CC_FILE_ERROR AsciiFilter::saveToFile(ccHObject* entity,
92  const QString& filename,
93  const SaveParameters& parameters) {
94  assert(entity && !filename.isEmpty());
95 
96  AsciiSaveDlg saveDialog(parameters.parentWidget);
97 
98  saveDialog.setCoordsPrecision(s_outputCoordPrecision);
99  saveDialog.setSfPrecision(s_outputSFPrecision);
100  saveDialog.setSeparatorIndex(s_outputSeparatorIndex);
101  saveDialog.enableSwapColorAndSF(s_saveSFBeforeColor);
102  saveDialog.enableSaveColumnsNamesHeader(s_saveColumnsNamesHeader);
103  saveDialog.enableSavePointCountHeader(s_savePointCountHeader);
104 
105  // if the dialog shouldn't be shown, we'll simply take the default values!
106  static bool s_showDialog = true;
107  if (parameters.alwaysDisplaySaveDialog && s_showDialog) {
108  if (!saveDialog.exec()) {
109  return CC_FERR_CANCELED_BY_USER;
110  }
111  s_outputCoordPrecision = saveDialog.coordsPrecision();
112  s_outputSFPrecision = saveDialog.sfPrecision();
113  s_outputSeparatorIndex = saveDialog.getSeparatorIndex();
114  s_saveSFBeforeColor = saveDialog.swapColorAndSF();
115  s_saveColumnsNamesHeader = saveDialog.saveColumnsNamesHeader();
116  s_savePointCountHeader = saveDialog.savePointCountHeader();
117  }
118 
119  if (!entity->isKindOf(CV_TYPES::POINT_CLOUD)) {
120  if (entity->isA(CV_TYPES::HIERARCHY_OBJECT)) // multiple clouds?
121  {
122  QFileInfo fi(filename);
123  QString extension = fi.suffix();
124  QString baseName = fi.completeBaseName();
125  QString path = fi.path();
126 
127  unsigned count = entity->getChildrenNumber();
128  // we count the number of clouds first
129  unsigned cloudCount = 0;
130  {
131  for (unsigned i = 0; i < count; ++i) {
132  ccHObject* child = entity->getChild(i);
133  if (child->isKindOf(CV_TYPES::POINT_CLOUD)) ++cloudCount;
134  }
135  }
136 
137  // we can now create the corresponding file(s)
138  if (cloudCount > 1) {
139  unsigned counter = 0;
140 
141  bool autoShow = s_showDialog;
142  s_showDialog = false;
143 
144  for (unsigned i = 0; i < count; ++i) {
145  ccHObject* child = entity->getChild(i);
146  if (child->isKindOf(CV_TYPES::POINT_CLOUD)) {
147  QString subFilename = path + QString("/");
148  subFilename += QString(baseName).replace(
149  "cloudname", child->getName(),
150  Qt::CaseInsensitive);
151  counter++;
152  assert(counter <= cloudCount);
153  subFilename += QString("_%1").arg(cloudCount - counter,
154  6, 10, QChar('0'));
155  if (!extension.isEmpty())
156  subFilename += QString(".") + extension;
157 
158  CC_FILE_ERROR result = saveToFile(
159  entity->getChild(i), subFilename, parameters);
160  if (result != CC_FERR_NO_ERROR) {
161  return result;
162  } else {
163  CVLog::Print(QString("[ASCII] Cloud '%1' has been "
164  "saved in: %2")
165  .arg(child->getName(),
166  subFilename));
167  }
168  } else {
169  CVLog::Warning(QString("[ASCII] Entity '%1' can't be "
170  "saved this way!")
171  .arg(child->getName()));
172  }
173  }
174 
175  // restore previous state
176  s_showDialog = autoShow;
177 
178  return CC_FERR_NO_ERROR;
179  }
180  } else {
181  return CC_FERR_BAD_ARGUMENT;
182  }
183  }
184 
185  QFile file(filename);
186  if (!file.open(QFile::WriteOnly | QFile::Truncate)) return CC_FERR_WRITING;
187  QTextStream stream(&file);
188 
189  ccGenericPointCloud* cloud = ccHObjectCaster::ToGenericPointCloud(entity);
190 
191  unsigned numberOfPoints = cloud->size();
192  bool writeColors = cloud->hasColors();
193  bool writeNorms = cloud->hasNormals();
194  std::vector<ccScalarField*> theScalarFields;
195  if (cloud->isKindOf(CV_TYPES::POINT_CLOUD)) {
196  ccPointCloud* ccCloud = static_cast<ccPointCloud*>(cloud);
197  for (unsigned i = 0; i < ccCloud->getNumberOfScalarFields(); ++i)
198  theScalarFields.push_back(
199  static_cast<ccScalarField*>(ccCloud->getScalarField(i)));
200  }
201  bool writeSF = (!theScalarFields.empty());
202 
203  // progress dialog
204  QScopedPointer<ecvProgressDialog> pDlg(nullptr);
205  if (parameters.parentWidget) {
206  pDlg.reset(new ecvProgressDialog(true, parameters.parentWidget));
207  pDlg->setMethodTitle(
208  QObject::tr("Saving cloud [%1]").arg(cloud->getName()));
209  pDlg->setInfo(QObject::tr("Number of points: %1").arg(numberOfPoints));
210  pDlg->start();
211  }
212  cloudViewer::NormalizedProgress nprogress(pDlg.data(), numberOfPoints);
213 
214  // non static parameters
215  int normalPrecision = 2 + sizeof(PointCoordinateType);
216  QChar separator(saveDialog.getSeparator());
217  bool saveFloatColors = saveDialog.saveFloatColors();
218  bool saveAlphaChannel = saveDialog.saveAlphaChannel();
219 
220  if (s_saveColumnsNamesHeader) {
221  QString header("//");
222  header.append(AsciiHeaderColumns::X());
223  header.append(separator);
224  header.append(AsciiHeaderColumns::Y());
225  header.append(separator);
226  header.append(AsciiHeaderColumns::Z());
227 
228  if (writeColors && !s_saveSFBeforeColor) {
229  header.append(separator);
230  header.append(saveFloatColors ? AsciiHeaderColumns::Rf()
231  : AsciiHeaderColumns::R());
232  header.append(separator);
233  header.append(saveFloatColors ? AsciiHeaderColumns::Gf()
234  : AsciiHeaderColumns::G());
235  header.append(separator);
236  header.append(saveFloatColors ? AsciiHeaderColumns::Bf()
237  : AsciiHeaderColumns::B());
238  if (saveAlphaChannel) {
239  header.append(separator);
240  header.append(saveFloatColors ? AsciiHeaderColumns::Af()
241  : AsciiHeaderColumns::A());
242  }
243  }
244 
245  if (writeSF) {
246  // add each associated SF name
247  for (std::vector<ccScalarField*>::const_iterator it =
248  theScalarFields.begin();
249  it != theScalarFields.end(); ++it) {
250  QString sfName((*it)->getName());
251  sfName.replace(separator, '_');
252  header.append(separator);
253  header.append(sfName);
254  }
255  }
256 
257  if (writeColors && s_saveSFBeforeColor) {
258  header.append(separator);
259  header.append(saveFloatColors ? AsciiHeaderColumns::Rf()
260  : AsciiHeaderColumns::R());
261  header.append(separator);
262  header.append(saveFloatColors ? AsciiHeaderColumns::Gf()
263  : AsciiHeaderColumns::G());
264  header.append(separator);
265  header.append(saveFloatColors ? AsciiHeaderColumns::Bf()
266  : AsciiHeaderColumns::B());
267  if (saveAlphaChannel) {
268  header.append(separator);
269  header.append(saveFloatColors ? AsciiHeaderColumns::Af()
270  : AsciiHeaderColumns::A());
271  }
272  }
273 
274  if (writeNorms) {
275  header.append(separator);
276  header.append(AsciiHeaderColumns::Nx());
277  header.append(separator);
278  header.append(AsciiHeaderColumns::Ny());
279  header.append(separator);
280  header.append(AsciiHeaderColumns::Nz());
281  }
282 
283  stream << header << "\n";
284  }
285 
286  if (s_savePointCountHeader) {
287  stream << QString::number(numberOfPoints) << "\n";
288  }
289 
290  CC_FILE_ERROR result = CC_FERR_NO_ERROR;
291  for (unsigned i = 0; i < numberOfPoints; ++i) {
292  // line for the current point
293  QString line;
294 
295  // write current point coordinates
296  const CCVector3* P = cloud->getPoint(i);
297  CCVector3d Pglobal = cloud->toGlobal3d<PointCoordinateType>(*P);
298  line.append(QString::number(Pglobal.x, 'f', s_outputCoordPrecision));
299  line.append(separator);
300  line.append(QString::number(Pglobal.y, 'f', s_outputCoordPrecision));
301  line.append(separator);
302  line.append(QString::number(Pglobal.z, 'f', s_outputCoordPrecision));
303 
304  QString colorLine;
305  if (writeColors) {
306  // add rgb color
307  const ecvColor::Rgb& col = cloud->getPointColor(i);
308  if (saveFloatColors) {
309  colorLine.append(separator);
310  colorLine.append(QString::number(static_cast<double>(col.r) /
311  ecvColor::MAX));
312  colorLine.append(separator);
313  colorLine.append(QString::number(static_cast<double>(col.g) /
314  ecvColor::MAX));
315  colorLine.append(separator);
316  colorLine.append(QString::number(static_cast<double>(col.b) /
317  ecvColor::MAX));
318  if (saveAlphaChannel) {
319  colorLine.append(separator);
320  colorLine.append(
321  QString::number(static_cast<double>(ecvColor::MAX) /
322  ecvColor::MAX));
323  }
324  } else {
325  colorLine.append(separator);
326  colorLine.append(QString::number(col.r));
327  colorLine.append(separator);
328  colorLine.append(QString::number(col.g));
329  colorLine.append(separator);
330  colorLine.append(QString::number(col.b));
331  if (saveAlphaChannel) {
332  colorLine.append(separator);
333  colorLine.append(QString::number(ecvColor::MAX));
334  }
335  }
336 
337  if (!s_saveSFBeforeColor) {
338  line.append(colorLine);
339  }
340  }
341 
342  if (writeSF) {
343  // add each associated SF values
344  for (std::vector<ccScalarField*>::const_iterator it =
345  theScalarFields.begin();
346  it != theScalarFields.end(); ++it) {
347  line.append(separator);
348  double sfVal = (*it)->getGlobalShift() + (*it)->getValue(i);
349  line.append(QString::number(sfVal, 'f', s_outputSFPrecision));
350  }
351  }
352 
353  if (writeColors && s_saveSFBeforeColor) line.append(colorLine);
354 
355  if (writeNorms) {
356  // add normal vector
357  const CCVector3& N = cloud->getPointNormal(i);
358  line.append(separator);
359  line.append(QString::number(N.x, 'f', normalPrecision));
360  line.append(separator);
361  line.append(QString::number(N.y, 'f', normalPrecision));
362  line.append(separator);
363  line.append(QString::number(N.z, 'f', normalPrecision));
364  }
365 
366  stream << line << "\n";
367 
368  if (pDlg && !nprogress.oneStep()) {
369  result = CC_FERR_CANCELED_BY_USER;
370  break;
371  }
372  }
373 
374  return result;
375 }

528  unsigned numberOfPoints,
529  int& maxIndex,
530  unsigned step = 1) {

532  if (!cloud || !cloud->reserveThePointsTable(numberOfPoints)) {
533  delete cloud;
534  return cloudAttributesDescriptor();
535  }
536 
537  if (step == 1)
538  cloud->setName("unnamed - Cloud");
539  else
540  cloud->setName(QString("unnamed - Cloud (part %1)").arg(step));
541 
542  cloudAttributesDescriptor cloudDesc;
543  cloudDesc.cloud = cloud;
544 
545  int seqSize = static_cast<int>(openSequence.size());
546  for (int i = 0; i < seqSize; ++i) {
547  switch (openSequence[i].type) {
548  case ASCII_OPEN_DLG_None:
549  break;
550  case ASCII_OPEN_DLG_X:
551  cloudDesc.xCoordIndex = i;
552  break;
553  case ASCII_OPEN_DLG_Y:
554  cloudDesc.yCoordIndex = i;
555  break;
556  case ASCII_OPEN_DLG_Z:
557  cloudDesc.zCoordIndex = i;
558  break;
559  case ASCII_OPEN_DLG_NX:
560  if (cloud->reserveTheNormsTable()) {
561  cloudDesc.xNormIndex = i;
562  cloudDesc.hasNorms = true;
563  cloud->showNormals(true);
564  } else {
565  CVLog::Warning(
566  "Failed to allocate memory for normals! (skipped)");
567  }
568  break;
569  case ASCII_OPEN_DLG_NY:
570  if (cloud->reserveTheNormsTable()) {
571  cloudDesc.yNormIndex = i;
572  cloudDesc.hasNorms = true;
573  cloud->showNormals(true);
574  } else {
575  CVLog::Warning(
576  "Failed to allocate memory for normals! (skipped)");
577  }
578  break;
579  case ASCII_OPEN_DLG_NZ:
580  if (cloud->reserveTheNormsTable()) {
581  cloudDesc.zNormIndex = i;
582  cloudDesc.hasNorms = true;
583  cloud->showNormals(true);
584  } else {
585  CVLog::Warning(
586  "Failed to allocate memory for normals! (skipped)");
587  }
588  break;
589  case ASCII_OPEN_DLG_Scalar: {
590  int sfIndex = cloud->getNumberOfScalarFields() + 1;
591  QString sfName = openSequence[i].header;
592  if (sfName.isEmpty()) {
593  sfName = "Scalar field";
594  if (sfIndex > 1) sfName += QString(" #%1").arg(sfIndex);
595  // if (isPositive)
596  // sfName += QString(" (positive)");
597  } else {
598  sfName.replace('_', ' ');
599  }
600 
601  ccScalarField* sf = new ccScalarField(qPrintable(sfName));
602  int sfIdx = cloud->addScalarField(sf);
603  if (sfIdx >= 0) {
604  cloudDesc.scalarIndexes.push_back(i);
605  cloudDesc.scalarFields.push_back(sf);
606  } else {
607  CVLog::Warning(
608  "Failed to add scalar field #%i to cloud #%i! "
609  "(skipped)",
610  sfIndex);
611  sf->release();
612  sf = nullptr;
613  }
614  } break;
615  case ASCII_OPEN_DLG_Rf:
616  cloudDesc.hasFloatRGBColors[0] = true;
617  case ASCII_OPEN_DLG_R:
618  if (cloud->reserveTheRGBTable()) {
619  cloudDesc.redIndex = i;
620  cloudDesc.hasRGBColors = true;
621  cloud->showColors(true);
622  } else {
623  CVLog::Warning(
624  "Failed to allocate memory for colors! (skipped)");
625  }
626  break;
627  case ASCII_OPEN_DLG_Gf:
628  cloudDesc.hasFloatRGBColors[1] = true;
629  case ASCII_OPEN_DLG_G:
630  if (cloud->reserveTheRGBTable()) {
631  cloudDesc.greenIndex = i;
632  cloudDesc.hasRGBColors = true;
633  cloud->showColors(true);
634  } else {
635  CVLog::Warning(
636  "Failed to allocate memory for colors! (skipped)");
637  }
638  break;
639  case ASCII_OPEN_DLG_Bf:
640  cloudDesc.hasFloatRGBColors[2] = true;
641  case ASCII_OPEN_DLG_B:
642  if (cloud->reserveTheRGBTable()) {
643  cloudDesc.blueIndex = i;
644  cloudDesc.hasRGBColors = true;
645  cloud->showColors(true);
646  } else {
647  CVLog::Warning(
648  "Failed to allocate memory for colors! (skipped)");
649  }
650  break;
651  case ASCII_OPEN_DLG_Af:
652  cloudDesc.hasFloatRGBColors[3] = true;
653  case ASCII_OPEN_DLG_A:
654  if (cloud->reserveTheRGBTable()) {
655  cloudDesc.alphaIndex = i;
656  cloudDesc.hasRGBColors = true;
657  cloud->showColors(true);
658  } else {
659  CVLog::Warning(
660  "Failed to allocate memory for colors! (skipped)");
661  }
662  break;
663  case ASCII_OPEN_DLG_RGB32i:
664  case ASCII_OPEN_DLG_RGB32f:
665  if (cloud->reserveTheRGBTable()) {
666  if (openSequence[i].type == ASCII_OPEN_DLG_RGB32i)
667  cloudDesc.iRgbaIndex = i;
668  else
669  cloudDesc.fRgbaIndex = i;
670  cloudDesc.hasRGBColors = true;
671  cloud->showColors(true);
672  } else {
673  CVLog::Warning(
674  "Failed to allocate memory for colors! (skipped)");
675  }
676  break;
677  case ASCII_OPEN_DLG_Grey:
678  if (cloud->reserveTheRGBTable()) {
679  cloudDesc.greyIndex = i;
680  cloud->showColors(true);
681  } else {
682  CVLog::Warning(
683  "Failed to allocate memory for colors! (skipped)");
684  }
685  break;
686  case ASCII_OPEN_DLG_Label:
687  assert(cloudDesc.labelIndex < 0); // There Can Be Only One
688  cloudDesc.labelIndex = i;
689  break;

691  // unhandled case?
692  assert(false);
693  break;
694  }

802  1.0, static_cast<double>(fileSize) / averageLineSize -
803  static_cast<double>(skipLines));

806  // default

809  std::max(static_cast<double>(cloudChunkPos +
810  cloudChunkSize) +
811  1.0,
812  static_cast<double>(pointsRead) * 1.02);
813  }

856  maxCloudSize, approximateNumberOfLines - cloudChunkPos);

1058  cloudDesc.scalarFields[j]->resizeSafe(cloudDesc.cloud->size(),
1059  true, NAN_VALUE);
1060  cloudDesc.scalarFields[j]->computeMinAndMax();
1061  }
1062  cloudDesc.cloud->setCurrentDisplayedScalarField(0);
1063  cloudDesc.cloud->showSF(true);
1064  }
1065 
1066  // position the labels

1080  }
1081  }
1082 
1083  container.addChild(cloudDesc.cloud);
1084  }
1085
