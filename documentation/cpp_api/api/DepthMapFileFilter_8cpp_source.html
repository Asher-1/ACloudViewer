44 CC_FILE_ERROR DepthMapFileFilter::saveToFile(ccHObject* entity,
45  const QString& filename,
46  const SaveParameters& parameters) {
47  if (!entity || filename.isEmpty()) return CC_FERR_BAD_ARGUMENT;
48 
49  ccHObject::Container sensors;
50  if (entity->isKindOf(CV_TYPES::GBL_SENSOR))
51  sensors.push_back(entity);
52  else
53  entity->filterChildren(sensors, true, CV_TYPES::GBL_SENSOR);
54 
55  if (sensors.empty()) {
56  CVLog::Error("No sensor in input selection!");
57  return CC_FERR_BAD_ENTITY_TYPE;
58  }
59 
60  // multiple filenames handling
61  QFileInfo fi(filename);
62  QString baseName = fi.baseName();
63  QString extension = fi.suffix();
64 
65  CC_FILE_ERROR result = CC_FERR_NO_ERROR;
66 
67  size_t sensorCount = sensors.size();
68  for (size_t i = 0; i < sensorCount && result == CC_FERR_NO_ERROR; ++i) {
69  // more than one sensor? we must generate auto filename
70  QString sensorFilename = (sensorCount > 1 ? QString("%1_%2.%3")
71  .arg(baseName)
72  .arg(i)
73  .arg(extension)
74  : filename);
75 
76  ccGBLSensor* sensor = static_cast<ccGBLSensor*>(sensors[i]);
77  if (sensor) {
78  result = saveToFile(sensorFilename, sensor);
79  // we stop at the first severe error
80  if (result != CC_FERR_NO_SAVE && result != CC_FERR_NO_ERROR) break;
81  }
82  }
83 
84  return result;
85 }

87 CC_FILE_ERROR DepthMapFileFilter::saveToFile(const QString& filename,
88  ccGBLSensor* sensor) {
89  assert(sensor);
90  if (!sensor) {
91  return CC_FERR_BAD_ARGUMENT;
92  }
93 
94  // the depth map associated to this sensor
95  const ccDepthBuffer& db = sensor->getDepthBuffer();
96  if (db.zBuff.empty()) {
97  CVLog::Warning(QString("[DepthMap] sensor '%1' has no associated depth "
98  "map (you must compute it first)")
99  .arg(sensor->getName()));
100  return CC_FERR_NO_SAVE; // this is not a severe error (the process can
101  // go on)
102  }
103 
104  ccGenericPointCloud* cloud =
105  ccHObjectCaster::ToGenericPointCloud(sensor->getParent());
106  if (!cloud) {
107  CVLog::Warning(QString("[DepthMap] sensor '%1' is not associated to a "
108  "point cloud!")
109  .arg(sensor->getName()));
110  // this is not a severe error (the process can go on)
111  }
112 
113  if (CheckForSpecialChars(filename)) {
114  CVLog::Warning(QString(
115  "[DepthMap] Output filename contains special characters. It "
116  "might be scrambled or rejected by the I/O filter..."));
117  }
118 
119  // opening file
120  FILE* fp = fopen(qPrintable(filename), "wt");
121  if (!fp) {
122  CVLog::Error(QString("[DepthMap] Can't open file '%1' for writing!")
123  .arg(filename));
124  return CC_FERR_WRITING;
125  }
126 
127  fprintf(fp, "// SENSOR DEPTH MAP\n");
128  fprintf(fp, "// %s\n", qPrintable(FileIO::createdBy()));
129  fprintf(fp, "// %s\n", qPrintable(FileIO::createdDateTime()));
130  fprintf(fp, "// Associated cloud: %s\n",
131  qPrintable(cloud ? cloud->getName() : "none"));
132  fprintf(fp, "// Pitch = %f [ %f : %f ]\n", sensor->getPitchStep(),
133  sensor->getMinPitch(), sensor->getMaxPitch());
134  fprintf(fp, "// Yaw = %f [ %f : %f ]\n", sensor->getYawStep(),
135  sensor->getMinYaw(), sensor->getMaxYaw());
136  fprintf(fp, "// Range = %f\n", sensor->getSensorRange());
137  fprintf(fp, "// L = %i\n", db.width);
138  fprintf(fp, "// H = %i\n", db.height);
139  fprintf(fp, "/////////////////////////\n");
140 
141  // an array of projected normals (same size a depth map)
142  ccGBLSensor::NormalGrid* theNorms = nullptr;
143  // an array of projected colors (same size a depth map)
144  ccGBLSensor::ColorGrid* theColors = nullptr;
145 
146  // if the sensor is associated to a "ccPointCloud", we may also extract
147  // normals and color!
148  if (cloud && cloud->isA(CV_TYPES::POINT_CLOUD)) {
149  ccPointCloud* pc = static_cast<ccPointCloud*>(cloud);
150 
151  unsigned nbPoints = cloud->size();
152  if (nbPoints == 0) {
153  CVLog::Warning(QString("[DepthMap] sensor '%1' is associated to an "
154  "empty cloud?!")
155  .arg(sensor->getName()));
156  // this is not a severe error (the process can go on)
157  } else {
158  // if possible, we create the array of projected normals
159  if (pc->hasNormals()) {
160  std::vector<CCVector3> decodedNorms;
161  try {
162  decodedNorms.reserve(nbPoints);
163  for (unsigned i = 0; i < nbPoints; ++i) {
164  decodedNorms.emplace_back(pc->getPointNormal(i));
165  }
166 
167  theNorms = sensor->projectNormals(pc, decodedNorms);
168  } catch (const std::bad_alloc&) {
169  CVLog::Warning(QString("[DepthMap] not enough memory to "
170  "load normals on sensor '%1'!")
171  .arg(sensor->getName()));
172  }
173  }
174 
175  // if possible, we create the array of projected colors
176  if (pc->hasColors()) {
177  try {
178  std::vector<ecvColor::Rgb> rgbColors;
179  rgbColors.reserve(nbPoints);
180  for (unsigned i = 0; i < nbPoints; ++i) {
181  // conversion from ColorCompType[3] to unsigned char[3]
182  rgbColors.emplace_back(pc->getPointColor(i));
183  }
184  theColors = sensor->projectColors(pc, rgbColors);
185  } catch (const std::bad_alloc&) {
186  CVLog::Warning(QString("[DepthMap] not enough memory to "
187  "load colors on sensor '%1'!")
188  .arg(sensor->getName()));
189  }
190  }
191  }
192  }
193 
194  const PointCoordinateType* _zBuff = db.zBuff.data();
195  unsigned index = 0;
196  for (unsigned k = 0; k < db.height; ++k) {
197  for (unsigned j = 0; j < db.width; ++j, ++_zBuff, ++index) {
198  // grid index and depth
199  fprintf(fp, "%u %u %.12f", j, k, *_zBuff);
200 
201  // color
202  if (theColors) {
203  const ecvColor::Rgb& C = theColors->at(index);
204  fprintf(fp, " %i %i %i", C.r, C.g, C.b);
205  }
206 
207  // normal
208  if (theNorms) {
209  const CCVector3& N = theNorms->at(index);
210  fprintf(fp, " %f %f %f", N.x, N.y, N.z);
211  }
212 
213  fprintf(fp, "\n");
214  }
215  }
216 
217  fclose(fp);
218  fp = nullptr;
219 
220  if (theNorms) {
221  delete theNorms;
222  theNorms = nullptr;
223  }
224  if (theColors) {
225  delete theColors;
226  theColors = nullptr;
227  }
228 
229  return CC_FERR_NO_ERROR;
230 }
