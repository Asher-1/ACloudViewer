72  inline static Eigen::Vector3d ToEigen(const Type col[3]) {
73  if (is_same_type<Type, float>()) {
74  return Eigen::Vector3d(col[0], col[1], col[2]);
75  } else if (is_same_type<Type, ColorCompType>()) {
76  return Eigen::Vector3d(col[0] / 255.0, col[1] / 255.0,
77  col[2] / 255.0);
78  } else {
79  assert(false);
80  return Eigen::Vector3d(col[0], col[1], col[2]);
81  }
82  }
83  inline static Eigen::Vector3d ToEigen(const RgbTpl<Type>& t) {
84  return ToEigen(t.rgb);
85  }
86  constexpr inline static RgbTpl FromEigen(const Eigen::Vector3d& t) {
87  Type newCol[3];
88  if (is_same_type<Type, float>()) {
89  newCol[0] = static_cast<Type>(std::min(1.0, std::max(0.0, t(0))));
90  newCol[1] = static_cast<Type>(std::min(1.0, std::max(0.0, t(1))));
91  newCol[2] = static_cast<Type>(std::min(1.0, std::max(0.0, t(2))));
92  } else {
93  if (t(0) > 1 + EPSILON_VALUE || t(1) > 1 + EPSILON_VALUE ||
94  t(2) > 1 + EPSILON_VALUE) {
95  cloudViewer::utility::LogWarning(
96  "[ecvColor] Find invalid color: ");
97  std::cout << t << std::endl;
98  }
99 
100  newCol[0] = static_cast<Type>(
101  std::min(255.0, std::max(0.0, t(0) * MAX)));
102  newCol[1] = static_cast<Type>(
103  std::min(255.0, std::max(0.0, t(1) * MAX)));
104  newCol[2] = static_cast<Type>(
105  std::min(255.0, std::max(0.0, t(2) * MAX)));
106  }
107 
108  return RgbTpl(newCol[0], newCol[1], newCol[2]);
109  }

266  CV_DB_LIB_API static Rgb hsl2rgb(float H, float S, float L) {
267  H /= 360;
268  float q = L < 0.5f ? L * (1.0f + S) : L + S - L * S;
269  float p = 2 * L - q;
270 
271  float r = hue2rgb(p, q, H + 1.0f / 3.0f);
272  float g = hue2rgb(p, q, H);
273  float b = hue2rgb(p, q, H - 1.0f / 3.0f);
274 
275  return Rgb(static_cast<ColorCompType>(r * ecvColor::MAX),
276  static_cast<ColorCompType>(g * ecvColor::MAX),
277  static_cast<ColorCompType>(b * ecvColor::MAX));
278  }
279

286  CV_DB_LIB_API static Rgb hsv2rgb(float H, float S, float V) {
287  float hi = 0;
288  float f = std::modf(H / 60.0f, &hi);
289 
290  float l = V * (1.0f - S);
291  float m = V * (1.0f - f * S);
292  float n = V * (1.0f - (1.0f - f) * S);
293 
294  Rgbf rgb(0, 0, 0);
295 
296  switch (static_cast<int>(hi) % 6) {
297  case 0:
298  rgb.r = V;
299  rgb.g = n;
300  rgb.b = l;
301  break;
302  case 1:
303  rgb.r = m;
304  rgb.g = V;
305  rgb.b = l;
306  break;
307  case 2:
308  rgb.r = l;
309  rgb.g = V;
310  rgb.b = n;
311  break;
312  case 3:
313  rgb.r = l;
314  rgb.g = m;
315  rgb.b = V;
316  break;
317  case 4:
318  rgb.r = n;
319  rgb.g = l;
320  rgb.b = V;
321  break;
322  case 5:
323  rgb.r = V;
324  rgb.g = l;
325  rgb.b = m;
326  break;
327  }
328 
329  return Rgb(static_cast<ColorCompType>(rgb.r * ecvColor::MAX),
330  static_cast<ColorCompType>(rgb.g * ecvColor::MAX),
331  static_cast<ColorCompType>(rgb.b * ecvColor::MAX));
332  }
333 
334 protected:
336  static float hue2rgb(float m1, float m2, float hue) {
337  if (hue < 0)
338  hue += 1.0f;
339  else if (hue > 1.0f)
340  hue -= 1.0f;
341 
342  if (6 * hue < 1.0f)
343  return m1 + (m2 - m1) * hue * 6;
344  else if (2 * hue < 1.0f)
345  return m2;
346  else if (3 * hue < 2.0f)
347  return m1 + (m2 - m1) * (4.0f - hue * 6);
348  else
349  return m1;
350  }
