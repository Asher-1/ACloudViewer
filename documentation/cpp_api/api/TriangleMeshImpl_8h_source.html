41  core::ParallelFor(normals.GetDevice(), n,
42  [=] CLOUDVIEWER_DEVICE(int64_t workload_idx) {
43  int64_t idx = 3 * workload_idx;
44  scalar_t x = ptr[idx];
45  scalar_t y = ptr[idx + 1];
46  scalar_t z = ptr[idx + 2];

51  } else {

54  x /= norm;
55  y /= norm;
56  z /= norm;

58  ptr[idx] = x;
59  ptr[idx + 1] = y;
