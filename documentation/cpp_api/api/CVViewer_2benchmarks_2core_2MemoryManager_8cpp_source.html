19 std::shared_ptr<MemoryManagerDevice> MakeMemoryManager(
20  const Device& device, const MemoryManagerBackend& backend) {
21  std::shared_ptr<MemoryManagerDevice> device_mm;
22  switch (device.GetType()) {
23  case Device::DeviceType::CPU:
24  device_mm = std::make_shared<MemoryManagerCPU>();
25  break;
26 #ifdef BUILD_CUDA_MODULE
27  case Device::DeviceType::CUDA:
28  device_mm = std::make_shared<MemoryManagerCUDA>();
29  break;
30 #endif
31  default:
32  utility::LogError("Unimplemented device {}.", device.ToString());
33  break;
34  }
35 
36  switch (backend) {
37  case MemoryManagerBackend::Direct:
38  return device_mm;
39  case MemoryManagerBackend::Cached:
40  return std::make_shared<MemoryManagerCached>(device_mm);
41  default:
42  utility::LogError("Unimplemented backend.");
43  break;
44  }
45 }
