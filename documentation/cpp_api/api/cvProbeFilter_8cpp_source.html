81 void cvProbeFilter::apply() {
82  VtkUtils::vtkInitOnce(m_probe);
83  m_probe->SetInputData(m_dataObject);
84 
85  switch (m_widgetType) {
86  case WT_Line: {
87  VTK_CREATE(vtkLineSource, ls);
88  ls->SetPoint1(m_linePoint1);
89  ls->SetPoint2(m_linePoint2);
90  ls->Update();
91  m_probe->SetSourceData(ls->GetOutput());
92  } break;
93  case WT_Sphere: {
94  VTK_CREATE(vtkSphereSource, ss);
95  ss->SetCenter(m_sphereCenter);
96  ss->Update();
97  m_probe->SetSourceData(ss->GetOutput());
98  } break;
99  case WT_Box:
100  break;
101  case WT_ImplicitPlane: {
102  VTK_CREATE(vtkPlaneSource, ps);
103  ps->SetOrigin(m_planeOrigin);
104  ps->SetNormal(m_planeNormal);
105  ps->Update();
106  m_probe->SetSourceData(ps->GetOutput());
107 
108  } break;
109  }
110 
111  m_probe->Update();
112 
113  vtkPointData* pointData = m_probe->GetOutput()->GetPointData();
114 
115  Utils::ColorSeries colorSeries;
116  colorSeries.setScheme(Utils::ColorSeries::Cool);
117 
118  m_plotWidget->clearGraphs();
119  int numOfArray = pointData->GetNumberOfArrays();
120  for (int i = 0; i < numOfArray; ++i) {
121  QString arrayName(pointData->GetArrayName(i));
122  // skip this kind of data
123  if (arrayName.compare("vtkValidPointMask", Qt::CaseInsensitive) == 0)
124  continue;
125 
126  m_plotWidget->addGraph();
127  m_plotWidget->graph(i)->setName(arrayName);
128  m_plotWidget->graph(i)->setPen(colorSeries.nextColor());
129  vtkDataArray* da = pointData->GetArray(i);
130  int numOfTuples = da->GetNumberOfTuples();
131  QVector<double> x(numOfTuples), y(numOfTuples);
132 
133 #ifdef USE_TBB
134  tbb::parallel_for(0, numOfTuples,
135  [&](int dataIndex)
136 #else
137 
138 #if defined(_OPENMP)
139 #pragma omp parallel for
140 #endif
141  for (int dataIndex = 0; dataIndex < numOfTuples; ++dataIndex)
142 #endif
143  {
144  x[dataIndex] = dataIndex;
145  double value = da->GetTuple1(dataIndex);
146  y[dataIndex] = value;
147  }
148 #ifdef USE_TBB
149  );
150 #endif
151 
152  m_plotWidget->graph(i)->setData(x, y);
153  m_plotWidget->graph(i)->rescaleAxes(true);
154  }
155 
156  m_plotWidget->replot();
157  update();
158 }

188 void cvProbeFilter::getInteractorBounds(ccBBox& bbox) {
189  double bounds[6];
190 
191  bool valid = true;
192  switch (m_widgetType) {
193  case WidgetType::WT_Line:
194  if (m_lineWidget) {
195  // xmin, xmax, ymin, ymax, zmin, zmax
196  m_lineWidget->GetProp3D()->GetBounds(bounds);
197  }
198  break;
199  case WidgetType::WT_Box:
200  if (m_boxWidget) {
201  m_boxWidget->GetProp3D()->GetBounds(bounds);
202  }
203  break;
204  case WidgetType::WT_ImplicitPlane:
205  if (m_implicitPlaneWidget) {
206  m_implicitPlaneWidget->GetProp3D()->GetBounds(bounds);
207  }
208  break;
209  case WidgetType::WT_Sphere:
210  if (m_sphereWidget) {
211  // xmin, xmax, ymin, ymax, zmin, zmax
212  m_sphereWidget->GetProp3D()->GetBounds(bounds);
213  }
214  break;
215  default:
216  valid = false;
217  break;
218  }
219 
220  if (!valid) {
221  bbox.setValidity(valid);
222  return;
223  }
224 
225  CCVector3 minCorner(bounds[0], bounds[2], bounds[4]);
226  CCVector3 maxCorner(bounds[1], bounds[3], bounds[5]);
227  bbox.minCorner() = minCorner;
228  bbox.maxCorner() = maxCorner;
229  bbox.setValidity(valid);

259 void cvProbeFilter::shift(const CCVector3d& v) {
260  switch (m_widgetType) {
261  case WidgetType::WT_Line:
262  if (m_lineWidget) {
263  double newPos1[3];
264  double newPos2[3];
265  CCVector3d::vadd(v.u, m_linePoint1, newPos1);
266  CCVector3d::vadd(v.u, m_linePoint2, newPos2);
267  onLineWidgetPointsChanged(newPos1, newPos2);
268  }
269 
270  break;
271  case WidgetType::WT_Box:
272  if (m_boxWidget) {
273  VTK_CREATE(vtkTransform, trans);
274  m_boxWidget->GetTransform(trans);
275  trans->Translate(v.u);
276  m_boxWidget->SetTransform(trans);
277  apply();
278  }
279  break;
280  case WidgetType::WT_ImplicitPlane:
281  if (m_implicitPlaneWidget) {
282  // Modify and update planeWidget
283  }
284  break;
285  case WidgetType::WT_Sphere:
286  if (m_sphereWidget) {
287  double newPos[3];
288  CCVector3d::vadd(v.u, m_sphereCenter, newPos);
289  onSphereWidgetCenterChanged(newPos);
290  }
291  break;
292  default:
293  break;
294  }

296  showProbeWidget();

328  // default output path (+ filename)

334  ecvFileUtils::defaultDocPath(), filters, &selectedFilter);
335 
336  if (selectedFilename.isEmpty()) {
337  // process cancelled by the user
338  return nullptr;

340

342  QString suffix = fileInfo.suffix();
343  bool result = false;
344  if (suffix.compare("png", Qt::CaseInsensitive) == 0) {
345  result = m_plotWidget->savePng(selectedFilename, 1920, 1080);
346  }
347  /*else if (suffix.compare("jpg", Qt::CaseInsensitive) == 0)
348  {
349  result = m_plotWidget->saveJpg(selectedFilename, 1920, 1080);
350  }*/
351  else if (suffix.compare("bmp", Qt::CaseInsensitive) == 0) {
352  result = m_plotWidget->saveBmp(selectedFilename, 1920, 1080);
353  } else if (suffix.compare("pdf", Qt::CaseInsensitive) == 0) {
354  result = m_plotWidget->savePdf(selectedFilename, true, 1920, 1080);
355  }
356 
357  if (result) {
358  CVLog::Print(tr("Probe figure has been saved in %1")
359  .arg(selectedFilename));

361  CVLog::Warning(tr("failed to export probe figure!"));

365  return nullptr;

485 void cvProbeFilter::showProbeWidget() {
486  switch (m_widgetType) {
487  case WT_Line:
488  if (!m_lineWidget) {
489  VtkUtils::vtkInitOnce(m_lineWidget);
490  m_lineWidget->SetInteractor(getInteractor());
491  m_lineWidget->SetPlaceFactor(1);
492  m_lineWidget->SetClampToBounds(1);
493  m_lineWidget->SetProp3D(m_modelActor);
494  m_lineWidget->PlaceWidget();
495  // m_lineWidget->AddObserver(vtkCommand::EndInteractionEvent,
496  // new LineCallback);
497  VtkUtils::LineWidgetObserver* observer =
498  new VtkUtils::LineWidgetObserver(this);
499  connect(observer, SIGNAL(pointsChanged(double*, double*)), this,
500  SLOT(onLineWidgetPointsChanged(double*, double*)));
501  observer->attach(m_lineWidget);
502  }
503 
504  m_lineWidget->SetPoint1(m_linePoint1);
505  m_lineWidget->SetPoint2(m_linePoint2);
506  m_lineWidget->On();
507 
508  // disable the other two
509  if (m_sphereWidget) m_sphereWidget->Off();
510  if (m_boxWidget) m_boxWidget->Off();
511  if (m_implicitPlaneWidget) m_implicitPlaneWidget->Off();
512 
513  break;
514 
515  case WT_Sphere:
516  if (!m_sphereWidget) {
517  VtkUtils::vtkInitOnce(m_sphereWidget);
518  m_sphereWidget->SetInteractor(getInteractor());
519  m_sphereWidget->SetPlaceFactor(1);
520  m_sphereWidget->SetProp3D(m_modelActor);
521  m_sphereWidget->PlaceWidget();
522  VtkUtils::SphereWidgetObserver* observer =
523  new VtkUtils::SphereWidgetObserver(this);
524  connect(observer, SIGNAL(centerChanged(double*)), this,
525  SLOT(onSphereWidgetCenterChanged(double*)));
526  observer->attach(m_sphereWidget);
527  m_sphereRadius = m_sphereWidget->GetRadius();
528  }
529  m_sphereWidget->On();
530  m_sphereWidget->SetCenter(m_sphereCenter);
531  m_sphereWidget->SetRadius(m_sphereRadius);
532 
533  // disable the other two
534  if (m_lineWidget) m_lineWidget->Off();
535  if (m_boxWidget) m_boxWidget->Off();
536  if (m_implicitPlaneWidget) m_implicitPlaneWidget->Off();
537 
538  break;
539 
540  case WT_Box:
541  if (!m_boxWidget) {
542  VtkUtils::vtkInitOnce(m_boxWidget);
543  m_boxWidget->SetInteractor(getInteractor());
544  m_boxWidget->SetPlaceFactor(1);
545  m_boxWidget->SetProp3D(m_modelActor);
546  m_boxWidget->PlaceWidget();
547  }
548  m_boxWidget->On();
549 
550  // disable the other two
551  if (m_lineWidget) m_lineWidget->Off();
552  if (m_sphereWidget) m_sphereWidget->Off();
553  if (m_implicitPlaneWidget) m_implicitPlaneWidget->Off();
554 
555  break;
556 
557  case WT_ImplicitPlane:
558  if (!m_implicitPlaneWidget) {
559  VtkUtils::vtkInitOnce(m_implicitPlaneWidget);
560  m_implicitPlaneWidget->SetInteractor(getInteractor());
561  m_implicitPlaneWidget->SetPlaceFactor(1);
562  m_implicitPlaneWidget->SetProp3D(m_modelActor);
563  m_implicitPlaneWidget->SetOutlineTranslation(
564  0); // make the outline non-movable
565  m_implicitPlaneWidget->GetPlaneProperty()->SetOpacity(0.7);
566  m_implicitPlaneWidget->GetPlaneProperty()->SetColor(.0, .0, .0);
567 
568  double bounds[6];
569  if (isValidPolyData()) {
570  vtkPolyData* polyData =
571  vtkPolyData::SafeDownCast(m_dataObject);
572  polyData->GetBounds(bounds);
573  m_implicitPlaneWidget->SetOrigin(polyData->GetCenter());
574  } else if (isValidDataSet()) {
575  vtkDataSet* dataSet =
576  vtkDataSet::SafeDownCast(m_dataObject);
577  dataSet->GetBounds(bounds);
578  m_implicitPlaneWidget->SetOrigin(dataSet->GetCenter());
579  }
580  // m_implicitPlaneWidget->SetOutsideBounds(0);
581  m_implicitPlaneWidget->PlaceWidget(bounds);
582 
583  VtkUtils::ImplicitPlaneWidgetObserver* observer =
584  new VtkUtils::ImplicitPlaneWidgetObserver(this);
585  connect(observer, SIGNAL(originChanged(double*)), this,
586  SLOT(onImplicitPlaneWidgetOriginChanged(double*)));
587  connect(observer, SIGNAL(normalChanged(double*)), this,
588  SLOT(onImplicitPlaneWidgetNormalChanged(double*)));
589  observer->attach(m_implicitPlaneWidget);
590  }
591  m_implicitPlaneWidget->On();
592  m_implicitPlaneWidget->UpdatePlacement();
593 
594  // disable the other two
595  if (m_lineWidget) m_lineWidget->Off();
596  if (m_sphereWidget) m_sphereWidget->Off();
597  if (m_boxWidget) m_boxWidget->Off();
598 
599  break;
600  }
601 
602  applyDisplayEffect();
