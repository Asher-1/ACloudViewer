74 QString ImageFileFilter::GetSaveFilename(const QString& dialogTitle,
75  const QString& baseName,
76  const QString& imageSavePath,
77  QWidget* parentWidget /*=0*/) {
78  // add images output file filters
79  QString filters;
80 
81  // we grab the list of supported image file formats (writing)
82  QList<QByteArray> formats = QImageWriter::supportedImageFormats();
83  if (formats.empty()) {
84  CVLog::Error(
85  "No image format supported by your system?!\n(check that the "
86  "'imageformats' directory is alongside the application "
87  "executable)");
88  return QString();
89  }
90 
91  // we convert this list into a proper "filters" string
92  QString pngFilter;
93  for (int i = 0; i < formats.size(); ++i) {
94  QString ext = QString(formats[i].data()).toUpper();
95  QString filter = QString("%1 image (*.%2)").arg(ext, formats[i].data());
96  filters.append(filter + QString("\n"));
97 
98  // find PNG by default
99  if (pngFilter.isEmpty() && ext == "PNG") {
100  pngFilter = filter;
101  }
102  }
103 
104  QString outputFilename = QFileDialog::getSaveFileName(
105  parentWidget, dialogTitle,
106  imageSavePath +
107  QString("/%1.%2").arg(baseName,
108  pngFilter.isEmpty()
109  ? QString(formats[0].data())
110  : QString("png")),
111  filters,
112  pngFilter.isEmpty() ? static_cast<QString*>(0) : &pngFilter);
113 
114  return outputFilename;
