60 void cvGlyphFilter::apply() {
61  if (!m_dataObject) return;
62 
63  VtkUtils::vtkInitOnce(m_glyph3d);
64  m_glyph3d->SetInputData(m_dataObject);
65  m_glyph3d->SetScaleModeToDataScalingOff();
66  m_glyph3d->SetScaling(0);
67 
68  switch (m_shape) {
69  case Arrow: {
70  VTK_CREATE(vtkArrowSource, as);
71  m_glyph3d->SetSourceConnection(as->GetOutputPort());
72  } break;
73 
74  case Cone: {
75  VTK_CREATE(vtkConeSource, cs);
76  m_glyph3d->SetSourceConnection(cs->GetOutputPort());
77  } break;
78 
79  case Line: {
80  VTK_CREATE(vtkLineSource, ls);
81  m_glyph3d->SetSourceConnection(ls->GetOutputPort());
82  } break;
83 
84  case Cylinder: {
85  VTK_CREATE(vtkCylinderSource, cs);
86  m_glyph3d->SetSourceConnection(cs->GetOutputPort());
87  } break;
88 
89  case Sphere: {
90  VTK_CREATE(vtkSphereSource, ss);
91  m_glyph3d->SetSourceConnection(ss->GetOutputPort());
92  } break;
93 
94  case Point: {
95  VTK_CREATE(vtkPointSource, ps);
96  ps->SetNumberOfPoints(1);
97  m_glyph3d->SetSourceConnection(ps->GetOutputPort());
98  } break;
99  }
100 
101  // Create a mapper and actor
102  VTK_CREATE(vtkPolyDataMapper, mapper);
103  mapper->SetInputConnection(m_glyph3d->GetOutputPort());
104  mapper->SetScalarVisibility(!m_glyphColor.isValid());
105 
106  if (!m_filterActor) {
107  VtkUtils::vtkInitOnce(m_filterActor);
108  m_filterActor->SetMapper(mapper);
109  addActor(m_filterActor);
110  }
111 
112  double vtkClr[3];
113  Utils::vtkColor(m_glyphColor, vtkClr);
114  m_filterActor->GetProperty()->SetColor(vtkClr);
115  m_filterActor->GetProperty()->SetPointSize(m_size);
116  showLabels(m_labelVisible);
117 
118  applyDisplayEffect();
119 }

201 void cvGlyphFilter::showLabels(bool show) {
202  if (!show && !m_labelActor) return;
203 
204  if (m_labelActor && !show) {
205  m_labelActor->SetVisibility(show);
206  update();
207  return;
208  }
209 
210  if (!m_labelActor) {
211  VtkUtils::vtkInitOnce(m_labelActor);
212  addActor(m_labelActor);
213  CVLog::Warning(QString("showing labels may reduce efficiency!"));
214  }
215 
216  VTK_CREATE(vtkLabeledDataMapper, labelMapper);
217  labelMapper->SetFieldDataName("Isovalues");
218  labelMapper->SetInputData(m_dataObject);
219  labelMapper->SetLabelMode(m_labelMode);
220  labelMapper->SetLabelFormat("%6.2f");
221 
222  m_labelActor->SetMapper(labelMapper);
223  m_labelActor->SetVisibility(show);
224  update();
225 }
