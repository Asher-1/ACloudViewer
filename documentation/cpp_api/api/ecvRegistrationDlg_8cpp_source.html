218 int ccRegistrationDlg::getTransformationFilters() const {
219  int filters = 0;
220  switch (rotComboBox->currentIndex()) {
221  case 1:
222  filters |= cloudViewer::RegistrationTools::SKIP_RYZ;
223  break;
224  case 2:
225  filters |= cloudViewer::RegistrationTools::SKIP_RXZ;
226  break;
227  case 3:
228  filters |= cloudViewer::RegistrationTools::SKIP_RXY;
229  break;
230  default:
231  // nothing to do
232  break;
233  }
234 
235  if (!TxCheckBox->isChecked())
236  filters |= cloudViewer::RegistrationTools::SKIP_TX;
237  if (!TyCheckBox->isChecked())
238  filters |= cloudViewer::RegistrationTools::SKIP_TY;
239  if (!TzCheckBox->isChecked())
240  filters |= cloudViewer::RegistrationTools::SKIP_TZ;
241 
242  return filters;
243 }
