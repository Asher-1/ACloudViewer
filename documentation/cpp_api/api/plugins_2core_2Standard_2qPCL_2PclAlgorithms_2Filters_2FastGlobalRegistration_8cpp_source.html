41 template <typename T>
42 static fgr::KDTree BuildKDTree(const vector<T>& data) {
43  const int rows = static_cast<int>(data.size());
44  const int dim = static_cast<int>(data.front().size());
45 
46  Distances dataset(rows * dim);
47  for (int i = 0; i < rows; i++)
48  for (int j = 0; j < dim; j++)
49  dataset[i * dim + j] = static_cast<float>(data[i][j]);
50 
51  flann::Matrix<float> dataset_mat(dataset.data(), rows, dim);
52  KDTree tree(dataset_mat, flann::KDTreeSingleIndexParams(15));
53  tree.buildIndex();
54 
55  return tree;
56 }

59 static void SearchKDTree(fgr::KDTree& tree,
60  const T& input,
61  int& nearestIndex,
62  float& smallestSquareDist) {
63  static const size_t Rows = 1;
64  static const size_t NN = 1;
65  const size_t cols = static_cast<size_t>(input.size());
66 
67  Distances query(Rows * cols);
68  for (size_t i = 0; i < cols; i++) query[i] = input(i);
69  flann::Matrix<float> query_mat(query.data(), Rows, cols);
70 
71  flann::Matrix<int> indices_mat(&nearestIndex, Rows, NN);
72  flann::Matrix<float> dists_mat(&smallestSquareDist, Rows, NN);
73 
74  tree.knnSearch(query_mat, indices_mat, dists_mat, NN,
75  flann::SearchParams(128));
76 }

78 static void TransformPoints(fgr::Points& points, const Eigen::Matrix4f& Trans) {
79  size_t npc = points.size();
80 
81  Matrix3f R = Trans.block<3, 3>(0, 0);
82  Vector3f t = Trans.block<3, 1>(0, 3);
83 
84  for (size_t cnt = 0; cnt < npc; cnt++) {
85  Vector3f temp = R * points[cnt] + t;
86  points[cnt] = temp;
87  }
88 }

90 static bool Similar(const Eigen::Vector3f& ACloud1,
91  const Eigen::Vector3f& BCloud1,
92  const Eigen::Vector3f& ACloud2,
93  const Eigen::Vector3f& BCloud2,
94  double maxRelativeScaleDiff) {
95  float ABCloud1 = (BCloud1 - ACloud1).norm();
96  float ABCloud2 = (BCloud2 - ACloud2).norm();
97  return (maxRelativeScaleDiff * ABCloud1 < ABCloud2) &&
98  (ABCloud2 * maxRelativeScaleDiff < ABCloud1);

101 void CApp::AdvancedMatching(bool crossCheck /*=true*/,
102  bool tupleConstraint /*=true*/) {
103  correspondances_.clear();
104 
105  // choose the cloud order (largest first)
106  int cloud1Index = ReferenceIndex;
107  int cloud2Index = AlignedIndex;
108  bool swapped = false;
109 
110  if (pointClouds_[cloud2Index].size() > pointClouds_[cloud1Index].size()) {
111  std::swap(cloud1Index, cloud2Index);
112  swapped = true;
113  }
114 
118 
119  const Features& cloud1Features = features_[cloud1Index];
120  const Features& cloud2Features = features_[cloud2Index];
121 
122  KDTree feature_tree_cloud1 = BuildKDTree(cloud1Features);
123  KDTree feature_tree_cloud2 = BuildKDTree(cloud2Features);
124 
128 
129  Correspondences corres;
130  {
131  const size_t cloud1Size = pointClouds_[cloud1Index].size();
132  const size_t cloud2Size = pointClouds_[cloud2Index].size();
133 
134  // First, look at correspondances from cloud2 to cloud1
135  Indexes cloud1_to_cloud2(cloud1Size, -1);
136  for (int indexInC2 = 0; indexInC2 < cloud2Size; ++indexInC2) {
137  int nearestIndexInC1 = -1;
138  float squareDistance = 0; // not used
139  SearchKDTree(feature_tree_cloud1, cloud2Features[indexInC2],
140  nearestIndexInC1, squareDistance);
141  assert(nearestIndexInC1 != -1);
142 
143  // Then check the 'backward' correspondance starting from the
144  // nearest point in cloud1 (if not already done)
145  if (cloud1_to_cloud2[nearestIndexInC1] == -1) {
146  int nearestIndexInC2 = -1;
147  SearchKDTree(feature_tree_cloud2,
148  cloud1Features[nearestIndexInC1], nearestIndexInC2,
149  squareDistance);
150  assert(nearestIndexInC2 != -1);
151  cloud1_to_cloud2[nearestIndexInC1] = nearestIndexInC2;
152 
153  if (!crossCheck) {
154  // add the corresponding 'backward' correspondance from
155  // cloud1 to cloud2 to the global correspondances list
156  corres.push_back(Pair(nearestIndexInC1, nearestIndexInC2));
157  }
158  }
159 
160  if (!crossCheck ||
161  indexInC2 == cloud1_to_cloud2[nearestIndexInC1]) {
162  // add the 'forward' correspondance from cloud2 to cloud1
163  corres.push_back(Pair(nearestIndexInC1, indexInC2));
164  }
165  }
166  corres.shrink_to_fit();
167  }
168 
169  CVLog::Print(QString("Number of pairs %1 cross-check: %2")
170  .arg(crossCheck ? "with" : "without")
171  .arg(corres.size()));
172 
176  if (tupleConstraint) {
177  srand(static_cast<unsigned>(time(NULL)));
178 
179  const size_t ncorr = corres.size();
180  const size_t number_of_trial = ncorr * 100;
181 
182  const Points& cloud1 = pointClouds_[cloud1Index];
183  const Points& cloud2 = pointClouds_[cloud2Index];
184 
185  size_t cnt = 0;
186  Correspondences corres_tuple;
187  corres_tuple.reserve(tuple_max_cnt_ * 3);
188  for (size_t i = 0; i < number_of_trial; i++) {
189  size_t rand0 = (static_cast<size_t>(rand()) % ncorr);
190  size_t rand1 = (static_cast<size_t>(rand()) % ncorr);
191  if (rand1 == rand0) continue;
192  size_t rand2 = (static_cast<size_t>(rand()) % ncorr);
193  if (rand2 == rand0 || rand2 == rand1) continue;
194 
195  const Pair& pair0 = corres[rand0];
196  const Pair& pair1 = corres[rand1];
197  const Pair& pair2 = corres[rand2];
198 
199  // collect 3 points from first cloud
200  const Eigen::Vector3f& ptCloud1_0 = cloud1[pair0.first];
201  const Eigen::Vector3f& ptCloud1_1 = cloud1[pair1.first];
202  const Eigen::Vector3f& ptCloud1_2 = cloud1[pair2.first];
203 
204  // collect the 3 corresponding points from second cloud
205  const Eigen::Vector3f& ptCloud2_0 = cloud2[pair0.second];
206  const Eigen::Vector3f& ptCloud2_1 = cloud2[pair1.second];
207  const Eigen::Vector3f& ptCloud2_2 = cloud2[pair2.second];
208 
209  if (Similar(ptCloud1_0, ptCloud1_1, ptCloud2_0, ptCloud2_1,
210  tuple_scale_) &&
211  Similar(ptCloud1_0, ptCloud1_2, ptCloud2_0, ptCloud2_2,
212  tuple_scale_) &&
213  Similar(ptCloud1_1, ptCloud1_2, ptCloud2_1, ptCloud2_2,
214  tuple_scale_)) {
215  // confirm the pairs
216  corres_tuple.push_back(Pair(pair0.first, pair0.second));
217  corres_tuple.push_back(Pair(pair1.first, pair1.second));
218  corres_tuple.push_back(Pair(pair2.first, pair2.second));
219 
220  if (++cnt >= tuple_max_cnt_) break;
221  }
222  }
223  corres_tuple.shrink_to_fit();
224  corres.swap(corres_tuple);
225 
226  CVLog::Print("Number of pairs after tuple constraint: " +
227  QString::number(corres.size()));
228  }
229 
230  if (swapped) {
231  for (Pair& pair : corres) {
232  std::swap(pair.first, pair.second);
233  }
234  }
235 
236  correspondances_.swap(corres);
237 }

239 void CApp::NormalizePoints() {
240  float scale = 0;
241 
242  for (int i = 0; i < 2; ++i) {
243  // compute mean
244  Vector3f mean(0, 0, 0);
245 
246  size_t npti = pointClouds_[i].size();
247  for (size_t ii = 0; ii < npti; ++ii) {
248  mean += pointClouds_[i][ii];
249  }
250  mean = mean / npti;
251  means_[i] = mean;
252 
253  // printf("normalize points :: mean[%d] = [%f %f %f]\n", i, mean(0),
254  // mean(1), mean(2));
255 
256  for (size_t ii = 0; ii < npti; ++ii) {
257  pointClouds_[i][ii] -= mean;
258  }
259 
260  // compute scale
261  float maxNorm = 0;
262  for (size_t ii = 0; ii < npti; ++ii) {
263  float norm = pointClouds_[i][ii].norm(); // because we extract mean
264  // in the previous stage.
265  if (norm > maxNorm) maxNorm = norm;
266  }
267 
268  if (maxNorm > scale) scale = maxNorm;
269  }
270 
272  if (use_absolute_scale_) {
273  globalScale_ = 1.0f;
274  startScale_ = scale;
275  } else {
276  globalScale_ = scale; // second choice: we keep the maximum scale.
277  startScale_ = 1.0f;
278  }
279  // printf("normalize points :: global scale : %f\n", globalScale_);
280 
281  for (int i = 0; i < 2; ++i) {
282  size_t npti = pointClouds_[i].size();
283  for (size_t ii = 0; ii < npti; ++ii) {
284  pointClouds_[i][ii] /= globalScale_;
285  }
286  }

289 bool CApp::OptimizePairwise(bool decrease_mu) {
290  // printf("Pairwise rigid pose optimization\n");
291  transOutput_ = Eigen::Matrix4f::Identity();
292 
293  if (correspondances_.size() < 10) return false;
294 
295  // make another copy of pointClouds_[AlignedIndex].
296  Points pcj_copy = pointClouds_[AlignedIndex];
297 
298  std::vector<double> s(correspondances_.size(), 1.0);
299 
300  Eigen::Matrix4f trans;
301  trans.setIdentity();
302  double par = startScale_;
303 
304  for (size_t itr = 0; itr < iteration_number_; itr++) {
305  // graduated non-convexity.
306  if (decrease_mu) {
307  if ((itr % 4) == 0 && (par > max_corr_dist_)) {
308  par /= div_factor_;
309  }
310  }
311 
312  static const int NVariable = 6; // 3 for rotation and 3 for translation
313  Eigen::MatrixXd JTJ(NVariable, NVariable);
314  JTJ.setZero();
315  Eigen::MatrixXd JTr(NVariable, 1);
316  JTr.setZero();
317 
318  // double r2 = 0.0;
319 
320  for (int c = 0; c < correspondances_.size(); c++) {
321  int ii = correspondances_[c].first;
322  int jj = correspondances_[c].second;
323  const Eigen::Vector3f& p = pointClouds_[ReferenceIndex][ii];
324  const Eigen::Vector3f& q = pcj_copy[jj];
325  Eigen::Vector3f rpq = p - q;
326 
327  double temp = par / (rpq.dot(rpq) + par);
328  s[c] = temp * temp;
329 
330  Eigen::MatrixXd J(NVariable, 1);
331  J.setZero();
332  J(1) = -q(2);
333  J(2) = q(1);
334  J(3) = -1;
335  double r = rpq(0);
336  JTJ += J * J.transpose() * s[c];
337  JTr += J * r * s[c];
338  // r2 += r * r * s[c];
339 
340  J.setZero();
341  J(2) = -q(0);
342  J(0) = q(2);
343  J(4) = -1;
344  r = rpq(1);
345  JTJ += J * J.transpose() * s[c];
346  JTr += J * r * s[c];
347  // r2 += r * r * s[c];
348 
349  J.setZero();
350  J(0) = -q(1);
351  J(1) = q(0);
352  J(5) = -1;
353  r = rpq(2);
354  JTJ += J * J.transpose() * s[c];
355  JTr += J * r * s[c];
356  // r2 += r * r * s[c];
357 
358  // r2 += (par * (1.0 - sqrt(s[c])) * (1.0 - sqrt(s[c])));
359  }
360 
361  Eigen::MatrixXd result = -JTJ.llt().solve(JTr);
362 
363  Eigen::Affine3d aff_mat;
364  aff_mat.linear() = static_cast<Eigen::Matrix3d>(
365  Eigen::AngleAxisd(result(2), Eigen::Vector3d::UnitZ()) *
366  Eigen::AngleAxisd(result(1), Eigen::Vector3d::UnitY()) *
367  Eigen::AngleAxisd(result(0), Eigen::Vector3d::UnitX()));
368  aff_mat.translation() =
369  Eigen::Vector3d(result(3), result(4), result(5));
370 
371  Eigen::Matrix4f delta = aff_mat.matrix().cast<float>();
372 
373  trans = delta * trans;
374  TransformPoints(pcj_copy, delta);
375  }
376 
377  transOutput_ = trans * transOutput_;
378 
379  return true;
380 }
