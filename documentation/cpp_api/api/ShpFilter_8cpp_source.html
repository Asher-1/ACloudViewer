221 static bool areVerticesCounterClockwise(
222  const cloudViewer::GenericIndexedCloudPersist *vertices,
223  int32_t numPoints,
224  unsigned char dim1,
225  unsigned char dim2) {
226  // http://stackoverflow.com/questions/1165647/how-to-determine-if-a-list-of-polygon-points-are-in-clockwise-order
227  // or http://en.wikipedia.org/wiki/Shoelace_formula
228  double sum = 0.0;
229  unsigned realNumPoints = vertices->size();
230  for (int32_t i = 0; i + 1 < numPoints; ++i) {
231  CCVector3 P1;
232  CCVector3 P2;
233  vertices->getPoint(i, P1);
234  vertices->getPoint((i + 1) % realNumPoints, P2);
235  sum += (P2.u[dim1] - P2.u[dim1]) * (P2.u[dim2] + P1.u[dim2]);
236  }
237 
238  // negative sum = counter-clockwise
239  return sum < 0.0;
240 }

276 CCVector2d mRangeOfContainer(ccHObject::Container &objects) {
277  CCVector2d range(std::numeric_limits<double>::max(),
278  std::numeric_limits<double>::min());
279 
280  auto updateRange =
281  [&range](const cloudViewer::GenericIndexedCloudPersist *cloud) {
282  if (!cloud->isScalarFieldEnabled()) return;
283  unsigned numPoints = cloud->size();
284  for (unsigned i = 0; i < numPoints; ++i) {
285  ScalarType val = cloud->getPointScalarValue(i);
286  if (val != NAN_VALUE) {
287  auto m = static_cast<double>(val);
288  range.x = std::min(range.x, m);
289  range.y = std::max(range.y, m);
290  }
291  }
292  };
293 
294  // call the same method on the first child so as to get its type
295  for (ccHObject *obj : objects) {
296  switch (obj->getClassID()) {
297  case CV_TYPES::POINT_CLOUD: {
298  const ccGenericPointCloud *cloud =
299  ccHObjectCaster::ToGenericPointCloud(obj);
300  updateRange(cloud);
301  break;
302  }
303  case CV_TYPES::POLY_LINE: {
304  const ccPolyline *poly = ccHObjectCaster::ToPolyline(obj);
305  const cloudViewer::GenericIndexedCloudPersist *vertices =
306  poly->getAssociatedCloud();
307  updateRange(vertices);
308  break;
309  }
310  default:
311  break;
312  }
313  }
314 
315  if (range.x == std::numeric_limits<double>::max()) {
316  range.x = 0.0;
317  range.y = 0.0;
318  }

346 CC_FILE_ERROR ShapeFileHeader::readFrom(QDataStream &sin) {
347  sin.setByteOrder(QDataStream::BigEndian);
348 
349  int32_t fileCode;
350  sin >> fileCode;
351  if (fileCode != ESRI_SHAPE_FILE_CODE) {
352  CVLog::Warning("[SHP] wrong file code (%d), is this a shape file?",
353  fileCode);
354  return CC_FERR_MALFORMED_FILE;
355  }
356 
357  sin.skipRawData(5 * sizeof(int32_t));
358 
359  sin >> fileLength;
360  fileLength *= 2; // fileLength is measured in 16-bit words
361 
362  sin.setByteOrder(QDataStream::LittleEndian);
363 
364  sin >> version >> shapeTypeInt;
365 
366  if (!isValidESRIShapeCode(shapeTypeInt)) {
367  CVLog::Warning("[SHP] invalid shape type code in header (%d)",
368  shapeTypeInt);
369  return CC_FERR_MALFORMED_FILE;
370  }
371 
372  sin >> pointMin.x >> pointMin.y >> pointMax.x >> pointMax.y;
373  sin >> pointMin.z >> pointMax.z;
374 
375  pointMin.z = std::isnan(pointMin.z) ? 0 : pointMin.z;
376  pointMax.z = std::isnan(pointMax.z) ? 0 : pointMax.z;
377 
378  sin >> mRange.x >> mRange.y;
379 
380  if (sin.status() != QDataStream::Ok) {
381  CVLog::Warning("[SHP] Something went wrong reading the shp header");
382  return CC_FERR_READING;
383  }
384 
385  return CC_FERR_NO_ERROR;
386 }

388 CC_FILE_ERROR ShapeFileHeader::writeTo(QDataStream &out) {
389  out.setByteOrder(QDataStream::BigEndian);
390 
391  out << ESRI_SHAPE_FILE_CODE;
392 
393  constexpr size_t numBytesToSkip = 5 * sizeof(int32_t);
394  constexpr char skipValues[numBytesToSkip] = {0};
395  out.writeRawData(skipValues, numBytesToSkip);
396 
397  out << fileLength;
398 
399  out.setByteOrder(QDataStream::LittleEndian);
400  out << version;
401  out << shapeTypeInt;
402  out << pointMin.x << pointMin.y << pointMax.x << pointMax.y;
403  out << pointMin.z << pointMax.z;
404 
405  out << mRange.x << mRange.y;
406 
407  assert(out.device()->pos() == ESRI_HEADER_SIZE);
408  return CC_FERR_NO_ERROR;
409 }

506 static void GetSupportedShapes(ccHObject *baseEntity,
507  ccHObject::Container &shapes,
508  ESRI_SHAPE_TYPE &shapeType) {
509  shapeType = ESRI_SHAPE_TYPE::NULL_SHAPE;
510  if (!baseEntity) {
511  assert(false);
512  shapes.clear();
513  return;
514  }
515 
516  switch (baseEntity->getClassID()) {
517  case CV_TYPES::POINT_CLOUD: {
518  unsigned count =
519  ccHObjectCaster::ToGenericPointCloud(baseEntity)->size();
520  if (count != 0) {
521  shapeType = ESRI_SHAPE_TYPE::MULTI_POINT_Z;
522  shapes.push_back(baseEntity);
523  }
524  break;
525  }
526  case CV_TYPES::POLY_LINE: {
527  shapeType = ESRI_SHAPE_TYPE::POLYLINE_Z;
528  shapes.push_back(baseEntity);
529  break;
530  }
531  case CV_TYPES::MESH: {
532  shapeType = ESRI_SHAPE_TYPE::MULTI_PATCH;
533  shapes.push_back(baseEntity);
534  break;
535  }
536  case CV_TYPES::HIERARCHY_OBJECT:
537  // we only allow groups with children of the same type!
538  if (baseEntity->getChildrenNumber()) {
539  ccHObject *child = baseEntity->getChild(0);
540  assert(child);
541  if (!child) return;
542 
543  // first we check that all entities have the same type
544  for (unsigned i = 1; i < baseEntity->getChildrenNumber(); ++i) {
545  if (baseEntity->getChild(i) &&
546  baseEntity->getChild(i)->getClassID() !=
547  child->getClassID()) {
548  // mixed shapes are not allowed in shape files
549  return;
550  }
551  }
552 
553  // call the same method on the first child so as to get its type
554  GetSupportedShapes(child, shapes, shapeType);
555  if (shapeType == ESRI_SHAPE_TYPE::NULL_SHAPE) return;
556 
557  // then add the remaining children
558  for (unsigned i = 1; i < baseEntity->getChildrenNumber(); ++i) {
559  ESRI_SHAPE_TYPE otherShapeType =
560  ESRI_SHAPE_TYPE::NULL_SHAPE;
561  ccHObject *child = baseEntity->getChild(i);
562  if (child)
563  GetSupportedShapes(child, shapes, otherShapeType);
564 
565  if (otherShapeType != shapeType) {
566  if (child)
567  CVLog::Warning(
568  QString("[SHP] Entity %1 has not the same "
569  "type (%2) as the others in the "
570  "selection (%3)! Can't mix "
571  "types...")
572  .arg(child->getName())
573  .arg(ToString(otherShapeType))
574  .arg(ToString(shapeType)));
575  // mixed shapes are not allowed in shape files
576  shapes.clear();
577  return;
578  }
579  }
580  }
581  break;
582  default:
583  // nothing to do
584  break;
585  }
586 }

627  int32_t numPoints) {
628  std::vector<ScalarType> scalarValues;
629  // M boundaries
630  double mMin;
631  double mMax;
632  shpStream >> mMin >> mMax;
633

635  try {
636  scalarValues.resize(numPoints);
637  } catch (const std::bad_alloc &) {

652  }

725 CC_FILE_ERROR buildPatches(ccHObject &container,
726  const std::vector<int32_t> &startIndexes,
727  const std::vector<int32_t> &partTypes,
728  const std::vector<CCVector3> &points,
729  const std::vector<ScalarType> &scalarValues) {
730  size_t numParts = startIndexes.size();
731  size_t numPoints = points.size();
732 
733  for (int32_t i = 0; i < numParts; ++i) {
734  if (!isValidEsriPartType(partTypes[i])) {
735  CVLog::Warning(
736  "[SHP] Multipatch part %d has an invalid part type (%d)", i,
737  partTypes[i]);
738  continue;
739  }
740  auto type = static_cast<ESRI_PART_TYPE>(partTypes[i]);
741 
742  const int32_t &firstIndex = startIndexes[i];
743  const int32_t &lastIndex = static_cast<const int32_t &>(
744  (i + 1 < numParts ? startIndexes[i + 1] : numPoints) - 1);
745  const int32_t vertCount = lastIndex - firstIndex + 1;
746 
747  switch (type) {
748  case ESRI_PART_TYPE::TRIANGLE_STRIP: {
749  ccMesh *mesh =
750  createMesh(points, scalarValues, firstIndex, lastIndex);
751  for (int32_t j = 2; j < vertCount; ++j) {

762  }
763  container.addChild(mesh);
764  break;
765  }
766  default:
767  CVLog::Print(QString("[SHP] Cannot handle Patch of type: %1")
768  .arg(ToString(type)));

770  }
771  }

963  bool preserveCoordinateShift,
964  bool load2DPolyAs3DPoly = true) {

973  if (startIndexes.empty()) {
974  return CC_FERR_NOT_ENOUGH_MEMORY;
975  }
976  // for each part, the index of its first point in the points array
977  // FIXME: we should use this information and create as many polylines as

979 
980  // Points (An array of length NumPoints)
981  std::vector<CCVector3> points = readPoints(shpStream, numPoints, Pshift);
982  if (points.empty()) {
983  return CC_FERR_NOT_ENOUGH_MEMORY;
984  }
985

987  bool is3D = isESRIShape3D(shapeType);
988  if (is3D) {
989  // Z boundaries
990  shpStream.skipRawData(2 * sizeof(double));
991 
992  // Z coordinates (an array of length NumPoints)
993  for (int32_t i = 0; i < numPoints; ++i) {
994  double z;
995  shpStream >> z;
996  points[i].z = static_cast<PointCoordinateType>(z + Pshift.z);
997  }
998  }
999

1001  std::vector<ScalarType> scalarValues;
1002  if (hasMeasurements(shapeType)) {
1003  scalarValues = readMeasures(shpStream, numPoints);
1004  if (scalarValues.empty()) {
1005  // not enough memory to load scalar values!

1007  "load scalar values!")

1015  const int32_t &lastIndex =
1016  (i + 1 < numParts ? startIndexes[i + 1] : numPoints) - 1;
1017  int32_t vertCount = lastIndex - firstIndex + 1;
1018 
1019  // test if the polyline is closed
1020  bool isClosed = false;
1021  if (vertCount > 2 &&
1022  cloudViewer::LessThanEpsilon(
1023  (points[firstIndex] - points[lastIndex]).norm())) {
1024  vertCount--;
1025  isClosed = true;
1026  }

1029  ccPointCloud *vertices = new ccPointCloud("vertices");
1030  if (!vertices->reserve(vertCount)) {
1031  delete vertices;
1032  return CC_FERR_NOT_ENOUGH_MEMORY;
1033  }
1034  for (int32_t j = 0; j < vertCount; ++j) {
1035  vertices->addPoint(points[firstIndex + j]);
1036  }
1037  vertices->setEnabled(false);
1038  if (preserveCoordinateShift) {
1039  vertices->setGlobalShift(Pshift);
1040  }
1041 
1042  // polyline
1043  ccPolyline *poly = new ccPolyline(vertices);
1044  poly->addChild(vertices);
1045  if (preserveCoordinateShift) {
1046  poly->setGlobalShift(
1047  Pshift); // shouldn't be necessary but who knows ;)
1048  }
1049 
1050  if (!poly->reserve(vertCount)) {
1051  delete poly;
1052  return CC_FERR_NOT_ENOUGH_MEMORY;
1053  }
1054  poly->addPointIndex(0, static_cast<unsigned>(vertCount));
1055  poly->showSF(vertices->sfShown());
1056  QString name = QString("Polyline #%1").arg(index);
1057  if (numParts != 1) name += QString(".%1").arg(i + 1);
1058  poly->setName(name);
1059  poly->setClosed(isClosed);
1060  poly->set2DMode(!is3D && !load2DPolyAs3DPoly);
1061 
1062  if (!scalarValues.empty()) {
1063  bool allNans =
1064  std::all_of(scalarValues.begin() + firstIndex,
1065  scalarValues.begin() + lastIndex,
1066  [](ScalarType v) { return std::isnan(v); });
1067  if (!allNans) {
1068  auto *sf = new ccScalarField("Measures");
1069  if (!sf->reserveSafe(vertCount)) {
1070  CVLog::Warning(QString("[SHP] Polyline #%1.%2: not enough "
1071  "memory to load scalar values!")
1072  .arg(index)
1073  .arg(i + 1));
1074  sf->release();
1075  sf = nullptr;
1076  }
1077  for (int32_t j = 0; j < vertCount; ++j) {
1078  sf->addElement(scalarValues[j + firstIndex]);
1079  }
1080  sf->computeMinAndMax();
1081  int sfIdx = vertices->addScalarField(sf);
1082  vertices->setCurrentDisplayedScalarField(sfIdx);
1083  vertices->showSF(true);
1084  vertices->enableScalarField();
1085  }
1086  }
1087  container.addChild(poly);
1088  }
1089  return CC_FERR_NO_ERROR;

1092 static CC_FILE_ERROR SavePolyline(ccPolyline *poly,
1093  QDataStream &out,
1094  int32_t &recordSize,
1095  int32_t recordNumber,
1096  ESRI_SHAPE_TYPE outputShapeType,
1097  int vertDim = 2) {
1098  assert(vertDim >= 0 && vertDim < 3);
1099 
1100  if (!poly) {
1101  assert(false);
1102  return CC_FERR_BAD_ENTITY_TYPE;
1103  }
1104 
1105  const unsigned char Z = static_cast<unsigned char>(vertDim);
1106  const unsigned char X = Z == 2 ? 0 : Z + 1;
1107  const unsigned char Y = X == 2 ? 0 : X + 1;
1108 
1109  cloudViewer::GenericIndexedCloudPersist *vertices =
1110  poly->getAssociatedCloud();
1111  if (!vertices) {
1112  return CC_FERR_BAD_ENTITY_TYPE;
1113  }
1114 
1115  unsigned realVertexCount = poly->size();
1116  switch (outputShapeType) {
1117  case ESRI_SHAPE_TYPE::POLYGON:
1118  case ESRI_SHAPE_TYPE::POLYGON_M:
1119  case ESRI_SHAPE_TYPE::POLYGON_Z:
1120  if (realVertexCount < 3) {
1121  CVLog::Warning(
1122  QObject::tr("[SHP] Polyline %1 does not have enough "
1123  "vertices to be saved as polygon entity")
1124  .arg(poly->getName()));
1125  return CC_FERR_BAD_ENTITY_TYPE;
1126  }
1127  break;
1128  case ESRI_SHAPE_TYPE::POLYLINE:
1129  case ESRI_SHAPE_TYPE::POLYLINE_M:
1130  case ESRI_SHAPE_TYPE::POLYLINE_Z:
1131  if (realVertexCount < 2) {
1132  CVLog::Warning(
1133  QObject::tr("[SHP] Polyline %1 does not have enough "
1134  "vertices to be saved as polyline entity")
1135  .arg(poly->getName()));
1136  return CC_FERR_BAD_ENTITY_TYPE;
1137  }
1138  break;
1139  default:
1140  assert(false);
1141  return CC_FERR_BAD_ENTITY_TYPE;
1142  }
1143 
1144  bool isClosed = poly->isClosed();
1145 
1146  if (static_cast<int64_t>(realVertexCount) + 1 >
1147  std::numeric_limits<int32_t>::max()) {
1148  CVLog::Warning(
1149  QObject::tr("[SHP] Polyline %1 has too many points to be saved")
1150  .arg(poly->getName()));
1151  return CC_FERR_BAD_ENTITY_TYPE;
1152  }
1153 
1154  int32_t iRealVertexCount = static_cast<int32_t>(realVertexCount);
1155  int32_t numPoints = iRealVertexCount + (isClosed ? 1 : 0);
1156  const int32_t numParts = 1;
1157 
1158  recordSize = sizeofPolyLine(outputShapeType, numPoints, numParts);
1159  out.setByteOrder(QDataStream::BigEndian);
1160  out << recordNumber << recordSize;
1161 
1162  qint64 recordStart = out.device()->pos();
1163  out.setByteOrder(QDataStream::LittleEndian);
1164 
1165  // Byte 0: Shape Type
1166  out << static_cast<int32_t>(outputShapeType);
1167 
1168  // Byte 4: Box
1169  ccHObject::GlobalBoundingBox globalBB = poly->getOwnGlobalBB();
1170  // The Bounding Box for the PolyLine stored in the order Xmin, Ymin, Xmax,
1171  // Ymax (24 bytes)
1172  out << globalBB.minCorner().u[X] << globalBB.minCorner().u[Y]
1173  << globalBB.maxCorner().u[X] << globalBB.maxCorner().u[Y];
1174 
1175  // Byte 36: NumParts (The number of parts in the PolyLine)
1176  out << numParts;
1177  // Byte 40: NumPoints (The total number of points for all parts)
1178  out << numPoints;
1179 
1180  // Byte 44: Parts (An array of length NumParts)
1181  //(for each part, the index of its first point in the points array)
1182  assert(numParts == 1);
1183  out << static_cast<int32_t>(0);
1184 
1185  // for polygons we must list the vertices in the right order:
1186  //"The neighborhood to the right of an observer walking along
1187  // the ring in vertex order is the inside of the polygon"
1188  //== clockwise order
1189  bool inverseOrder = false;
1190  if (outputShapeType == ESRI_SHAPE_TYPE::POLYGON ||
1191  outputShapeType == ESRI_SHAPE_TYPE::POLYGON_Z) {
1192  assert(isClosed);
1193  assert(numPoints > 2);
1194 
1195  unsigned char dim1 = X;
1196  unsigned char dim2 = Y;
1197  if (outputShapeType == ESRI_SHAPE_TYPE::POLYGON_Z) {
1198  CCVector3d diag = globalBB.getDiagVec();
1199 
1200  // in 3D we have to guess the 'flat' dimension
1201  unsigned char minDim = diag.u[1] < diag.u[0] ? 1 : 0;
1202  if (diag.u[2] < diag.u[minDim]) minDim = 2;
1203  dim1 = minDim == 2 ? 0 : minDim + 1;
1204  dim2 = dim1 == 2 ? 0 : dim1 + 1;
1205  }
1206 
1207  inverseOrder =
1208  areVerticesCounterClockwise(vertices, numPoints, dim1, dim2);
1209  }
1210 
1211  // Points (An array of length NumPoints)
1212  for (int32_t i = 0; i < numPoints; ++i) {
1213  int32_t ii = (inverseOrder ? numPoints - 1 - i : i);
1214  const CCVector3 *P = vertices->getPoint(
1215  ii % iRealVertexCount); // warning: handle loop if polyline is
1216  // closed
1217  CCVector3d Pg = poly->toGlobal3d(*P);
1218  // 2D point (16 bytes)
1219  out << Pg.u[X] << Pg.u[Y];
1220  }
1221 
1222  // 3D polylines
1223  if (isESRIShape3D(outputShapeType)) {
1224  // Z boundaries (16 bytes)
1225  out << globalBB.minCorner().u[Z] << globalBB.maxCorner().u[Z];
1226 
1227  // Z coordinates (for each part - just one here)
1228  for (int32_t i = 0; i < numPoints; ++i) {
1229  int32_t ii = (inverseOrder ? numPoints - 1 - i : i);
1230  const CCVector3 *P = vertices->getPoint(
1231  ii % iRealVertexCount); // warning: handle loop if polyline
1232  // is closed
1233  CCVector3d Pg = poly->toGlobal3d(*P);
1234  out << Pg.u[Z];
1235  }
1236  }
1237 
1238  if (hasMeasurements(outputShapeType)) {
1239  // M boundaries (16 bytes)
1240  bool hasSF = vertices->isScalarFieldEnabled();
1241  CCVector2d minMax = minMaxOfEnabledScalarField(vertices);
1242  out << minMax.x << minMax.y;
1243 
1244  // M values (for each part - just one here)
1245  if (hasSF) {
1246  for (int32_t i = 0; i < numPoints; ++i) {
1247  ScalarType scalar =
1248  vertices->getPointScalarValue(i % iRealVertexCount);
1249  out << (ccScalarField::ValidValue(scalar)
1250  ? ESRI_NO_DATA
1251  : static_cast<double>(scalar));
1252  }
1253  } else {
1254  for (int32_t i = 0; i < numPoints; ++i) {
1255  out << ESRI_NO_DATA;
1256  }
1257  }
1258  }
1259 
1260  assert(out.device()->pos() == recordStart + recordSize * 2);
1261  return CC_FERR_NO_ERROR;
1262 }

1797  ccPointCloud *singlePoints = nullptr;

1799  QMap<ccPolyline *, int32_t> polyIDs;
1800  int32_t maxPolyID = 0;

1803  while (hdr.fileLength - shpStream.device()->pos() > 0) {
1804  if (shpStream.status() != QDataStream::Ok) {
1805  CVLog::Warning("[SHP] Something went wrong reading the file");

1813  recordSize *= 2; // recordSize is measured in 16-bit words
1814  shpStream.setByteOrder(QDataStream::LittleEndian);

1827  .arg(recordNumber)
1828  .arg(ToString(shapeType))
1829  .arg(recordSize));

1836  case ESRI_SHAPE_TYPE::POLYLINE_Z:
1837  case ESRI_SHAPE_TYPE::POLYGON_Z:

1839  case ESRI_SHAPE_TYPE::POLYLINE:

1842  case ESRI_SHAPE_TYPE::POLYGON_M: {

1845  shapeType, Pshift,

1849  unsigned childCountAfter = container.getChildrenNumber();
1850  // warning: we can load mutliple polylines for a single
1851  // record!
1852  for (unsigned i = childCountBefore; i < childCountAfter;
1853  ++i) {

1855  assert(child && child->isA(CV_TYPES::POLY_LINE));

1863  case ESRI_SHAPE_TYPE::MULTI_POINT_M:
1864  is3DShape = true;
1865  case ESRI_SHAPE_TYPE::MULTI_POINT:

1876  maxPointID = recordNumber;
1877  }
