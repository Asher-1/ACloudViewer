52 CC_FILE_ERROR STLFilter::saveToFile(ccHObject* entity,
53  const QString& filename,
54  const SaveParameters& parameters) {
55  if (!entity) return CC_FERR_BAD_ARGUMENT;
56 
57  if (!entity->isKindOf(CV_TYPES::MESH)) return CC_FERR_BAD_ENTITY_TYPE;
58 
59  ccGenericMesh* mesh = ccHObjectCaster::ToGenericMesh(entity);
60  if (!mesh || mesh->size() == 0) {
61  CVLog::Warning(QString("[STL] No facet in mesh '%1'!")
62  .arg(mesh ? mesh->getName()
63  : QStringLiteral("unnamed")));
64  return CC_FERR_NO_ERROR;
65  }
66 
67  // ask for output format
68  bool binaryMode = true;
69  if (parameters.alwaysDisplaySaveDialog) {
70  QMessageBox msgBox(QMessageBox::Question, "Choose output format",
71  "Save in BINARY or ASCII format?");
72  QPushButton* binaryButton =
73  msgBox.addButton("BINARY", QMessageBox::AcceptRole);
74  msgBox.addButton("ASCII", QMessageBox::AcceptRole);
75  msgBox.exec();
76  binaryMode = (msgBox.clickedButton() == binaryButton);
77  }
78 
79  // try to open file for saving
80  FILE* theFile = fopen(qPrintable(filename), "wb");
81  if (!theFile) return CC_FERR_WRITING;
82 
83  CC_FILE_ERROR result = CC_FERR_NO_ERROR;
84  if (binaryMode) {
85  result = saveToBINFile(mesh, theFile);
86  } else // if (msgBox.clickedButton() == asciiButton)
87  {
88  result = saveToASCIIFile(mesh, theFile);
89  }
90 
91  fclose(theFile);
92 
93  return result;
94 }

96 CC_FILE_ERROR STLFilter::saveToBINFile(ccGenericMesh* mesh,
97  FILE* theFile,
98  QWidget* parentWidget /*=0*/) {
99  assert(theFile && mesh && mesh->size() != 0);
100  unsigned faceCount = mesh->size();
101 
102  // progress
103  QScopedPointer<ecvProgressDialog> pDlg(nullptr);
104  if (parentWidget) {
105  pDlg.reset(new ecvProgressDialog(true, parentWidget));
106  pDlg->setMethodTitle(
107  QObject::tr("Saving mesh [%1]").arg(mesh->getName()));
108  pDlg->setInfo(QObject::tr("Number of facets: %1").arg(faceCount));
109  pDlg->start();
110  QApplication::processEvents();
111  }
112  cloudViewer::NormalizedProgress nprogress(pDlg.data(), faceCount);
113 
114  // header
115  {
116  char header[80];
117  memset(header, 0, 80);
118  strcpy(header, "Binary STL file generated by ACloudViewer!");
119  if (fwrite(header, 80, 1, theFile) < 1) return CC_FERR_WRITING;
120  }
121 
122  // UINT32 Number of triangles
123  {
124  uint32_t tmpInt32 = static_cast<uint32_t>(faceCount);
125  if (fwrite((const void*)&tmpInt32, 4, 1, theFile) < 1)
126  return CC_FERR_WRITING;
127  }
128 
129  ccGenericPointCloud* vertices = mesh->getAssociatedCloud();
130  assert(vertices);
131 
132  // Can't save global shift information....
133  if (vertices->isShifted()) {
134  CVLog::Warning(
135  "[STL] Global shift information can't be restored in STL "
136  "Binary format! (too low precision)");
137  }
138 
139  mesh->placeIteratorAtBeginning();
140  for (unsigned i = 0; i < faceCount; ++i) {
141  cloudViewer::VerticesIndexes* tsi = mesh->getNextTriangleVertIndexes();
142 
143  const CCVector3* A = vertices->getPointPersistentPtr(tsi->i1);
144  const CCVector3* B = vertices->getPointPersistentPtr(tsi->i2);
145  const CCVector3* C = vertices->getPointPersistentPtr(tsi->i3);
146  // compute face normal (right hand rule)
147  CCVector3 N = (*B - *A).cross(*C - *A);
148 
149  // REAL32[3] Normal vector
150  CCVector3f buffer = CCVector3f::fromArray(
151  N.u); // convert to an explicit float array (as
152  // PointCoordinateType may be a double!)
153  assert(sizeof(float) == 4);
154  if (fwrite((const void*)buffer.u, 4, 3, theFile) < 3)
155  return CC_FERR_WRITING;
156 
157  // REAL32[3] Vertex 1,2 & 3
158  buffer = CCVector3f::fromArray(
159  A->u); // convert to an explicit float array (as
160  // PointCoordinateType may be a double!)
161  if (fwrite((const void*)buffer.u, 4, 3, theFile) < 3)
162  return CC_FERR_WRITING;
163  buffer = CCVector3f::fromArray(
164  B->u); // convert to an explicit float array (as
165  // PointCoordinateType may be a double!)
166  if (fwrite((const void*)buffer.u, 4, 3, theFile) < 3)
167  return CC_FERR_WRITING;
168  buffer = CCVector3f::fromArray(
169  C->u); // convert to an explicit float array (as
170  // PointCoordinateType may be a double!)
171  if (fwrite((const void*)buffer.u, 4, 3, theFile) < 3)
172  return CC_FERR_WRITING;
173 
174  // UINT16 Attribute byte count (not used)
175  {
176  char byteCount[2] = {0, 0};
177  if (fwrite(byteCount, 2, 1, theFile) < 1) return CC_FERR_WRITING;
178  }
179 
180  // progress
181  if (pDlg && !nprogress.oneStep()) {
182  return CC_FERR_CANCELED_BY_USER;
183  }
184  }
185 
186  if (pDlg) {
187  pDlg->stop();
188  }
189 
190  return CC_FERR_NO_ERROR;
191 }

193 CC_FILE_ERROR STLFilter::saveToASCIIFile(ccGenericMesh* mesh,
194  FILE* theFile,
195  QWidget* parentWidget /*=0*/) {
196  assert(theFile && mesh && mesh->size() != 0);
197  unsigned faceCount = mesh->size();
198 
199  // progress
200  QScopedPointer<ecvProgressDialog> pDlg(nullptr);
201  if (parentWidget) {
202  pDlg.reset(new ecvProgressDialog(true, parentWidget));
203  pDlg->setMethodTitle(
204  QObject::tr("Saving mesh [%1]").arg(mesh->getName()));
205  pDlg->setInfo(QObject::tr("Number of facets: %1").arg(faceCount));
206  pDlg->start();
207  QApplication::processEvents();
208  }
209  cloudViewer::NormalizedProgress nprogress(pDlg.data(), faceCount);
210 
211  if (fprintf(theFile, "solid %s\n", qPrintable(mesh->getName())) <
212  0) // empty names are acceptable!
213  {
214  return CC_FERR_WRITING;
215  }
216 
217  // vertices
218  ccGenericPointCloud* vertices = mesh->getAssociatedCloud();
219 
220  mesh->placeIteratorAtBeginning();
221  for (unsigned i = 0; i < faceCount; ++i) {
222  cloudViewer::VerticesIndexes* tsi = mesh->getNextTriangleVertIndexes();
223 
224  const CCVector3* A = vertices->getPointPersistentPtr(tsi->i1);
225  const CCVector3* B = vertices->getPointPersistentPtr(tsi->i2);
226  const CCVector3* C = vertices->getPointPersistentPtr(tsi->i3);
227  // compute face normal (right hand rule)
228  CCVector3 N = (*B - *A).cross(*C - *A);
229 
230  //%e = scientific notation
231  if (fprintf(theFile, "facet normal %e %e %e\n", N.x, N.y, N.z) < 0)
232  return CC_FERR_WRITING;
233  if (fprintf(theFile, "outer loop\n") < 0) return CC_FERR_WRITING;
234 
235  CCVector3d Aglobal = vertices->toGlobal3d<PointCoordinateType>(*A);
236  if (fprintf(theFile, "vertex %e %e %e\n", Aglobal.x, Aglobal.y,
237  Aglobal.z) < 0)
238  return CC_FERR_WRITING;
239  CCVector3d Bglobal = vertices->toGlobal3d<PointCoordinateType>(*B);
240  if (fprintf(theFile, "vertex %e %e %e\n", Bglobal.x, Bglobal.y,
241  Bglobal.z) < 0)
242  return CC_FERR_WRITING;
243  CCVector3d Cglobal = vertices->toGlobal3d<PointCoordinateType>(*C);
244  if (fprintf(theFile, "vertex %e %e %e\n", Cglobal.x, Cglobal.y,
245  Cglobal.z) < 0)
246  return CC_FERR_WRITING;
247  if (fprintf(theFile, "endloop\nendfacet\n") < 0) return CC_FERR_WRITING;
248 
249  // progress
250  if (pDlg && !nprogress.oneStep()) {
251  return CC_FERR_CANCELED_BY_USER;
252  }
253  }
254 
255  if (fprintf(theFile, "endsolid %s\n", qPrintable(mesh->getName())) <
256  0) // empty names are acceptable!
257  {
258  return CC_FERR_WRITING;
259  }
260 
261  return CC_FERR_NO_ERROR;
262 }

264 CC_FILE_ERROR STLFilter::loadFile(const QString& filename,
265  ccHObject& container,
266  LoadParameters& parameters) {
267  CVLog::Print(QString("[STL] Loading '%1'").arg(filename));
268 
269  // try to open the file
270  QFile fp(filename);
271  if (!fp.open(QIODevice::ReadOnly)) return CC_FERR_READING;
272 
273  // ASCII OR BINARY?
274  QString name("mesh");
275  bool ascii = true;
276  {
277  // buffer
278  char header[80] = {0};
279  qint64 sz = fp.read(header, 80);
280  if (sz < 80) {
281  // either ASCII or BINARY STL FILES are always > 80 bytes
282  return sz == 0 ? CC_FERR_READING : CC_FERR_MALFORMED_FILE;
283  }
284  // normally, binary files shouldn't start by 'solid'
285  if (!QString(header).trimmed().toUpper().startsWith("SOLID")) {
286  ascii = false;
287  } else //... but sadly some BINARY files does start by SOLID?!!!! (wtf)
288  {
289  // go back to the beginning of the file
290  fp.seek(0);
291 
292  QTextStream stream(&fp);
293  // skip first line
294  stream.readLine();
295  // we look if the second line (if any) starts by 'facet'
296  QString line = stream.readLine();
297  ascii = true;
298  if (line.isEmpty() || fp.error() != QFile::NoError ||
299  !QString(line).trimmed().toUpper().startsWith("FACET")) {
300  ascii = false;
301  }
302  }
303  // go back to the beginning of the file
304  fp.seek(0);
305  }
306  CVLog::Print("[STL] Detected format: %s", ascii ? "ASCII" : "BINARY");
307 
308  // vertices
309  ccPointCloud* vertices = new ccPointCloud("vertices");
310  // mesh
311  ccMesh* mesh = new ccMesh(vertices);
312  mesh->setName(name);
313  // add normals
314  mesh->setTriNormsTable(new NormsIndexesTableType());
315 
316  CC_FILE_ERROR error = CC_FERR_NO_ERROR;
317  if (ascii)
318  error = loadASCIIFile(fp, mesh, vertices, parameters);
319  else
320  error = loadBinaryFile(fp, mesh, vertices, parameters);
321 
322  if (error != CC_FERR_NO_ERROR) {
323  return CC_FERR_MALFORMED_FILE;
324  }
325 
326  unsigned vertCount = vertices->size();
327  unsigned faceCount = mesh->size();
328  CVLog::Print("[STL] %i points, %i face(s)", vertCount, faceCount);
329 
330  // do some cleaning
331  {
332  vertices->shrinkToFit();
333  mesh->shrinkToFit();
334  NormsIndexesTableType* normals = mesh->getTriNormsTable();
335  if (normals) {
336  normals->shrink_to_fit();
337  }
338  }
339 
340  // remove duplicated vertices
341  mesh->mergeDuplicatedVertices(ccMesh::DefaultMergeDulicateVerticesLevel,
342  parameters.parentWidget);
343  vertices = nullptr; // warning, after this point, 'vertices' is not valid
344  // anymore
345 
346  NormsIndexesTableType* normals = mesh->getTriNormsTable();
347  if (normals) {
348  // normals->link();
349  // mesh->addChild(normals); //automatically done by setTriNormsTable
350  mesh->showNormals(true);
351  } else {
352  // DGM: normals can be per-vertex or per-triangle so it's better to let
353  // the user do it himself later Moreover it's not always good idea if
354  // the user doesn't want normals (especially in ccViewer!) if
355  // (mesh->computeNormals()) mesh->showNormals(true); else
356  // CVLog::Warning("[STL] Failed to compute per-vertex normals...");
357  CVLog::Warning(
358  "[STL] Mesh has no normal! You can manually compute them "
359  "(select it then call \"Edit > Normals > Compute\")");
360  }
361  ccGenericPointCloud* meshVertices = mesh->getAssociatedCloud();
362  meshVertices->setEnabled(false);
363  meshVertices->setLocked(
364  false); // DGM: no need to lock it as it is only used by one mesh!
365  mesh->addChild(meshVertices);
366 
367  container.addChild(mesh);
368 
369  return CC_FERR_NO_ERROR;
370 }

372 CC_FILE_ERROR STLFilter::loadASCIIFile(QFile& fp,
373  ccMesh* mesh,
374  ccPointCloud* vertices,
375  LoadParameters& parameters) {
376  assert(fp.isOpen() && mesh && vertices);
377 
378  // text stream
379  QTextStream stream(&fp);
380 
381  // 1st line: 'solid name'
382  QString name("mesh");
383  {
384  QString currentLine = stream.readLine();
385  if (currentLine.isEmpty() || fp.error() != QFile::NoError) {
386  return CC_FERR_READING;
387  }
388  QStringList tokens = currentLine.simplified().split(
389  QChar(' '), QtCompat::SkipEmptyParts);
390  if (tokens.empty() || tokens[0].toUpper() != "SOLID") {
391  CVLog::Warning("[STL] File should begin by 'solid [name]'!");
392  return CC_FERR_MALFORMED_FILE;
393  }
394  // Extract name
395  if (tokens.size() > 1) {
396  tokens.removeAt(0);
397  name = tokens.join(" ");
398  }
399  }
400  mesh->setName(name);
401 
402  // progress dialog
403  QScopedPointer<ecvProgressDialog> pDlg(nullptr);
404  if (parameters.parentWidget) {
405  pDlg.reset(new ecvProgressDialog(true, parameters.parentWidget));
406  pDlg->setMethodTitle(QObject::tr("(ASCII) STL file"));
407  pDlg->setInfo(QObject::tr("Loading in progress..."));
408  pDlg->setRange(0, 0);
409  pDlg->start();
410  QApplication::processEvents();
411  }
412 
413  // current vertex shift
414  CCVector3d Pshift(0, 0, 0);
415 
416  unsigned pointCount = 0;
417  unsigned faceCount = 0;
418  static const unsigned s_defaultMemAllocCount = 65536;
419  bool normalWarningAlreadyDisplayed = false;
420  NormsIndexesTableType* normals = mesh->getTriNormsTable();
421 
422  CC_FILE_ERROR result = CC_FERR_NO_ERROR;
423 
424  unsigned lineCount = 1;
425  while (true) {
426  CCVector3 N;
427  bool normalIsOk = false;
428 
429  // 1st line of a 'facet': "facet normal ni nj nk" / or 'endsolid' (i.e.
430  // end of file)
431  {
432  QString currentLine = stream.readLine();
433  if (currentLine.isEmpty()) {
434  break;
435  } else if (fp.error() != QFile::NoError) {
436  result = CC_FERR_READING;
437  break;
438  }
439  ++lineCount;
440 
441  QStringList tokens = currentLine.simplified().split(
442  QChar(' '), QtCompat::SkipEmptyParts);
443  if (tokens.empty() || tokens[0].toUpper() != "FACET") {
444  if (tokens[0].toUpper() != "ENDSOLID") {
445  CVLog::Warning(
446  "[STL] Error on line #%i: line should start by "
447  "'facet'!",
448  lineCount);
449  return CC_FERR_MALFORMED_FILE;
450  }
451  break;
452  }
453 
454  if (normals && tokens.size() >= 5) {
455  // let's try to read normal
456  if (tokens[1].toUpper() == "NORMAL") {
457  N.x = static_cast<PointCoordinateType>(
458  tokens[2].toDouble(&normalIsOk));
459  if (normalIsOk) {
460  N.y = static_cast<PointCoordinateType>(
461  tokens[3].toDouble(&normalIsOk));
462  if (normalIsOk) {
463  N.z = static_cast<PointCoordinateType>(
464  tokens[4].toDouble(&normalIsOk));
465  }
466  }
467  if (!normalIsOk && !normalWarningAlreadyDisplayed) {
468  CVLog::Warning(
469  "[STL] Error on line #%i: failed to read "
470  "'normal' values!",
471  lineCount);
472  normalWarningAlreadyDisplayed = true;
473  }
474  } else if (!normalWarningAlreadyDisplayed) {
475  CVLog::Warning(
476  "[STL] Error on line #%i: expecting 'normal' after "
477  "'facet'!",
478  lineCount);
479  normalWarningAlreadyDisplayed = true;
480  }
481  } else if (tokens.size() > 1 && !normalWarningAlreadyDisplayed) {
482  CVLog::Warning(
483  "[STL] Error on line #%i: incomplete 'normal' "
484  "description!",
485  lineCount);
486  normalWarningAlreadyDisplayed = true;
487  }
488  }
489 
490  // 2nd line: 'outer loop'
491  {
492  QString currentLine = stream.readLine();
493  if (currentLine.isEmpty() || fp.error() != QFile::NoError ||
494  !QString(currentLine)
495  .trimmed()
496  .toUpper()
497  .startsWith("OUTER LOOP")) {
498  CVLog::Warning(
499  "[STL] Error: expecting 'outer loop' on line #%i",
500  lineCount + 1);
501  result = CC_FERR_READING;
502  break;
503  }
504  ++lineCount;
505  }
506 
507  // 3rd to 5th lines: 'vertex vix viy viz'
508  unsigned vertIndexes[3];
509  // unsigned pointCountBefore = pointCount;
510  for (unsigned i = 0; i < 3; ++i) {
511  QString currentLine = stream.readLine();
512  if (currentLine.isEmpty() || fp.error() != QFile::NoError ||
513  !QString(currentLine)
514  .trimmed()
515  .toUpper()
516  .startsWith("VERTEX")) {
517  CVLog::Warning(
518  "[STL] Error: expecting a line starting by 'vertex' on "
519  "line #%i",
520  lineCount + 1);
521  result = CC_FERR_MALFORMED_FILE;
522  break;
523  }
524  ++lineCount;
525 
526  QStringList tokens =
527  QString(currentLine)
528  .simplified()
529  .split(QChar(' '), QtCompat::SkipEmptyParts);
530  if (tokens.size() < 4) {
531  CVLog::Warning(
532  "[STL] Error on line #%i: incomplete 'vertex' "
533  "description!",
534  lineCount);
535  result = CC_FERR_MALFORMED_FILE;
536  break;
537  }
538 
539  // read vertex
540  CCVector3d Pd(0, 0, 0);
541  {
542  bool vertexIsOk = false;
543  Pd.x = tokens[1].toDouble(&vertexIsOk);
544  if (vertexIsOk) {
545  Pd.y = tokens[2].toDouble(&vertexIsOk);
546  if (vertexIsOk) Pd.z = tokens[3].toDouble(&vertexIsOk);
547  }
548  if (!vertexIsOk) {
549  CVLog::Warning(
550  "[STL] Error on line #%i: failed to read 'vertex' "
551  "coordinates!",
552  lineCount);
553  result = CC_FERR_MALFORMED_FILE;
554  break;
555  }
556  }
557 
558  // first point: check for 'big' coordinates
559  if (pointCount == 0) {
560  bool preserveCoordinateShift = true;
561  if (HandleGlobalShift(Pd, Pshift, preserveCoordinateShift,
562  parameters)) {
563  if (preserveCoordinateShift) {
564  vertices->setGlobalShift(Pshift);
565  }
566  CVLog::Warning(
567  "[STLFilter::loadFile] Cloud has been recentered! "
568  "Translation: (%.2f ; %.2f ; %.2f)",
569  Pshift.x, Pshift.y, Pshift.z);
570  }
571  }
572 
573  CCVector3 P = CCVector3::fromArray((Pd + Pshift).u);
574 
575  // cloud is already full?
576  if (vertices->capacity() == pointCount &&
577  !vertices->reserve(pointCount + s_defaultMemAllocCount))
578  return CC_FERR_NOT_ENOUGH_MEMORY;
579 
580  // insert new point
581  vertIndexes[i] = pointCount++;
582  vertices->addPoint(P);
583  }
584 
585  // we have successfully read the 3 vertices
586  // let's add a new triangle
587  {
588  // mesh is full?
589  if (mesh->capacity() == faceCount) {
590  if (!mesh->reserve(faceCount + s_defaultMemAllocCount)) {
591  result = CC_FERR_NOT_ENOUGH_MEMORY;
592  break;
593  }
594 
595  if (normals) {
596  bool success = normals->reserveSafe(mesh->capacity());
597  if (success &&
598  faceCount == 0) // specific case: allocate per triangle
599  // normal indexes the first time!
600  {
601  success = mesh->reservePerTriangleNormalIndexes();
602  }
603 
604  if (!success) {
605  CVLog::Warning(
606  "[STL] Not enough memory: can't store "
607  "normals!");
608  mesh->removePerTriangleNormalIndexes();
609  mesh->setTriNormsTable(nullptr);
610  normals->release();
611  normals = nullptr;
612  }
613  }
614  }
615 
616  mesh->addTriangle(vertIndexes[0], vertIndexes[1], vertIndexes[2]);
617  ++faceCount;
618  }
619 
620  // and a new normal?
621  if (normals) {
622  int index = -1;
623  if (normalIsOk) {
624  // compress normal
625  index = static_cast<int>(normals->currentSize());
626  CompressedNormType nIndex = ccNormalVectors::GetNormIndex(N.u);
627  normals->addElement(nIndex);
628  }
629  mesh->addTriangleNormalIndexes(index, index, index);
630  }
631 
632  // 6th line: 'endloop'
633  {
634  QString currentLine = stream.readLine();
635  if (currentLine.isEmpty() || fp.error() != QFile::NoError ||
636  !QString(currentLine)
637  .trimmed()
638  .toUpper()
639  .startsWith("ENDLOOP")) {
640  CVLog::Warning("[STL] Error: expecting 'endnloop' on line #%i",
641  lineCount + 1);
642  result = CC_FERR_MALFORMED_FILE;
643  break;
644  }
645  ++lineCount;
646  }
647 
648  // 7th and last line: 'endfacet'
649  {
650  QString currentLine = stream.readLine();
651  if (currentLine.isEmpty() || fp.error() != QFile::NoError ||
652  !QString(currentLine)
653  .trimmed()
654  .toUpper()
655  .startsWith("ENDFACET")) {
656  CVLog::Warning("[STL] Error: expecting 'endfacet' on line #%i",
657  lineCount + 1);
658  result = CC_FERR_MALFORMED_FILE;
659  break;
660  }
661  ++lineCount;
662  }
663 
664  // progress
665  if (pDlg && (faceCount % 1024) == 0) {
666  if (pDlg->wasCanceled()) break;
667  pDlg->setValue(static_cast<int>(faceCount >> 10));
668  }
669  }
670 
671  if (normalWarningAlreadyDisplayed) {
672  CVLog::Warning("[STL] Failed to read some 'normal' values!");
673  }
674 
675  if (pDlg) {
676  pDlg->close();

682 CC_FILE_ERROR STLFilter::loadBinaryFile(QFile& fp,
683  ccMesh* mesh,
684  ccPointCloud* vertices,
685  LoadParameters& parameters) {
686  assert(fp.isOpen() && mesh && vertices);
687 
688  unsigned pointCount = 0;
689  unsigned faceCount = 0;
690 
691  // UINT8[80] Header (we skip it)
692  fp.seek(80);
693  mesh->setName("Mesh"); // hard to guess solid name with binary files!
694 
695  // UINT32 Number of triangles
696  {
697  unsigned tmpInt32;
698  if (fp.read((char*)&tmpInt32, 4) < 4) return CC_FERR_READING;
699  faceCount = tmpInt32;
700  }
701 
702  if (!mesh->reserve(faceCount)) return CC_FERR_NOT_ENOUGH_MEMORY;
703  if (!vertices->reserve(3 * faceCount)) return CC_FERR_NOT_ENOUGH_MEMORY;
704  NormsIndexesTableType* normals = mesh->getTriNormsTable();
705  if (normals && (!normals->reserveSafe(faceCount) ||
706  !mesh->reservePerTriangleNormalIndexes())) {
707  CVLog::Warning("[STL] Not enough memory: can't store normals!");
708  mesh->removePerTriangleNormalIndexes();
709  mesh->setTriNormsTable(nullptr);
710  }
711 
712  // progress dialog
713  QScopedPointer<ecvProgressDialog> pDlg(nullptr);
714  if (parameters.parentWidget) {
715  pDlg.reset(new ecvProgressDialog(true, parameters.parentWidget));
716  pDlg->setMethodTitle(QObject::tr("Loading binary STL file"));
717  pDlg->setInfo(QObject::tr("Loading %1 faces").arg(faceCount));
718  pDlg->start();
719  QApplication::processEvents();
720  }
721  cloudViewer::NormalizedProgress nProgress(pDlg.data(), faceCount);
722 
723  // current vertex shift
724  CCVector3d Pshift(0, 0, 0);
725 
726  for (unsigned f = 0; f < faceCount; ++f) {
727  // REAL32[3] Normal vector
728  assert(sizeof(float) == 4);
729  CCVector3 N;
730  if (fp.read((char*)N.u, 12) < 12) return CC_FERR_READING;
731 
732  // 3 vertices
733  unsigned vertIndexes[3];
734  for (unsigned i = 0; i < 3; ++i) {
735  // REAL32[3] Vertex 1,2 & 3
736  float Pf[3];
737  if (fp.read((char*)Pf, 12) < 0) return CC_FERR_READING;
738 
739  // first point: check for 'big' coordinates
740  CCVector3d Pd(Pf[0], Pf[1], Pf[2]);
741  if (pointCount == 0) {
742  bool preserveCoordinateShift = true;
743  if (HandleGlobalShift(Pd, Pshift, preserveCoordinateShift,
744  parameters)) {
745  if (preserveCoordinateShift) {
746  vertices->setGlobalShift(Pshift);
747  }
748  CVLog::Warning(
749  "[STLFilter::loadFile] Cloud has been recentered! "
750  "Translation: (%.2f ; %.2f ; %.2f)",
751  Pshift.x, Pshift.y, Pshift.z);
752  }
753  }
754 
755  CCVector3 P = CCVector3::fromArray((Pd + Pshift).u);
756 
757  // insert new point
758  vertIndexes[i] = pointCount++;
759  vertices->addPoint(P);
760  }
761 
762  // UINT16 Attribute byte count (not used)
763  {
764  char a[2];
765  if (fp.read(a, 2) < 0) return CC_FERR_READING;
766  }
767 
768  // we have successfully read the 3 vertices
769  // let's add a new triangle
770  { mesh->addTriangle(vertIndexes[0], vertIndexes[1], vertIndexes[2]); }
771 
772  // and a new normal?
773  if (normals) {
774  // compress normal
775  int index = static_cast<int>(normals->currentSize());
776  CompressedNormType nIndex = ccNormalVectors::GetNormIndex(N.u);
777  normals->addElement(nIndex);
778  mesh->addTriangleNormalIndexes(index, index, index);
779  }
780 
781  // progress
782  if (pDlg && !nProgress.oneStep()) {
783  break;
784  }
785  }
786 
787  if (pDlg) {
788  pDlg->stop();
789  }
790 
791  return CC_FERR_NO_ERROR;
792 }
