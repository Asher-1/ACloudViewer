70 void PclFiltersTool::initialize(ecvGenericVisualizer3D* viewer) {
71  assert(viewer);
72  this->setVisualizer(viewer);
73 
74  switch (m_filterType) {
75  case ecvGenericFiltersTool::CLIP_FILTER:
76  m_filter = new cvClipFilter();
77  break;
78  case ecvGenericFiltersTool::SLICE_FILTER:
79  m_filter = new cvSliceFilter();
80  break;
81  case ecvGenericFiltersTool::DECIMATE_FILTER:
82  m_filter = new cvDecimateFilter();
83  break;
84  case ecvGenericFiltersTool::ISOSURFACE_FILTER:
85  m_filter = new cvIsoSurfaceFilter();
86  break;
87  case ecvGenericFiltersTool::THRESHOLD_FILTER:
88  m_filter = new cvThresholdFilter();
89  break;
90  case ecvGenericFiltersTool::SMOOTH_FILTER:
91  m_filter = new cvSmoothFilter();
92  break;
93  case ecvGenericFiltersTool::PROBE_FILTER:
94  m_filter = new cvProbeFilter();
95  break;
96  case ecvGenericFiltersTool::STREAMLINE_FILTER:
97  m_filter = new cvStreamlineFilter();
98  break;
99  case ecvGenericFiltersTool::GLYPH_FILTER:
100  m_filter = new cvGlyphFilter();
101  break;
102  default:
103  CVLog::Error(QString("unknown filter type"));
104  break;
105  }
106 
107  if (m_filter && m_viewer) {
108  m_filter->setUpViewer(m_viewer);
109  }
110 
111  resetMode();
112  m_cloudLabel = nullptr;
113 }

229 void PclFiltersTool::areaPickingEventProcess(
230  const std::vector<int>& new_selected_slice) {
231  if (new_selected_slice.empty() || !m_viewer) return;
232 
233  int s = m_viewer->getRenderWindowInteractor()->GetShiftKey();
234  int a = m_viewer->getRenderWindowInteractor()->GetControlKey();
235 
236  // remove ground points
237  std::vector<int> selected_slice;
238  for (auto x : new_selected_slice) {
239  if (m_cloudLabel[x] != GROUND_POINT) {
240  selected_slice.push_back(x);
241  }
242  }
243 
244  if ((s && a) || m_intersectMode) { // intersection
245  m_last_selected_slice = ecvTools::IntersectionVector(
246  m_last_selected_slice, selected_slice);
247  } else if (s || m_unionMode) { // union
248  m_last_selected_slice =
249  ecvTools::UnionVector(m_last_selected_slice, selected_slice);
250  } else if (a || m_trimMode) { // remove
251  m_last_selected_slice =
252  ecvTools::DiffVector(m_last_selected_slice, selected_slice);
253  } else { // new
254  m_last_selected_slice = selected_slice;
255  }
256 
257  update();
258 }
