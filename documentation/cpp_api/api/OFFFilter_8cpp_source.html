46 CC_FILE_ERROR OFFFilter::saveToFile(ccHObject* entity,
47  const QString& filename,
48  const SaveParameters& parameters) {
49  if (!entity) return CC_FERR_BAD_ARGUMENT;
50 
51  if (!entity->isKindOf(CV_TYPES::MESH)) {
52  CVLog::Warning("[OFF] This filter can only save one mesh at a time!");
53  return CC_FERR_BAD_ENTITY_TYPE;
54  }
55 
56  ccGenericMesh* mesh = ccHObjectCaster::ToGenericMesh(entity);
57  if (!mesh || mesh->size() == 0) {
58  CVLog::Warning("[OFF] Input mesh is empty!");
59  return CC_FERR_NO_SAVE;
60  }
61  ccGenericPointCloud* vertices = mesh->getAssociatedCloud();
62  if (!vertices || vertices->size() == 0) {
63  CVLog::Warning("[OFF] Input mesh has no vertices?!");
64  return CC_FERR_NO_SAVE;
65  }
66 
67  // try to open file for saving
68  QFile fp(filename);
69  if (!fp.open(QIODevice::WriteOnly | QIODevice::Text))
70  return CC_FERR_WRITING;
71 
72  QTextStream stream(&fp);
73  stream.setRealNumberNotation(QTextStream::FixedNotation);
74  stream.setRealNumberPrecision(12); // TODO: ask the user?
75 
76  // header: "OFF"
77  stream << "OFF" << QtCompat::endl;
78 
79  // 2nd line: vertices count / faces count / edges count
80  unsigned vertCount = vertices->size();
81  unsigned triCount = mesh->size();
82  stream << vertCount << ' ' << triCount << ' ' << 0 << QtCompat::endl;
83 
84  // save vertices
85  {
86  for (unsigned i = 0; i < vertCount; ++i) {
87  const CCVector3* P = vertices->getPoint(i);
88  CCVector3d Pglobal = vertices->toGlobal3d<PointCoordinateType>(*P);
89  stream << Pglobal.x << ' ' << Pglobal.y << ' ' << Pglobal.z
90  << QtCompat::endl;
91  }
92  }
93 
94  // save triangles
95  {
96  for (unsigned i = 0; i < triCount; ++i) {
97  const cloudViewer::VerticesIndexes* tsi =
98  mesh->getTriangleVertIndexes(i);
99  stream << "3 " << tsi->i1 << ' ' << tsi->i2 << ' ' << tsi->i3
100  << QtCompat::endl;
101  }
102  }
103 
104  return CC_FERR_NO_ERROR;
105 }

107 static QString GetNextLine(QTextStream& stream) {
108  QString currentLine;
109  // skip comments
110  do {
111  currentLine = stream.readLine();
112  // end of file?
113  if (currentLine.isNull()) return QString();
114  } while (currentLine.startsWith("#") || currentLine.isEmpty());
115 
116  return currentLine;
117 }

119 CC_FILE_ERROR OFFFilter::loadFile(const QString& filename,
120  ccHObject& container,
121  LoadParameters& parameters) {
122  // try to open file
123  QFile fp(filename);
124  if (!fp.open(QIODevice::ReadOnly | QIODevice::Text)) return CC_FERR_READING;
125 
126  QTextStream stream(&fp);
127 
128  QString currentLine = stream.readLine();
129  if (!currentLine.toUpper().startsWith("OFF")) return CC_FERR_MALFORMED_FILE;
130 
131  // check if the number of vertices/faces/etc. are on the first line (yes it
132  // happens :( )
133  QStringList tokens =
134  qtCompatSplitRegex(currentLine, "\\s+", QtCompat::SkipEmptyParts);
135  if (tokens.size() == 4) {
136  tokens.removeAt(0);
137  } else {
138  currentLine = GetNextLine(stream);
139 
140  // end of file already?!
141  if (currentLine.isNull()) return CC_FERR_MALFORMED_FILE;
142 
143  // read the number of vertices/faces
144  tokens = qtCompatSplitRegex(currentLine, "\\s+",
145  QtCompat::SkipEmptyParts);
146  if (tokens.size() <
147  2 /*3*/) // should be 3 but we only use the 2 firsts...
148  return CC_FERR_MALFORMED_FILE;
149  }
150 
151  bool ok = false;
152  unsigned vertCount = tokens[0].toUInt(&ok);
153  if (!ok) return CC_FERR_MALFORMED_FILE;
154  unsigned triCount = tokens[1].toUInt(&ok);
155  if (!ok) return CC_FERR_MALFORMED_FILE;
156 
157  // create cloud and reserve some memory
158  ccPointCloud* vertices = new ccPointCloud("vertices");
159  if (!vertices->reserve(vertCount)) {
160  delete vertices;
161  return CC_FERR_NOT_ENOUGH_MEMORY;
162  }
163 
164  // read vertices
165  {
166  CCVector3d Pshift(0, 0, 0);
167  for (unsigned i = 0; i < vertCount; ++i) {
168  currentLine = GetNextLine(stream);
169  tokens = qtCompatSplitRegex(currentLine, "\\s+",
170  QtCompat::SkipEmptyParts);
171  if (tokens.size() < 3) {
172  delete vertices;
173  return CC_FERR_MALFORMED_FILE;
174  }
175 
176  // read vertex
177  CCVector3d Pd(0, 0, 0);
178  {
179  bool vertexIsOk = false;
180  Pd.x = tokens[0].toDouble(&vertexIsOk);
181  if (vertexIsOk) {
182  Pd.y = tokens[1].toDouble(&vertexIsOk);
183  if (vertexIsOk) Pd.z = tokens[2].toDouble(&vertexIsOk);
184  }
185  if (!vertexIsOk) {
186  delete vertices;
187  return CC_FERR_MALFORMED_FILE;
188  }
189  }
190 
191  // first point: check for 'big' coordinates
192  if (i == 0) {
193  bool preserveCoordinateShift = true;
194  if (HandleGlobalShift(Pd, Pshift, preserveCoordinateShift,
195  parameters)) {
196  if (preserveCoordinateShift) {
197  vertices->setGlobalShift(Pshift);
198  }
199  CVLog::Warning(
200  "[OFF] Cloud has been recentered! Translation: "
201  "(%.2f ; %.2f ; %.2f)",
202  Pshift.x, Pshift.y, Pshift.z);
203  }
204  }
205 
206  CCVector3 P = CCVector3::fromArray((Pd + Pshift).u);
207  vertices->addPoint(P);
208  }
209  }
210 
211  ccMesh* mesh = new ccMesh(vertices);
212  // Set mesh name to filename (ParaView-style)
213  mesh->setName(QFileInfo(filename).fileName());
214  mesh->addChild(vertices);
215  if (!mesh->reserve(triCount)) {
216  delete mesh;
217  return CC_FERR_NOT_ENOUGH_MEMORY;
218  }
219 
220  // load triangles
221  {
222  bool ignoredPolygons = false;
223  for (unsigned i = 0; i < triCount; ++i) {
224  currentLine = GetNextLine(stream);
225  tokens = qtCompatSplitRegex(currentLine, "\\s+",
226  QtCompat::SkipEmptyParts);
227  if (tokens.size() < 3) {
228  delete mesh;
229  return CC_FERR_MALFORMED_FILE;
230  }
231 
232  unsigned polyVertCount = tokens[0].toUInt(&ok);
233  if (!ok || static_cast<int>(polyVertCount) >= tokens.size()) {
234  delete mesh;
235  return CC_FERR_MALFORMED_FILE;
236  }
237  if (polyVertCount == 3 || polyVertCount == 4) {
238  // decode indexes
239  unsigned indexes[4];
240  for (unsigned j = 0; j < polyVertCount; ++j) {
241  indexes[j] = tokens[j + 1].toUInt(&ok);
242  if (!ok) {
243  delete mesh;
244  return CC_FERR_MALFORMED_FILE;
245  }
246  }
247 
248  // reserve memory if necessary
249  unsigned polyTriCount = polyVertCount - 2;
250  if (mesh->size() + polyTriCount > mesh->capacity()) {
251  if (!mesh->reserve(mesh->size() + polyTriCount +
252  256)) // use some margin to avoid too
253  // many allocations
254  {
255  delete mesh;
256  return CC_FERR_NOT_ENOUGH_MEMORY;
257  }
258  }
259 
260  // triangle or quad only
261  mesh->addTriangle(indexes[0], indexes[1], indexes[2]);
262  if (polyVertCount == 4)
263  mesh->addTriangle(indexes[0], indexes[2], indexes[3]);
264 
265  } else {
266  ignoredPolygons = true;
267  }
268  }
269 
270  if (ignoredPolygons) {
271  CVLog::Warning(
272  "[OFF] Some polygons with an unhandled size (i.e. > 4) "
273  "were ignored!");
274  }
275  }
276 
277  if (mesh->size() == 0) {
278  CVLog::Warning("[OFF] Failed to load any polygon!");
279  mesh->detachChild(vertices);
280  delete mesh;
281  mesh = 0;
282 
283  container.addChild(vertices);
284  vertices->setEnabled(true);
285  } else {
286  mesh->shrinkToFit();
287 
288  // DGM: normals can be per-vertex or per-triangle so it's better to let
289  // the user do it himself later Moreover it's not always good idea if
290  // the user doesn't want normals (especially in ccViewer!) if
291  // (mesh->computeNormals()) mesh->showNormals(true); else
292  // CVLog::Warning("[OFF] Failed to compute per-vertex normals...");
293  CVLog::Warning(
294  "[OFF] Mesh has no normal! You can manually compute them "
295  "(select it then call \"Tools > Normals > Compute\")");
296 
297  vertices->setEnabled(false);
298  // vertices->setLocked(true); //DGM: no need to lock it as it is only
299  // used by one mesh!
300  container.addChild(mesh);
301  }
302 
303  return CC_FERR_NO_ERROR;
304 }
