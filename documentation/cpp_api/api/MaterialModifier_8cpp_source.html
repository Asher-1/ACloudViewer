76 void TextureSamplerParameters::SetAnisotropy(std::uint8_t a) {
77  // Set anisotropy to the largest power-of-two less than or eaqual to a.
78  anisotropy = a;
79  for (std::uint8_t b = 1 << 7; b > 0; b >>= 1) {
