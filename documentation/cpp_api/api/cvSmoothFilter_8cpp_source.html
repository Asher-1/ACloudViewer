34 void cvSmoothFilter::apply() {
35  if (!m_dataObject) {
36  CVLog::Error(QString("cvSmoothFilter::apply: null data object."));
37  return;
38  }
39 
40  if (!m_meshMode) {
41  CVLog::Error(QString("cvSmoothFilter::apply mesh supported only!"));
42  return;
43  }
44 
45  if (m_keepMode) {
46  return;
47  }
48 
49  VtkUtils::vtkInitOnce(m_smoothFilter);
50  m_smoothFilter->SetInputData(m_dataObject);
51  m_smoothFilter->SetNumberOfIterations(m_numberOfIterations);
52  m_smoothFilter->SetFeatureEdgeSmoothing(m_featureEdgeSmoothing);
53  m_smoothFilter->SetBoundarySmoothing(m_boundarySmoothing);
54  m_smoothFilter->SetConvergence(m_convergence);
55  m_smoothFilter->SetFeatureAngle(m_featureAngle);
56  m_smoothFilter->SetEdgeAngle(m_edgeAngle);
57 
58  m_smoothFilter->Update();
59 
60  if (!m_filterActor) {
61  VTK_CREATE(vtkPolyDataMapper, mapper);
62  mapper->SetInputConnection(m_smoothFilter->GetOutputPort());
63 
64  VtkUtils::vtkInitOnce(m_filterActor);
65  m_filterActor->SetMapper(mapper);
66  addActor(m_filterActor);
67  }
68 
69  update();
70 }
