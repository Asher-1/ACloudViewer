200 void cvCutFilter::updateCutWidget() {
201  double bounds[6];
202  if (isValidPolyData()) {
203  vtkPolyData* polyData = vtkPolyData::SafeDownCast(m_dataObject);
204  polyData->GetBounds(bounds);
205  } else if (isValidDataSet()) {
206  vtkDataSet* dataSet = vtkDataSet::SafeDownCast(m_dataObject);
207  dataSet->GetBounds(bounds);
208  }
209 
210  switch (m_cutType) {
211  case Plane: {
212  if (!m_planeWidget) {
213  VtkUtils::vtkInitOnce(m_planeWidget);
214  m_planeWidget->SetInteractor(getInteractor());
215  m_planeWidget->SetPlaceFactor(1);
216  m_planeWidget->SetProp3D(m_modelActor);
217  m_planeWidget->SetOutlineTranslation(
218  0); // make the outline non-movable
219  m_planeWidget->GetPlaneProperty()->SetOpacity(0.7);
220  m_planeWidget->GetPlaneProperty()->SetColor(.0, .0, .0);
221 
222  VtkUtils::ImplicitPlaneWidgetObserver* observer =
223  new VtkUtils::ImplicitPlaneWidgetObserver(this);
224  connect(observer, SIGNAL(originChanged(double*)), this,
225  SLOT(onOriginChanged(double*)));
226  connect(observer, SIGNAL(normalChanged(double*)), this,
227  SLOT(onNormalChanged(double*)));
228  observer->attach(m_planeWidget);
229 
230  resetPlaneWidget();
231  }
232 
233  m_planeWidget->On();
234  m_planeWidget->SetOrigin(m_origin);
235  m_planeWidget->SetNormal(m_normal);
236  m_planeWidget->PlaceWidget(bounds);
237 
238  safeOff(m_sphereWidget);
239  safeOff(m_boxWidget);
240 
241  m_configUi->planeGroupBox->show();
242  m_configUi->sphereGroupBox->hide();
243  cvGenericFilter::updateSize();
244  } break;
245 
246  case Sphere: {
247  if (!m_sphereWidget) {
248  VtkUtils::vtkInitOnce(m_sphereWidget);
249  m_sphereWidget->SetInteractor(getInteractor());
250  m_sphereWidget->SetProp3D(m_modelActor);
251  m_sphereWidget->PlaceWidget();
252 
253  VtkUtils::SphereWidgetObserver* observer =
254  new VtkUtils::SphereWidgetObserver(this);
255  observer->attach(m_sphereWidget);
256  connect(observer, SIGNAL(centerChanged(double*)), this,
257  SLOT(onCenterChanged(double*)));
258  connect(observer, SIGNAL(radiusChanged(double)), this,
259  SLOT(onRadiusChanged(double)));
260  resetSphereWidget();
261  }
262 
263  m_sphereWidget->On();
264  m_sphereWidget->PlaceWidget(bounds);
265  m_sphereWidget->SetCenter(m_center);
266  m_sphereWidget->SetRadius(m_radius);
267  safeOff(m_planeWidget);
268  safeOff(m_boxWidget);
269 
270  m_configUi->planeGroupBox->hide();
271  m_configUi->sphereGroupBox->show();
272  cvGenericFilter::updateSize();
273  } break;
274 
275  case Box: {
276  if (!m_boxWidget) {
277  VtkUtils::vtkInitOnce(m_boxWidget);
278  m_boxWidget->SetInteractor(getInteractor());
279  m_boxWidget->SetPlaceFactor(1.0);
280  m_boxWidget->SetProp3D(m_modelActor);
281  m_boxWidget->PlaceWidget();
282 
283  VtkUtils::BoxWidgetObserver* observer =
284  new VtkUtils::BoxWidgetObserver(this);
285  observer->attach(m_boxWidget);
286  connect(observer, SIGNAL(planesChanged(vtkPlanes*)), this,
287  SLOT(onPlanesChanged(vtkPlanes*)));
288  }
289 
290  m_boxWidget->On();
291  m_boxWidget->PlaceWidget(bounds);
292  safeOff(m_planeWidget);
293  safeOff(m_sphereWidget);
294 
295  m_configUi->planeGroupBox->hide();
296  m_configUi->sphereGroupBox->hide();
297  cvGenericFilter::updateSize();
298  } break;
299  }

402 void cvCutFilter::showContourLines(bool show) {
403  if (!m_meshMode) {
404  return;
405  }
406 
407  if (!show && !m_dataObject) return;
408 
409  if (!show && m_contourLinesActor) {
410  m_contourLinesActor->SetVisibility(show);
411  update();
412  return;
413  }
414 
415  vtkPolyData* polyData = vtkPolyData::SafeDownCast(m_dataObject);
416  if (!polyData) return;
417 
418  vtkPointData* pointData = polyData->GetPointData();
419 
420  // store contour names
421  int numOfArrays = pointData->GetNumberOfArrays();
422  if (!numOfArrays) return;
423 
424  //
425  QMap<QString, vtkDataArray*> scalarNameDataMap;
426  QString contourName;
427  for (int i = 0; i < numOfArrays; ++i) {
428  contourName = pointData->GetArrayName(i);
429  scalarNameDataMap.insert(
430  contourName,
431  pointData->GetArray(contourName.toStdString().c_str()));
432  }
433 
434  //
435  VTK_CREATE(vtkDoubleArray, scalars);
436  double range[2];
437 
438  // now extracting contour data
439  for (int i = 0; i < numOfArrays; ++i) {
440  vtkDataArray* da = pointData->GetArray(contourName.toUtf8().data());
441  if (da) {
442  int numOfTuple = da->GetNumberOfTuples();
443  da->GetRange(range);
444  for (int j = 0; j < numOfTuple; ++j) {
445  double val = da->GetTuple1(j);
446  scalars->InsertNextTuple1(val);
447  }
448  break;
449  }
450  }
451 
453  polyData->GetPointData()->SetScalars(scalars);
454  VTK_CREATE(vtkAppendPolyData, appendFilledContours);
455  int numberOfContours = 10;
456 
457  double delta =
458  (range[1] - range[0]) / static_cast<double>(numberOfContours - 1);
459 
460  // Keep the clippers alive
461  std::vector<vtkSmartPointer<vtkClipPolyData>> clippersLo;
462  std::vector<vtkSmartPointer<vtkClipPolyData>> clippersHi;
463 
464  for (int i = 0; i < numberOfContours; i++) {
465  double valueLo = range[0] + static_cast<double>(i) * delta;
466  double valueHi = range[0] + static_cast<double>(i + 1) * delta;
467  clippersLo.push_back(vtkSmartPointer<vtkClipPolyData>::New());
468  clippersLo[i]->SetValue(valueLo);
469  if (i == 0)
470  clippersLo[i]->SetInputData(polyData);
471  else
472  clippersLo[i]->SetInputConnection(
473  clippersHi[i - 1]->GetOutputPort(1));
474  clippersLo[i]->InsideOutOff();
475  clippersLo[i]->Update();
476 
477  clippersHi.push_back(vtkSmartPointer<vtkClipPolyData>::New());
478  clippersHi[i]->SetValue(valueHi);
479  clippersHi[i]->SetInputConnection(clippersLo[i]->GetOutputPort());
480  clippersHi[i]->GenerateClippedOutputOn();
481  clippersHi[i]->InsideOutOn();
482  clippersHi[i]->Update();
483  if (clippersHi[i]->GetOutput()->GetNumberOfCells() == 0) continue;
484 
485  VTK_CREATE(vtkFloatArray, cd);
486  cd->SetNumberOfComponents(1);
487  cd->SetNumberOfTuples(clippersHi[i]->GetOutput()->GetNumberOfCells());
488  cd->FillComponent(0, valueLo);
489 
490  clippersHi[i]->GetOutput()->GetCellData()->SetScalars(cd);
491  appendFilledContours->AddInputConnection(
492  clippersHi[i]->GetOutputPort());
493  }
494 
495  VTK_CREATE(vtkCleanPolyData, filledContours);
496  filledContours->SetInputConnection(appendFilledContours->GetOutputPort());
497 
498  setResultData(filledContours->GetOutput());
499 
500  VTK_CREATE(vtkLookupTable, lut);
501  lut->SetNumberOfColors(numberOfContours);
502  lut->SetRange(range[0], range[1]);
503  lut->SetHueRange(0.6667, 0);
504  lut->Build();
505 
506  VTK_CREATE(vtkPolyDataMapper, contourMapper);
507  contourMapper->SetInputConnection(filledContours->GetOutputPort());
508  contourMapper->SetScalarRange(range[0], range[1]);
509  contourMapper->SetScalarModeToUseCellData();
510  contourMapper->SetLookupTable(lut);
511 
512  if (!m_contourLinesActor) m_contourLinesActor = vtkActor::New();
513  m_contourLinesActor->SetMapper(contourMapper);
514  m_contourLinesActor->SetVisibility(show);
515 
516  addActor(m_contourLinesActor);
517  update();
518 }

525 void cvCutFilter::modelReady() {

528  double center[3];
529  double scalarRange[2];
530  if (isValidPolyData()) {
531  vtkPolyData* polyData = vtkPolyData::SafeDownCast(m_dataObject);
532  polyData->GetBounds(bounds);
533  polyData->GetCenter(center);
534  polyData->GetScalarRange(scalarRange);
535  } else if (isValidDataSet()) {
536  vtkDataSet* dataSet = vtkDataSet::SafeDownCast(m_dataObject);
537  dataSet->GetBounds(bounds);
538  dataSet->GetCenter(center);
539  dataSet->GetScalarRange(scalarRange);
540  }
541 
542  setScalarRange(scalarRange[0], scalarRange[1]);
543 
544  double xRange = bounds[1] - bounds[0];
545  double yRange = bounds[3] - bounds[2];
546  double zRange = bounds[5] - bounds[4];
547  double minRange = qMin(xRange, qMin(yRange, zRange));
548  setRadius((double)minRange / 2);
549 
550  Utils::ArrayAssigner<double>()(m_center, center);
551  Utils::ArrayAssigner<double>()(m_origin, center);
552 
553  m_normal[0] = 1;
554  m_normal[1] = 0;
555  m_normal[2] = 0;
556 
557  VtkUtils::SignalBlocker sb(m_configUi->centerXSpinBox);
558  sb.addObject(m_configUi->centerYSpinBox);
559  sb.addObject(m_configUi->centerZSpinBox);
560  sb.addObject(m_configUi->originZSpinBox);
561  sb.addObject(m_configUi->originYSpinBox);
562  sb.addObject(m_configUi->originZSpinBox);
563  sb.addObject(m_configUi->radiusSpinBox);
564 
565  m_configUi->centerXSpinBox->setValue(center[0]);
566  m_configUi->centerYSpinBox->setValue(center[1]);
567  m_configUi->centerZSpinBox->setValue(center[2]);
568  m_configUi->originXSpinBox->setValue(center[0]);
569  m_configUi->originYSpinBox->setValue(center[1]);
570  m_configUi->originZSpinBox->setValue(center[2]);
571  m_configUi->radiusSpinBox->setValue(m_radius);
572 
573  if (m_meshMode) {

575  }
576 
577  updateCutWidget();
578 
579  cvGenericFilter::modelReady();
580 }
