61 void NormalizedProgress::scale(unsigned totalSteps,
62  unsigned totalPercentage /*=100*/,
63  bool updateCurrentProgress /*=false*/) {
64  if (progressCallback) {
65  if (totalSteps == 0 || totalPercentage == 0) {
66  m_step = 1;
67  m_percentAdd = 0;
68  return;
69  }
70
