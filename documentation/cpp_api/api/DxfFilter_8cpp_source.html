70  CCVector3 convertPoint(double x, double y, double z) {
71  CCVector3d P(x, y, z);
72  if (m_firstPoint) {
73  if ((!m_preserveCoordinateShift ||
74  ecvGlobalShiftManager::NeedShift(P + m_globalShift)) &&
75  FileIOFilter::HandleGlobalShift(P, m_globalShift,
76  m_preserveCoordinateShift,
77  m_loadParameters)) {
78  CVLog::Warning(
79  "[DxfImporter] All points/vertices will be recentered! "
80  "Translation: (%.2f ; %.2f ; %.2f)",
81  m_globalShift.x, m_globalShift.y, m_globalShift.z);
82  }
83  m_firstPoint = false;
84  }
85 
86  P += m_globalShift;
87  return CCVector3::fromArray(P.u);
88  }

104  if (!m_points) {
105  m_points = new ccPointCloud("Points");
106  m_root->addChild(m_points);
107  }

199  void add3dFace(const DL_3dFaceData& face) override {
200  // TODO: understand what this really is?!
201  CCVector3 P[4];
202  for (unsigned i = 0; i < 4; ++i) {
203  P[i] = convertPoint(face.x[i], face.y[i], face.z[i]);
204  }
205 
206  // create the 'faces' mesh if necessary
207  if (!m_faces) {
208  ccPointCloud* vertices = new ccPointCloud("vertices");
209  m_faces = new ccMesh(vertices);
210  m_faces->setName("Faces");
211  m_faces->addChild(vertices);
212  m_faces->setVisible(true);
213  vertices->setEnabled(false);
214  // vertices->setLocked(true); //DGM: no need to lock it as it is
215  // only used by one mesh!
216  if (m_preserveCoordinateShift) {
217  vertices->setGlobalShift(m_globalShift);
218  }
219 
220  m_root->addChild(m_faces);
221  }
222 
223  ccPointCloud* vertices =
224  dynamic_cast<ccPointCloud*>(m_faces->getAssociatedCloud());
225  if (!vertices) {
226  assert(false);
227  return;
228  }
229 
230  int vertIndexes[4] = {-1, -1, -1, -1};
231  unsigned addedVertCount = 4;
232  // check if the two last vertices are the same
233  if (P[2].x == P[3].x && P[2].y == P[3].y && P[2].z == P[3].z)
234  addedVertCount = 3;
235 
236  // current face color
237  ecvColor::Rgb col;
238  ecvColor::Rgb* faceCol = nullptr;
239  if (getCurrentColour(col)) faceCol = &col;
240 
241  // look for already defined vertices
242  unsigned vertCount = vertices->size();
243  if (vertCount) {
244  // DGM TODO: could we be smarter?
245  for (unsigned i = 0; i < addedVertCount; ++i) {
246  for (unsigned j = 0; j < vertCount; ++j) {
247  const CCVector3* Pj = vertices->getPoint(j);
248  if (P[i].x == Pj->x && P[i].y == Pj->y && P[i].z == Pj->z) {
249  bool useCurrentVertex = true;
250 
251  // We must also check that the color is the same (if
252  // any)
253  if (faceCol || vertices->hasColors()) {
254  const ecvColor::Rgb* _faceCol =
255  faceCol ? faceCol : &ecvColor::white;
256  const ecvColor::Rgb* _vertCol =
257  vertices->hasColors()
258  ? &vertices->getPointColor(j)
259  : &ecvColor::white;
260  useCurrentVertex = (_faceCol->r == _vertCol->r &&
261  _faceCol->g == _vertCol->g &&
262  _faceCol->b == _vertCol->b);
263  }
264 
265  if (useCurrentVertex) {
266  vertIndexes[i] = static_cast<int>(j);
267  break;
268  }
269  }
270  }
271  }
272  }
273 
274  // now create new vertices
275  unsigned createdVertCount = 0;
276  {
277  for (unsigned i = 0; i < addedVertCount; ++i)
278  if (vertIndexes[i] < 0) ++createdVertCount;
279  }
280 
281  if (createdVertCount != 0) {
282  // reserve memory for the new vertices
283  if (!vertices->reserve(vertCount + createdVertCount)) {
284  CVLog::Error("[DxfImporter] Not enough memory!");
285  return;
286  }
287 
288  for (unsigned i = 0; i < addedVertCount; ++i) {
289  if (vertIndexes[i] < 0) {
290  vertIndexes[i] = static_cast<int>(vertCount++);
291  vertices->addPoint(P[i]);
292  }
293  }
294  }
295 
296  // number of triangles to add
297  unsigned addTriCount = (addedVertCount == 3 ? 1 : 2);
298 
299  // now add the corresponding face(s)
300  if (!m_faces->reserve(m_faces->size() + addTriCount)) {
301  CVLog::Error("[DxfImporter] Not enough memory!");
302  return;
303  }
304  m_faces->addTriangle(vertIndexes[0], vertIndexes[1], vertIndexes[2]);
305  if (addedVertCount == 4)
306  m_faces->addTriangle(vertIndexes[0], vertIndexes[2],
307  vertIndexes[3]);
308 
309  // add per-triangle normals
310  {
311  // normals table
312  NormsIndexesTableType* triNormsTable = m_faces->getTriNormsTable();
313  bool firstTime = false;
314  if (!triNormsTable) {
315  triNormsTable = new NormsIndexesTableType();
316  m_faces->setTriNormsTable(triNormsTable);
317  firstTime = true;
318  }
319 
320  // add 1 or 2 new entries
321  unsigned triNormCount = triNormsTable->currentSize();
322  if (!triNormsTable->reserveSafe(triNormsTable->currentSize() +
323  addTriCount)) {
324  CVLog::Error("[DxfImporter] Not enough memory!");
325  return;
326  }
327 
328  CCVector3 N = (P[1] - P[0]).cross(P[2] - P[0]);
329  N.normalize();
330  triNormsTable->addElement(ccNormalVectors::GetNormIndex(N.u));
331  if (addTriCount == 2) {
332  N = (P[2] - P[0]).cross(P[3] - P[0]);
333  N.normalize();
334  triNormsTable->addElement(ccNormalVectors::GetNormIndex(N.u));
335  }
336 
337  // per-triangle normals indexes
338  if (firstTime) {
339  if (!m_faces->reservePerTriangleNormalIndexes()) {
340  CVLog::Error("[DxfImporter] Not enough memory!");
341  return;
342  }
343  m_faces->showNormals(true);
344  }
345  int n1 = static_cast<int>(triNormCount);
346  m_faces->addTriangleNormalIndexes(n1, n1, n1);
347  if (addTriCount == 2) {
348  int n2 = static_cast<int>(triNormCount + 1);
349  m_faces->addTriangleNormalIndexes(n2, n2, n2);
350  }
351  }
352 
353  // and now for the color
354  if (faceCol) {
355  // RGB field already instantiated?
356  if (vertices->hasColors()) {
357  for (unsigned i = 0; i < createdVertCount; ++i)
358  vertices->addRGBColor(*faceCol);
359  }
360  // otherwise, reserve memory and set all previous points to white by
361  // default
362  else if (vertices->setRGBColor(ecvColor::white)) {
363  // then replace the last color(s) by the current one
364  for (unsigned i = 0; i < createdVertCount; ++i)
365  vertices->setPointColor(
366  vertCount - 1 - i,
367  ecvColor::Rgba(*faceCol, ecvColor::MAX));
368  m_faces->showColors(true);
369  }
370  } else if (vertices->hasColors()) {
371  // add default color if none is defined!
372  for (unsigned i = 0; i < createdVertCount; ++i)
373  vertices->addRGBColor(ecvColor::white);
374  }
375  }

381  void addArc(const DL_ArcData& data) override {
382  // we load arc as simple polylines!
383  ccPointCloud* polyVertices = new ccPointCloud("vertices");
384  ccPolyline* poly = new ccPolyline(polyVertices);
385  poly->addChild(polyVertices);
386 
387  double arcLength_deg = data.angle2 - data.angle1;
388  unsigned vertexCount =
389  1 +
390  static_cast<unsigned>(std::max(
391  1.0, arcLength_deg)); // we use a 1 degree resolution
392  // by default for now
393  double step_deg = 1.0;
394  if (arcLength_deg < 360.0) {
395  assert(vertexCount >= 2);
396  step_deg = arcLength_deg / (vertexCount - 1);
397  } else {
398  vertexCount = 360;
399  step_deg = 1.0;
400  }
401 
402  if (!polyVertices->reserve(vertexCount) ||
403  !poly->reserve(vertexCount)) {
404  CVLog::Error("[DxfImporter] Not enough memory!");
405  delete poly;
406  return;
407  }
408  polyVertices->setEnabled(false);
409  poly->setVisible(true);
410  poly->setName("Arc");
411  poly->addPointIndex(0, vertexCount);
412  poly->setClosed(arcLength_deg >= 360.0);
413 
414  // some entities can have small coordinates (drawings, origin, etc.)
415  // hiding the fact that other polylines have large coordinates!
416  m_firstPoint = true;
417 
418  CCVector3 Clocal = convertPoint(data.cx, data.cy, data.cz);
419 
420  if (m_preserveCoordinateShift) {
421  polyVertices->setGlobalShift(m_globalShift);
422  poly->setGlobalShift(m_globalShift);
423  }
424 
425  for (unsigned i = 0; i < vertexCount; ++i) {
426  double angle_deg = data.angle1 + i * step_deg;
427  double angle_rad = cloudViewer::DegreesToRadians(angle_deg);
428  CCVector3 P =
429  Clocal + CCVector3(static_cast<PointCoordinateType>(
430  data.radius * cos(angle_rad)),
431  static_cast<PointCoordinateType>(
432  data.radius * sin(angle_rad)),
433  0);
434  polyVertices->addPoint(P);
435  }
436 
437  // flags
438  poly->setClosed(false);
439 
440  // color
441  ecvColor::Rgb col;
442  if (getCurrentColour(col)) {
443  poly->setColor(col);
444  poly->showColors(true);
445  }
446 
447  m_root->addChild(poly);
448  }

542  bool getCurrentColour(ecvColor::Rgb& ccColour) {
543  const DL_Attributes attributes = getAttributes();
544 
545  int colourIndex = attributes.getColor();
546 
547  if (colourIndex == 0) {
548  // TODO Colours BYBLOCK not handled
549  return false;
550  } else if (colourIndex == 256) {
551  // an attribute of 256 means the colours are BYLAYER, so grab it
552  // from our map instead
553  const int defaultIndex = -1;
554  colourIndex = m_layerColourMap.value(attributes.getLayer().c_str(),
555  defaultIndex);
556 
557  // if we don't have any information on the current layer
558  if (colourIndex == defaultIndex) return false;
559  }
560 
561  ccColour.r = static_cast<ColorCompType>(dxfColors[colourIndex][0] *
562  ecvColor::MAX);
563  ccColour.g = static_cast<ColorCompType>(dxfColors[colourIndex][1] *
564  ecvColor::MAX);
565  ccColour.b = static_cast<ColorCompType>(dxfColors[colourIndex][2] *
566  ecvColor::MAX);
567 
568  return true;
569  }

589 CC_FILE_ERROR DxfFilter::saveToFile(ccHObject* root,
590  const QString& filename,
591  const SaveParameters& parameters) {
592 #ifndef CV_DXF_SUPPORT
593 
594  CVLog::Error(
595  "[DXF] DXF format not supported! Check compilation parameters!");
596  return CC_FERR_CONSOLE_ERROR;
597 
598 #else
599 
600  if (!root || filename.isEmpty()) return CC_FERR_BAD_ARGUMENT;
601 
602  ccHObject::Container polylines;
603  root->filterChildren(polylines, true, CV_TYPES::POLY_LINE);
604  if (root->isKindOf(CV_TYPES::POLY_LINE)) polylines.push_back(root);
605  ccHObject::Container meshes;
606  root->filterChildren(meshes, true, CV_TYPES::MESH);
607  if (root->isKindOf(CV_TYPES::MESH)) meshes.push_back(root);
608  ccHObject::Container clouds;
609  root->filterChildren(clouds, true, CV_TYPES::POINT_CLOUD,
610  true); // we don't want polylines!
611  if (root->isKindOf(CV_TYPES::POINT_CLOUD)) clouds.push_back(root);
612 
613  if (!clouds.empty()) {
614  // remove the polylines vertices
615  ccHObject::Container realClouds;
616  try {
617  for (ccHObject* cloud : clouds) {
618  ccHObject* parent = cloud->getParent();
619  if (parent) {
620  if (parent->isA(CV_TYPES::POLY_LINE)) {
621  if (std::find(polylines.begin(), polylines.end(),
622  parent) != polylines.end()) {
623  // the parent is already in the 'polylines' set
624  continue;
625  }
626  } else if (parent->isKindOf(CV_TYPES::MESH)) {
627  if (std::find(meshes.begin(), meshes.end(), parent) !=
628  meshes.end()) {
629  // the parent is already in the 'meshes' set
630  continue;
631  }
632  }
633  }
634 
635  realClouds.push_back(cloud);
636  }
637  clouds = realClouds;
638  } catch (const std::bad_alloc&) {
639  return CC_FERR_NOT_ENOUGH_MEMORY;
640  }
641  }
642 
643  // only polylines and meshes are handled for now
644  size_t polyCount = polylines.size();
645  size_t meshCount = meshes.size();
646  size_t cloudCount = clouds.size();
647  if (polyCount + meshCount + cloudCount == 0) return CC_FERR_NO_SAVE;
648 
649  // get global bounding box
650  ccHObject::GlobalBoundingBox globalBB;
651  {
652  ccHObject::Container* containers[3] = {&polylines, &meshes, &clouds};
653 
654  bool firstEntity = true;
655  for (int j = 0; j < 3; ++j) {
656  const ccHObject::Container& container = *containers[j];
657  for (size_t i = 0; i < container.size(); ++i) {
658  ccHObject::GlobalBoundingBox bb =
659  container[i]->getOwnGlobalBB();
660  // update global BB
661  globalBB += bb;
662  }
663  }
664  }
665 
666  CCVector3d diag = globalBB.getDiagVec();
667  double baseSize = std::max(diag.x, diag.y);
668  double lineWidth = baseSize / 40.0;
669  double pageMargin = baseSize / 20.0;
670 
671  if (CheckForSpecialChars(filename)) {
672  CVLog::Warning(QString(
673  "[DXF] Output filename contains special characters. It might "
674  "be scrambled or rejected by the third party library..."));
675  }
676 
677  DL_Dxf dxf;
678  DL_WriterA* dw = dxf.out(qPrintable(filename), DL_VERSION_R12);
679  if (!dw) {
680  return CC_FERR_WRITING;
681  }
682 
683  CC_FILE_ERROR result = CC_FERR_NO_ERROR;
684 
685  try {
686  dxf.writeComment(*dw, FileIO::createdBy().toStdString());
687  dxf.writeComment(*dw, FileIO::createdDateTime().toStdString());
688 
689  // write header
690  dxf.writeHeader(*dw);
691 
692  // add dimensions
693  const CCVector3d& bbMinCorner = globalBB.minCorner();
694  const CCVector3d& bbMaxCorner = globalBB.maxCorner();
