178 template <typename... Args>
179 inline std::string FormatString(const std::string& format, Args... args) {
180  int size_s = std::snprintf(nullptr, 0, format.c_str(), args...) +
181  1; // Extra space for '\0'
182  if (size_s <= 0) {
183  throw std::runtime_error("Error during formatting.");
184  }
185  auto size = static_cast<size_t>(size_s);
186  auto buf = std::make_unique<char[]>(size);
187  std::snprintf(buf.get(), size, format.c_str(), args...);
188  return std::string(buf.get(),
189  buf.get() + size - 1); // We don't want the '\0' inside
190 };
191 
193 template <typename... Args>
194 inline std::string FastFormatString(const std::string& format, Args... args) {
195 #ifdef _WIN32
196  return FormatString(format, args...);
197 #else
198  char* buffer = nullptr;
199  int size_s = asprintf(&buffer, format.c_str(), args...);
200  if (size_s == -1) {
201  throw std::runtime_error("Error during formatting.");
202  }
203  auto ret = std::string(buffer,
204  buffer + size_s); // no + 1 since we ignore the \0
205  std::free(buffer); // asprintf calls malloc
206  return ret;
207 #endif // _WIN32
208 };
