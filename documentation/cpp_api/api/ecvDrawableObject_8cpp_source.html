81 void ccDrawableObject::getDrawingParameters(glDrawParams& params) const {

165 void ccDrawableObject::popDisplayState(bool apply /*=true*/) {
166  if (!m_displayStateStack.empty()) {
167  const DisplayState::Shared state = m_displayStateStack.back();
168  if (state && apply) {
169  applyDisplayState(*state);
170  }
171  m_displayStateStack.pop_back();
172  }
173 }

175 void ccDrawableObject::applyDisplayState(const DisplayState& state) {
176  if (state.visible != m_visible) setVisible(state.visible);
177  if (state.colorsDisplayed != m_colorsDisplayed)
178  showColors(state.colorsDisplayed);
179  if (state.normalsDisplayed != m_normalsDisplayed)
180  showNormals(state.normalsDisplayed);
181  if (state.sfDisplayed != m_sfDisplayed) showSF(state.sfDisplayed);
182  if (state.colorIsOverridden != m_colorIsOverridden)
183  enableTempColor(state.colorIsOverridden);
184  if (state.showNameIn3D != m_showNameIn3D) showNameIn3D(state.showNameIn3D);
185 }
