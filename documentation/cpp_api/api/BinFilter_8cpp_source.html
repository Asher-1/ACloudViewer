174 CC_FILE_ERROR BinFilter::SaveFileV2(QFile& out, ccHObject* object) {
175  if (!object) return CC_FERR_BAD_ARGUMENT;
176 
177  // About BIN versions:
178  //- 'original' version (file starts by the number of clouds - no header)
179  //- 'new' evolutive version, starts by 4 bytes ("CCB2") + save the current
180  // ccObject version
181 
182  CC_FILE_ERROR result = CC_FERR_NO_ERROR;
183 
184  // we check if all linked entities are in the sub tree we are going to save
185  //(such as vertices for a mesh!)
186  ccHObject::Container toCheck;
187  toCheck.push_back(object);
188  while (!toCheck.empty()) {
189  ccHObject* currentObject = toCheck.back();
190  assert(currentObject);
191  toCheck.pop_back();
192 
193  // we check objects that have links to other entities (meshes,
194  // polylines, etc.)
195  std::unordered_set<const ccHObject*> dependencies;
196  if (currentObject->isA(CV_TYPES::MESH) ||
197  currentObject->isKindOf(CV_TYPES::PRIMITIVE)) {
198  ccMesh* mesh = ccHObjectCaster::ToMesh(currentObject);
199  if (mesh->getAssociatedCloud())
200  dependencies.insert(mesh->getAssociatedCloud());
201  if (mesh->getMaterialSet())
202  dependencies.insert(mesh->getMaterialSet());
203  if (mesh->getTexCoordinatesTable())
204  dependencies.insert(mesh->getTexCoordinatesTable());
205  } else if (currentObject->isA(CV_TYPES::SUB_MESH)) {
206  dependencies.insert(currentObject->getParent());
207  } else if (currentObject->isKindOf(CV_TYPES::POLY_LINE)) {
208  cloudViewer::GenericIndexedCloudPersist* cloud =
209  static_cast<ccPolyline*>(currentObject)
210  ->getAssociatedCloud();
211  ccPointCloud* pc = dynamic_cast<ccPointCloud*>(cloud);
212  if (pc)
213  dependencies.insert(pc);
214  else
215  CVLog::Warning(QString("[BIN] Poyline '%1' is associated to an "
216  "unhandled vertices structure?!")
217  .arg(currentObject->getName()));
218  } else if (currentObject->isKindOf(CV_TYPES::SENSOR)) {
219  ccIndexedTransformationBuffer* buffer =
220  static_cast<ccSensor*>(currentObject)->getPositions();
221  if (buffer) dependencies.insert(buffer);
222  } else if (currentObject->isA(CV_TYPES::LABEL_2D)) {
223  cc2DLabel* label = static_cast<cc2DLabel*>(currentObject);
224  for (unsigned i = 0; i < label->size(); ++i) {
225  const cc2DLabel::PickedPoint& pp = label->getPickedPoint(i);
226  if (pp.cloud)
227  dependencies.insert(pp.cloud);
228  else if (pp.mesh)
229  dependencies.insert(pp.mesh);
230  }
231  } else if (currentObject->isA(CV_TYPES::FACET)) {
232  ccFacet* facet = static_cast<ccFacet*>(currentObject);
233  if (facet->getOriginPoints())
234  dependencies.insert(facet->getOriginPoints());
235  if (facet->getContourVertices())
236  dependencies.insert(facet->getContourVertices());
237  if (facet->getPolygon()) dependencies.insert(facet->getPolygon());
238  if (facet->getContour()) dependencies.insert(facet->getContour());
239  } else if (currentObject->isKindOf(CV_TYPES::IMAGE)) {
240  ccImage* image = static_cast<ccImage*>(currentObject);
241  if (image->getAssociatedSensor())
242  dependencies.insert(image->getAssociatedSensor());
243  }
244 
245  for (std::unordered_set<const ccHObject*>::const_iterator it =
246  dependencies.begin();
247  it != dependencies.end(); ++it) {
248  if (!object->find((*it)->getUniqueID())) {
249  CVLog::Warning(
250  QString("[BIN] Dependency broken: entity '%1' must "
251  "also be in selection in order to save '%2'")
252  .arg((*it)->getName(),
253  currentObject->getName()));
254  result = CC_FERR_BROKEN_DEPENDENCY_ERROR;
255  }
256  }
257  // release some memory...
258  dependencies.clear();
259 
260  for (unsigned i = 0; i < currentObject->getChildrenNumber(); ++i)
261  toCheck.push_back(currentObject->getChild(i));
262  }
263 
264  if (result != CC_FERR_NO_ERROR) {
265  return result;
266  }
267 
268  // header
269  // Since ver 2.5.2, the 4th character of the header corresponds to
270  //'deserialization flags' (see ccSerializableObject::DeserializationFlags)
271  char firstBytes[5] = "CCB2";
272  {
273  char flags = 0;
274  if (sizeof(PointCoordinateType) == 8) {
275  flags |= static_cast<char>(
276  ccSerializableObject::DF_POINT_COORDS_64_BITS);
277  }
278  // internal representation of scalar fields is now always floats
279  flags |= static_cast<char>(ccSerializableObject::DF_SCALAR_VAL_32_BITS);
280  assert(flags <= 8);
281  firstBytes[3] = 48 + flags; // 48 = ASCII("0")
282  }
283 
284  if (out.write(firstBytes, 4) < 0) return CC_FERR_WRITING;
285 
286  // Current BIN file version
287  short dataVersion = object->minimumFileVersion();
288  {
289  CVLog::Print(QString("[BIN] Output file version: %1.%2 (automatically "
290  "deduced from selected entities)")
291  .arg(dataVersion / 10)
292  .arg(dataVersion % 10));
293  uint32_t binVersion_u32 = dataVersion;
294  if (out.write((char*)&binVersion_u32, 4) < 0) return CC_FERR_WRITING;
295  }
296 
297  if (!object->toFile(out, dataVersion)) {
298  result = CC_FERR_CONSOLE_ERROR;
299  }
300 
301  s_lastSavedFileBinVersion = dataVersion;
302 
303  out.close();
304 
305  return result;
306 }

308 CC_FILE_ERROR BinFilter::loadFile(const QString& filename,
309  ccHObject& container,
310  LoadParameters& parameters) {
311  CVLog::Print(QString("[BIN] Opening file '%1'...").arg(filename));
312 
313  // opening file
314  QFile in(filename);
315  if (!in.open(QIODevice::ReadOnly)) return CC_FERR_READING;
316 
317  uint32_t firstBytes = 0;
318  if (in.read((char*)&firstBytes, 4) < 0) return CC_FERR_READING;
319  bool v1 = (strncmp((char*)&firstBytes, "CCB", 3) != 0);
320 
321  if (v1) {
322  return LoadFileV1(
323  in, container, static_cast<unsigned>(firstBytes),
324  parameters); // firstBytes == number of scans for V1 files!
325  } else {
326  // Since ver 2.5.2, the 4th character of the header corresponds to 'load
327  // flags'
328  int flags = 0;
329  {
330  QChar c(reinterpret_cast<char*>(&firstBytes)[3]);
331  bool ok;
332  flags = QString(c).toInt(&ok);
333  if (!ok || flags > 8) {
334  CVLog::Error(QString("Invalid file header (4th byte is '%1'?!)")
335  .arg(c));
336  return CC_FERR_WRONG_FILE_TYPE;
337  }
338  }
339 
340  return BinFilter::LoadFileV2(in, container, flags,
341  parameters.alwaysDisplayLoadDialog,
342  parameters.parentWidget);
343  }

353 ccHObject* FindRobust(ccHObject* root,
354  ccHObject* source,
355  const ccObject::LoadedIDMap& oldToNewIDMap,
356  unsigned oldUniqueID,
357  CV_CLASS_ENUM expectedType) {
358  ccObject::LoadedIDMap::const_iterator it = oldToNewIDMap.find(oldUniqueID);
359  while (it != oldToNewIDMap.end() && it.key() == oldUniqueID) {
360  unsigned uniqueID = it.value();
361  ++it;
362 
363  if (source) {
364  // 1st test the parent
365  ccHObject* parent = source->getParent();
366  if (Match(parent, uniqueID, expectedType)) return parent;
367 
368  // now test the children
369  for (unsigned i = 0; i < source->getChildrenNumber(); ++i) {
370  ccHObject* child = source->getChild(i);
371  if (Match(child, uniqueID, expectedType)) return child;
372  }
373  }
374 
375  // now test the whole DB
376  ccHObject* object = root->find(uniqueID);
377  // if we've found an object, we must also test its type!
378  if (object && object->isKindOf(expectedType)) {
379  return object;
380  }
381  }
382 
383  // no entity found!
384  return nullptr;
385 }

413 CC_FILE_ERROR BinFilter::LoadFileV2(QFile& in,
414  ccHObject& container,
415  int flags,
416  bool parallel,
417  QWidget* parentWidget /*=nullptr*/) {
418  assert(in.isOpen());
419 
420  uint32_t binVersion = 20;
421  if (in.read((char*)&binVersion, 4) < 0) {
422  return CC_FERR_READING;
423  }
424 
425  if (binVersion < 20) // should be superior to 2.0!
426  {
427  return CC_FERR_MALFORMED_FILE;
428  }
429 
430  QString coordsFormat =
431  ((flags & ccSerializableObject::DF_POINT_COORDS_64_BITS) ? "double"
432  : "float");
433  QString scalarFormat =
434  ((flags & ccSerializableObject::DF_SCALAR_VAL_32_BITS) ? "float"
435  : "double");
436  CVLog::Print(QString("[BIN] Version %1.%2 (coords: %3 / scalar: %4)")
437  .arg(binVersion / 10)
438  .arg(binVersion % 10)
439  .arg(coordsFormat)
440  .arg(scalarFormat));
441 
442  if (ccObject::GetCurrentDBVersion() < binVersion) {
443  CVLog::Error(
444  "This version of CloudViewer is too old and can't load this "
445  "file, sorry");
446  return CC_FERR_CONSOLE_ERROR;
447  }
448 
449  // we read the first entity type
450  CV_CLASS_ENUM classID =
451  ccObject::ReadClassIDFromFile(in, static_cast<short>(binVersion));
452  if (classID == CV_TYPES::OBJECT) {
453  return CC_FERR_CONSOLE_ERROR;
454  }
455 
456  // call the CC object factory
457  ccHObject* root = ccHObject::New(classID);
458  if (!root) {
459  return CC_FERR_MALFORMED_FILE;
460  }
461 
462  QScopedPointer<ecvProgressDialog> pDlg(nullptr);
463  if (parallel && parentWidget) {
464  pDlg.reset(new ecvProgressDialog(false, parentWidget));
465  pDlg->setMethodTitle(QObject::tr("BIN file"));
466  pDlg->setInfo(QObject::tr("Loading: %1").arg(in.fileName()));
467  pDlg->setRange(0, 0);
468  pDlg->show();
469  }
470 
471  ccObject::LoadedIDMap oldToNewIDMap;
472 
473  if (classID == CV_TYPES::CUSTOM_H_OBJECT) {
474  // store seeking position
475  size_t original_pos = in.pos();
476  // we need to load it as plain ccCustomHobject
477  root->fromFileNoChildren(in, static_cast<short>(binVersion), flags,
478  oldToNewIDMap); // this will load it
479  in.seek(original_pos); // reseek back the file
480 
481  QString classId = root->getMetaData("class_name").toString();
482  QString pluginId = root->getMetaData("plugin_name").toString();
483 
484  // get rid of the previous ccCustomHobject instance
485  delete root;
486  root = nullptr;
487 
488  // try to get a new object from external factories
489  ccHObject* new_child = ccHObject::New(pluginId, classId);
490  if (new_child) // found a plugin that can deserialize it
491  root = new_child;
492  else
493  return CC_FERR_FILE_WAS_WRITTEN_BY_UNKNOWN_PLUGIN;
494  }
495 
496  bool success = false;
497 
498  if (parallel) {
499  // concurrent call in a separate thread
500  QFuture<bool> future = QtConcurrent::run([&]() {
501  return root->fromFile(in, static_cast<short>(binVersion), flags,
502  oldToNewIDMap);
503  });
504 
505  while (!future.isFinished()) {
506 #if defined(CV_WINDOWS)
507  ::Sleep(500);
508 #else
509  usleep(500 * 1000);
510 #endif
511  if (pDlg) {
512  pDlg->setValue(pDlg->value() + 1);
513  }
514  // pDlg.setValue(static_cast<int>(in.pos())); //DGM: in fact, the
515  // file reading part is just half of the work!
516  QApplication::processEvents();
517  }
518 
519  success = future.result();
520  } else {
521  success = root->fromFile(in, static_cast<short>(binVersion), flags,
522  oldToNewIDMap);
523  }
524 
525  bool forceLoadAfterError = false;
526  if (!success) {
527  // delete root; DGM: can't delete it, too dangerous (bad pointers ;)
528  CVLog::Error(QString("Failed to read file (file position: %1 / %2")
529  .arg(in.pos())
530  .arg(in.size()));
531 
532  if (!root->isA(CV_TYPES::HIERARCHY_OBJECT) ||
533  root->getChildrenNumber() != 0) {
534  ContinueAfterError(forceLoadAfterError, true);
535  }
536 
537  if (!forceLoadAfterError) {
538  return CC_FERR_CONSOLE_ERROR;
539  }
540  }
541 
542  CC_FILE_ERROR result = CC_FERR_NO_ERROR;
543 
544  // re-link objects (and check errors)
545  bool checkErrors = true;
546  ccHObject* orphans = new ccHObject("Orphans (CORRUPTED FILE)");
547  ccHObject::Container toCheck;
548  toCheck.push_back(root);
549  while (!toCheck.empty()) {
550  ccHObject* currentObject = toCheck.back();
551  toCheck.pop_back();
552 
553  assert(currentObject);
554 
555  // we check objects that have links to other entities (meshes,
556  // polylines, etc.)
557  if (currentObject->isKindOf(CV_TYPES::MESH)) {
558  // specific case: mesh groups are deprecated!
559  if (currentObject->isA(CV_TYPES::MESH_GROUP)) {
560  // TODO
561  CVLog::Warning(QString("[BIN] Mesh groups are deprecated! "
562  "Entity %1 should be ignored...")
563  .arg(currentObject->getName()));
564  } else if (currentObject->isA(CV_TYPES::SUB_MESH)) {
565  ccSubMesh* subMesh = ccHObjectCaster::ToSubMesh(currentObject);
566 
567  // normally, the associated mesh should be the sub-mesh's
568  // parent! however we have its ID so we will look for it just to
569  // be sure
570  intptr_t meshID = (intptr_t)subMesh->getAssociatedMesh();
571  if (meshID > 0) {
572  ccHObject* mesh = FindRobust(root, subMesh, oldToNewIDMap,
573  static_cast<unsigned>(meshID),
574  CV_TYPES::MESH);
575  if (mesh) {
576  subMesh->setAssociatedMesh(
577  ccHObjectCaster::ToMesh(mesh),
578  false); //'false' because previous mesh is not
579  // null (= real mesh ID)!!!
580  } else {
581  // we have a problem here ;)
582  // normally, the associated mesh should be the
583  // sub-mesh's parent!
584  if (subMesh->getParent() &&
585  subMesh->getParent()->isA(CV_TYPES::MESH)) {
586  subMesh->setAssociatedMesh(
587  ccHObjectCaster::ToMesh(
588  subMesh->getParent()),
589  false); //'false' because previous mesh is
590  // not null (= real mesh ID)!!!
591  } else {
592  subMesh->setAssociatedMesh(
593  nullptr,
594  false); //'false' because previous mesh is
595  // not null (= real mesh ID)!!!
596  // DGM: can't delete it, too dangerous (bad pointers
597  // ;) delete subMesh;
598  CVLog::Warning(
599  QString("[BIN] Couldn't find associated "
600  "mesh (ID=%1) for sub-mesh '%2' in "
601  "the file!")
602  .arg(meshID)
603  .arg(subMesh->getName()));
604  if (!ContinueAfterError(forceLoadAfterError)) {
605  return CC_FERR_MALFORMED_FILE;
606  }
607  }
608  }
609  }
610  } else if (currentObject->isA(CV_TYPES::MESH) ||
611  currentObject->isKindOf(
612  CV_TYPES::PRIMITIVE)) // CV_TYPES::MESH or
613  // CV_TYPES::PRIMITIVE!
614  {
615  ccMesh* mesh = ccHObjectCaster::ToMesh(currentObject);
616  assert(mesh);
617 
618  // vertices
619  intptr_t cloudID = (intptr_t)mesh->getAssociatedCloud();
620  if (cloudID > 0) {
621  ccHObject* cloud =
622  FindRobust(root, mesh, oldToNewIDMap,
623  static_cast<unsigned>(cloudID),
624  CV_TYPES::POINT_CLOUD);
625  if (cloud) {
626  ccGenericPointCloud* genericCloud =
627  ccHObjectCaster::ToGenericPointCloud(cloud);
628  assert(genericCloud);
629  mesh->setAssociatedCloud(genericCloud);
630  } else {
631  // we have a problem here ;)
632  mesh->setAssociatedCloud(nullptr);
633  if (mesh->getMaterialSet()) {
634  mesh->setMaterialSet(nullptr, false);
635  }
636  // DGM: can't delete it, too dangerous (bad pointers ;)
637  // delete mesh;
638  if (mesh->getParent()) {
639  mesh->getParent()->removeDependencyWith(mesh);
640  mesh->getParent()->removeChild(mesh);
641  }
642  CVLog::Warning(
643  QString("[BIN] Couldn't find vertices (ID=%1) "
644  "for mesh '%2' in the file!")
645  .arg(cloudID)
646  .arg(mesh->getName()));
647  if (root == mesh &&
648  !ContinueAfterError(forceLoadAfterError)) {
649  return CC_FERR_MALFORMED_FILE;
650  }
651  currentObject = mesh = nullptr;
652  }
653  } else {
654  CVLog::Warning(QString("[BIN] Mesh '%1' has no vertices! "
655  "It will be ignored.")
656  .arg(mesh->getName()));
657  if (mesh->getParent()) {
658  mesh->getParent()->removeChild(mesh);
659  } else {
660  if (root == mesh) {
661  delete mesh;
662  if (!ContinueAfterError(forceLoadAfterError)) {
663  return CC_FERR_MALFORMED_FILE;
664  }
665  } else {
666  // we'll try to live with that
667  delete mesh;
668  }
669  }
670  currentObject = mesh = nullptr;
