41  // make the transform a little bit cleaner (necessary as it's read from
42  // ASCII!)
43  {
44  // #ifdef QT_DEBUG
45  // //test the matrix quality
46  // ccGLMatrixd before = mat;
47  // CCVector3d X0(before.getColumn(0));
48  // CCVector3d Y0(before.getColumn(1));
49  // CCVector3d Z0(before.getColumn(2));
50  // double normX0 = X0.norm();

52  // double normZ0 = Z0.norm();

80 CC_FILE_ERROR PTXFilter::loadFile(const QString& filename,
81  ccHObject& container,
82  LoadParameters& parameters) {
83  // open ASCII file for reading
84  QFile file(filename);
85  if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
86  return CC_FERR_READING;
87  }
88 
89  QTextStream inFile(&file);
90 
91  CCVector3d PshiftTrans(0, 0, 0);
92  CCVector3d PshiftCloud(0, 0, 0);
93  bool preserveCoordinateShift = true;
94 
95  CC_FILE_ERROR result = CC_FERR_NO_LOAD;
96  ScalarType minIntensity = 0;
97  ScalarType maxIntensity = 0;
98 
99  // progress dialog
100  QScopedPointer<ecvProgressDialog> pDlg(nullptr);
101  if (parameters.parentWidget) {
102  pDlg.reset(new ecvProgressDialog(true, parameters.parentWidget));
103  pDlg->setMethodTitle(QObject::tr("Loading PTX file"));
104  pDlg->setAutoClose(false);
105  }
106 
107  // progress dialog (for normals computation)
108  QScopedPointer<ecvProgressDialog> normalsProgressDlg(nullptr);
109  if (parameters.parentWidget && parameters.autoComputeNormals) {
110  normalsProgressDlg.reset(
111  new ecvProgressDialog(true, parameters.parentWidget));
112  normalsProgressDlg->setAutoClose(false);
113  normalsProgressDlg->hide();
114  }
115 
116  for (unsigned cloudIndex = 0;
117  result == CC_FERR_NO_ERROR || result == CC_FERR_NO_LOAD;
118  cloudIndex++) {
119  unsigned width = 0, height = 0;
120  ccGLMatrixd sensorTransD, cloudTransD;
121 
122  // read header
123  {
124  QString line = inFile.readLine();
125  if (line.isNull() &&
126  container.getChildrenNumber() != 0) // end of file?
127  break;
128 
129  // read the width (number of columns) and the height (number of
130  // rows) on the two first lines (DGM: we transpose the matrix right
131  // away)
132  bool ok;
133  height = line.toUInt(&ok);
134  if (!ok) return CC_FERR_MALFORMED_FILE;
135  line = inFile.readLine();
136  width = line.toUInt(&ok);
137  if (!ok) return CC_FERR_MALFORMED_FILE;
138 
139  CVLog::Print(QString("[PTX] Scan #%1 - grid size: %2 x %3")
140  .arg(cloudIndex + 1)
141  .arg(height)
142  .arg(width));
143 
144  // read sensor transformation matrix
145  for (int i = 0; i < 4; ++i) {
146  line = inFile.readLine();
147  QStringList tokens = line.split(" ", QtCompat::SkipEmptyParts);
148  if (tokens.size() != 3) return CC_FERR_MALFORMED_FILE;
149 
150  double* colDest = nullptr;
151  if (i == 0) {
152  // Translation
153  colDest = sensorTransD.getTranslation();
154  } else {
155  // X, Y and Z axis
156  colDest = sensorTransD.getColumn(i - 1);
157  }
158 
159  for (int j = 0; j < 3; ++j) {
160  assert(colDest);
161  colDest[j] = tokens[j].toDouble(&ok);
162  if (!ok) return CC_FERR_MALFORMED_FILE;
163  }
164  }
165  // make the transform a little bit cleaner (necessary as it's read
166  // from ASCII!)
167  CleanMatrix(sensorTransD);
168 
169  // read cloud transformation matrix
170  for (int i = 0; i < 4; ++i) {
171  line = inFile.readLine();
172  QStringList tokens = line.split(" ", QtCompat::SkipEmptyParts);
173  if (tokens.size() != 4) return CC_FERR_MALFORMED_FILE;
174 
175  double* col = cloudTransD.getColumn(i);
176  for (int j = 0; j < 4; ++j) {
177  col[j] = tokens[j].toDouble(&ok);
178  if (!ok) return CC_FERR_MALFORMED_FILE;
179  }
180  }
181  // make the transform a little bit cleaner (necessary as it's read
182  // from ASCII!)
183  CleanMatrix(cloudTransD);
184 
185  // handle Global Shift directly on the first cloud's translation!
186  if (cloudIndex == 0) {
187  if (HandleGlobalShift(cloudTransD.getTranslationAsVec3D(),
188  PshiftTrans, preserveCoordinateShift,
189  parameters)) {
190  CVLog::Warning(
191  "[PTXFilter::loadFile] Cloud has be recentered! "
192  "Translation: (%.2f ; %.2f ; %.2f)",
193  PshiftTrans.x, PshiftTrans.y, PshiftTrans.z);
194  }
195  }
196 
197  //'remove' global shift from the sensor and cloud transformation
198  // matrices
199  cloudTransD.setTranslation(cloudTransD.getTranslationAsVec3D() +
200  PshiftTrans);
201  sensorTransD.setTranslation(sensorTransD.getTranslationAsVec3D() +
202  PshiftTrans);
203  }
204 
205  // now we can read the grid cells
206  ccPointCloud* cloud = new ccPointCloud();
207  if (container.getChildrenNumber() == 0) {
208  cloud->setName("unnamed - Cloud");
209  } else {
210  if (container.getChildrenNumber() == 1) {
211  container.getChild(0)->setName(
212  "unnamed - Cloud 1"); // update previous cloud name!
213  }
214  cloud->setName(QString("unnamed - Cloud %1")
215  .arg(container.getChildrenNumber() + 1));
216  }
217 
218  unsigned gridSize = width * height;
219  if (!cloud->reserve(gridSize)) {
220  result = CC_FERR_NOT_ENOUGH_MEMORY;
221  delete cloud;
222  cloud = nullptr;
223  break;
224  }
225 
226  // set global shift
227  if (preserveCoordinateShift) {
228  cloud->setGlobalShift(PshiftTrans);
229  }
230 
231  // intensities
232  ccScalarField* intensitySF =
233  new ccScalarField(CC_PTX_INTENSITY_FIELD_NAME);
234  if (!intensitySF->reserveSafe(static_cast<unsigned>(gridSize))) {
235  CVLog::Warning("[PTX] Not enough memory to load intensities!");
236  intensitySF->release();
237  intensitySF = nullptr;
238  }
239 
240  // grid structure
241  ccPointCloud::Grid::Shared grid(new ccPointCloud::Grid);
242  grid->w = width;
243  grid->h = height;
244  bool hasIndexGrid = true;
245  try {
246  grid->indexes.resize(gridSize, -1); //-1 means no cell/point
247  } catch (const std::bad_alloc&) {
248  CVLog::Warning(
249  "[PTX] Not enough memory to load the grid structure");
250  hasIndexGrid = false;
251  }
252 
253  // read points
254  {
255  cloudViewer::NormalizedProgress nprogress(pDlg.data(), gridSize);
256  if (pDlg) {
257  pDlg->setInfo(qPrintable(
258  QString("Number of cells: %1").arg(gridSize)));
259  pDlg->start();
260  }
261 
262  bool firstPoint = true;
263  bool hasColors = false;
264  bool loadColors = false;
265  bool loadGridColors = false;
266  size_t gridIndex = 0;
267 
268  for (unsigned j = 0; j < height; ++j) {
269  for (unsigned i = 0; i < width; ++i, ++gridIndex) {
270  QString line = inFile.readLine();
271  QStringList tokens =
272  line.split(" ", QtCompat::SkipEmptyParts);
273 
274  if (firstPoint) {
275  hasColors = (tokens.size() == 7);
276  if (hasColors) {
277  loadColors = cloud->reserveTheRGBTable();
278  if (!loadColors) {
279  CVLog::Warning(
280  "[PTX] Not enough memory to load RGB "
281  "colors!");
282  } else if (hasIndexGrid) {
283  // we also load the colors into the grid (as
284  // invalid/missing points can have colors!)
285  try {
286  grid->colors.resize(gridSize,
287  ecvColor::Rgb(0, 0, 0));
288  loadGridColors = true;
289  } catch (const std::bad_alloc&) {
290  CVLog::Warning(
291  "[PTX] Not enough memory to load "
292  "the grid colors");
293  }
294  }
295  }
296  }
297  if ((hasColors && tokens.size() != 7) ||
298  (!hasColors && tokens.size() != 4)) {
299  result = CC_FERR_MALFORMED_FILE;
300  // early stop
301  j = height;
302  break;
303  }
304 
305  double values[4];
306  for (int v = 0; v < 4; ++v) {
307  bool ok;
308  values[v] = tokens[v].toDouble(&ok);
309  if (!ok) {
310  result = CC_FERR_MALFORMED_FILE;
311  // early stop
312  j = height;
313  break;
314  }
315  }
316 
317  // we skip "empty" cells
318  bool pointIsValid =
319  (CCVector3d::fromArray(values).norm2() != 0);
320  if (pointIsValid) {
321  const double* Pd = values;
322  // first point: check for 'big' coordinates
323  if (firstPoint) {
324  if (cloudIndex == 0 &&
325  !cloud->isShifted()) // in case the trans.
326  // matrix was ok!
327  {
328  CCVector3d P(Pd);
329  if (HandleGlobalShift(P, PshiftCloud,
330  preserveCoordinateShift,
331  parameters)) {
332  if (preserveCoordinateShift) {
333  cloud->setGlobalShift(PshiftCloud);
334  }
335  CVLog::Warning(
336  "[PTXFilter::loadFile] Cloud has "
337  "been recentered! Translation: "
338  "(%.2f ; %.2f ; %.2f)",
339  PshiftCloud.x, PshiftCloud.y,
340  PshiftCloud.z);
341  }
342  }
343  firstPoint = false;
344  }
345 
346  // update index grid
347  if (hasIndexGrid) {
348  grid->indexes[gridIndex] = static_cast<int>(
349  cloud->size()); // = index (default value =
350  // -1, means no point)
351  }
352 
353  // add point
354  cloud->addPoint(CCVector3(
355  static_cast<PointCoordinateType>(Pd[0] +
356  PshiftCloud.x),
357  static_cast<PointCoordinateType>(Pd[1] +
358  PshiftCloud.y),
359  static_cast<PointCoordinateType>(
360  Pd[2] + PshiftCloud.z)));
361 
362  // add intensity
363  if (intensitySF) {
364  intensitySF->addElement(
365  static_cast<ScalarType>(values[3]));
366  }
367  }
368 
369  // color
370  if (loadColors && (pointIsValid || loadGridColors)) {
371  ecvColor::Rgb color;
372  for (int c = 0; c < 3; ++c) {
373  bool ok;
374  unsigned temp = tokens[4 + c].toUInt(&ok);
375  ok &= (temp <= 255);
376  if (ok) {
377  color.rgb[c] = static_cast<unsigned char>(temp);
378  } else {
379  result = CC_FERR_MALFORMED_FILE;
380  // early stop
381  j = height;
382  break;
383  }
384  }
385 
386  if (pointIsValid) {
387  cloud->addRGBColor(color);
388  }
389  if (loadGridColors) {
390  assert(!grid->colors.empty());
391  grid->colors[gridIndex] = color;
392  }
393  }
394 
395  if (parameters.parentWidget && !nprogress.oneStep()) {
396  result = CC_FERR_CANCELED_BY_USER;
397  break;
398  }
399  }
400  }
401  }
402 
403  // is there at least one valid point in this grid?
404  if (cloud->size() == 0) {
405  delete cloud;
406  cloud = nullptr;
407  if (intensitySF) {
408  intensitySF->release();
409  intensitySF = nullptr;
410  }
411 
412  CVLog::Warning(
413  QString("[PTX] Scan #%1 is empty?!").arg(cloudIndex + 1));
414  } else {
415  if (result == CC_FERR_NO_LOAD)
416  result = CC_FERR_NO_ERROR; // to make clear that we have loaded
417  // at least something!
418 
419  cloud->resize(cloud->size());
420  if (intensitySF) {
421  assert(intensitySF->currentSize() == cloud->size());
422  intensitySF->resize(cloud->size());
423  intensitySF->computeMinAndMax();
424  int intensitySFIndex = cloud->addScalarField(intensitySF);
425 
426  // keep track of the min and max intensity
427  if (container.getChildrenNumber() == 0) {
428  minIntensity = intensitySF->getMin();
429  maxIntensity = intensitySF->getMax();
430  } else {
431  minIntensity =
432  std::min(minIntensity, intensitySF->getMin());
433  maxIntensity =
434  std::max(maxIntensity, intensitySF->getMax());
435  }
436 
437  cloud->showSF(true);
438  cloud->setCurrentDisplayedScalarField(intensitySFIndex);
439  }
440 
441  ccGBLSensor* sensor = 0;
442  if (hasIndexGrid && result != CC_FERR_CANCELED_BY_USER) {
443  // determine best sensor parameters (mainly yaw and pitch steps)
444  ccGLMatrix cloudToSensorTrans(
445  (sensorTransD.inverse() * cloudTransD).data());
446  sensor = ccGriddedTools::ComputeBestSensor(cloud, grid,
447  &cloudToSensorTrans);
448  }
449 
450  // we apply the transformation
451  ccGLMatrix cloudTrans(cloudTransD.data());
452  cloud->applyGLTransformation_recursive(&cloudTrans);
453  // this transformation is of no interest for the user
454  // cloud->resetGLTransformationHistory_recursive();
455 
456  if (sensor) {
457  ccGLMatrix sensorTrans(sensorTransD.data());
458  sensor->setRigidTransformation(
459  sensorTrans); // after
460  // cloud->applyGLTransformation_recursive!
461  cloud->addChild(sensor);
462  }
463 
464  // scan grid
465  if (hasIndexGrid) {
466  grid->validCount = static_cast<unsigned>(cloud->size());
467  grid->minValidIndex = 0;
468  grid->maxValidIndex = grid->validCount - 1;
469  grid->sensorPosition = sensorTransD;
470  cloud->addGrid(grid);
471 
472  // by default we don't compute normals without asking the user
473  if (parameters.autoComputeNormals) {
474  cloud->computeNormalsWithGrids(1.0,
475  normalsProgressDlg.data());
476  }
477  }
478 
479  cloud->setVisible(true);
480  cloud->showColors(cloud->hasColors());
481  cloud->showNormals(cloud->hasNormals());
482 
483  container.addChild(cloud);
484 
485 #ifdef QT_DEBUG
486  // break;
487 #endif
488  }
489  }
490 
491  // update scalar fields saturation (globally!)
492  {
493  bool validIntensityRange = true;
494  if (minIntensity < 0 || maxIntensity > 1.0) {
495  CVLog::Warning(
496  "[PTX] Intensity values are invalid (they should all fall "
497  "in [0 ; 1])");
498  validIntensityRange = false;
499  }
500 
501  for (unsigned i = 0; i < container.getChildrenNumber(); ++i) {
502  ccHObject* obj = container.getChild(i);
503  assert(obj && obj->isA(CV_TYPES::POINT_CLOUD));
504  cloudViewer::ScalarField* sf =
505  static_cast<ccPointCloud*>(obj)->getScalarField(0);
506  if (sf) {
507  ccScalarField* ccSF = static_cast<ccScalarField*>(sf);
508  ccSF->setColorScale(ccColorScalesManager::GetDefaultScale(
509  validIntensityRange
510  ? ccColorScalesManager::ABS_NORM_GREY
511  : ccColorScalesManager::GREY));
512  ccSF->setSaturationStart(0 /*minIntensity*/);
513  ccSF->setSaturationStop(maxIntensity);
514  }
515  }
516  }
517 
518  return result;
519 }
