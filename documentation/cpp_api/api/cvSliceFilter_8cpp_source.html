44 void cvSliceFilter::apply() {
45  if (!m_dataObject || m_keepMode) return;
46 
47  VtkUtils::vtkInitOnce(m_cutter);
48  m_cutter->SetInputData(m_dataObject);
49 
50  switch (cutType()) {
51  case cvCutFilter::Plane: {
52  VTK_CREATE(vtkPlane, plane);
53  plane->SetOrigin(m_origin);
54  plane->SetNormal(m_normal);
55  m_cutter->SetCutFunction(plane);
56  } break;
57 
58  case cvCutFilter::Box: {
59  m_cutter->SetCutFunction(m_planes);
60  } break;
61 
62  case cvCutFilter::Sphere: {
63  VTK_CREATE(vtkSphere, sphere);
64  sphere->SetCenter(m_center);
65  sphere->SetRadius(m_radius);
66  m_cutter->SetCutFunction(sphere);
67  } break;
68  }
69 
70  if (!m_filterActor) {
71  VtkUtils::vtkInitOnce(m_filterActor);
72  vtkSmartPointer<vtkLookupTable> lut =
73  createLookupTable(scalarMin(), scalarMax());
74  VTK_CREATE(vtkPolyDataMapper, mapper);
75  mapper->SetInputConnection(m_cutter->GetOutputPort());
76  mapper->SetLookupTable(lut);
77  m_filterActor->SetMapper(mapper);
78  addActor(m_filterActor);
79  }
80 
81  // update contour lines
82  if (m_contourLinesActor && m_contourLinesActor->GetVisibility())
83  showContourLines();
84 
85  applyDisplayEffect();
86 }

88 ccHObject* cvSliceFilter::getOutput() {
89  if (!m_cutter) return nullptr;
90 
91  // update Cutter
92  m_cutter->GenerateCutScalarsOff();
93  m_cutter->Update();
94 
95  // get cut strips polyData
96  VtkUtils::vtkInitOnce(m_cutStrips);
97  m_cutStrips->SetInputData(m_cutter->GetOutput());
98  m_cutStrips->Update();
99  vtkPolyData* polydata = m_cutStrips->GetOutput();
100  if (nullptr == polydata) return nullptr;
101  ccHObject* result = new ccHObject();
102  ccHObject::Container container =
103  vtk2cc::ConvertToMultiPolylines(polydata, "Slice", ecvColor::green);
104 
105  if (!container.empty() && m_entity) {
106  for (auto& obj : container) {
107  if (!obj) {
108  continue;
109  }
110  ccPolyline* poly = ccHObjectCaster::ToPolyline(obj);
111  if (!poly) continue;
112 
113  if (m_entity->isKindOf(CV_TYPES::POINT_CLOUD)) {
114  ccPointCloud* ccCloud = ccHObjectCaster::ToPointCloud(m_entity);
115  poly->setGlobalScale(ccCloud->getGlobalScale());
116  poly->setGlobalShift(ccCloud->getGlobalShift());
117  } else if (m_entity->isKindOf(CV_TYPES::MESH)) {
118  ccMesh* mesh = ccHObjectCaster::ToMesh(m_entity);
119  poly->setGlobalScale(
120  mesh->getAssociatedCloud()->getGlobalScale());
121  poly->setGlobalShift(
122  mesh->getAssociatedCloud()->getGlobalShift());
123  }
124 
125  poly->setName(m_entity->getName() + "-" + poly->getName());
126  result->addChild(poly);
127  }
128  } else {
129  delete result;
130  result = nullptr;
131  }
132  return result;
133 }
