52 //----------------------------------------------------------------------------
53 vtkInformation* vtkPVPostFilterExecutive::GetPostArrayToProcessInformation(int idx)
54 {
55  // add this info into the algorithms info object
56  vtkInformationVector* inArrayVec =
57  this->Algorithm->GetInformation()->Get(POST_ARRAYS_TO_PROCESS());
58  if (!inArrayVec)
59  {
60  inArrayVec = vtkInformationVector::New();
61  this->Algorithm->GetInformation()->Set(POST_ARRAYS_TO_PROCESS(), inArrayVec);
62  inArrayVec->Delete();
63  }
64  vtkInformation* inArrayInfo = inArrayVec->GetInformationObject(idx);
65  if (!inArrayInfo)
66  {
67  inArrayInfo = vtkInformation::New();
68  inArrayVec->SetInformationObject(idx, inArrayInfo);
69  inArrayInfo->Delete();
70  }
71  return inArrayInfo;
72 }

85 //----------------------------------------------------------------------------
86 bool vtkPVPostFilterExecutive::MatchingPropertyInformation(
87  vtkInformation* inputArrayInfo, vtkInformation* postArrayInfo)
88 {
89  return (inputArrayInfo && postArrayInfo && inputArrayInfo->Has(vtkDataObject::FIELD_NAME()) &&
90  postArrayInfo->Has(vtkDataObject::FIELD_NAME()) &&
91  inputArrayInfo->Get(vtkAlgorithm::INPUT_PORT()) ==
92  postArrayInfo->Get(vtkAlgorithm::INPUT_PORT()) &&
93  inputArrayInfo->Get(vtkAlgorithm::INPUT_CONNECTION()) ==
94  postArrayInfo->Get(vtkAlgorithm::INPUT_CONNECTION()) &&
95  inputArrayInfo->Get(vtkDataObject::FIELD_ASSOCIATION()) ==
96  postArrayInfo->Get(vtkDataObject::FIELD_ASSOCIATION()) &&
97  strcmp(inputArrayInfo->Get(vtkDataObject::FIELD_NAME()),
98  postArrayInfo->Get(vtkDataObject::FIELD_NAME())) == 0);
99 }
