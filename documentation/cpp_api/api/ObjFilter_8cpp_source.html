59 CC_FILE_ERROR ObjFilter::saveToFile(ccHObject* entity,
60  const QString& filename,
61  const SaveParameters& parameters) {
62  if (!entity) return CC_FERR_BAD_ARGUMENT;
63 
64  if (!entity->isKindOf(CV_TYPES::MESH)) {
65  CVLog::Warning(
66  "[OBJ] This filter can only save one mesh (optionally with "
67  "sub-meshes) at a time!");
68  return CC_FERR_BAD_ENTITY_TYPE;
69  }
70 
71  // mesh
72  ccGenericMesh* mesh = ccHObjectCaster::ToGenericMesh(entity);
73  if (!mesh || mesh->size() == 0) {
74  CVLog::Warning("[OBJ] Input mesh is empty!");
75  return CC_FERR_NO_SAVE;
76  }
77 
78  // vertices
79  ccGenericPointCloud* vertices = mesh->getAssociatedCloud();
80  if (!vertices || vertices->size() == 0) {
81  CVLog::Warning("[OBJ] Input mesh has no vertices?!");
82  return CC_FERR_NO_SAVE;
83  }
84  unsigned nbPoints = vertices->size();
85 
86  // try to open file for saving
87  QFile file(filename);
88  if (!file.open(QFile::Text | QFile::WriteOnly)) return CC_FERR_WRITING;
89 
90  // progress
91  QScopedPointer<ecvProgressDialog> pDlg(nullptr);
92  if (parameters.parentWidget) {
93  pDlg.reset(new ecvProgressDialog(true, parameters.parentWidget));
94  pDlg->setMethodTitle(
95  QObject::tr("Saving mesh [%1]").arg(mesh->getName()));
96  pDlg->setInfo(QObject::tr("Writing %1 vertices").arg(nbPoints));
97  pDlg->setAutoClose(
98  false); // don't close dialogue when progress bar is full
99  pDlg->start();
100  }
101  cloudViewer::NormalizedProgress nprogress(pDlg.data(), nbPoints);
102 
103  QTextStream stream(&file);
104  stream.setRealNumberNotation(QTextStream::FixedNotation);
105  stream.setRealNumberPrecision(
106  sizeof(PointCoordinateType) == 4 && !vertices->isShifted() ? 8
107  : 12);
108 
109  stream << "# " << FileIO::createdBy() << QtCompat::endl;
110  stream << "# " << FileIO::createdDateTime() << QtCompat::endl;
111 
112  if (file.error() != QFile::NoError) return CC_FERR_WRITING;
113 
114  for (unsigned i = 0; i < nbPoints; ++i) {
115  const CCVector3* P = vertices->getPoint(i);
116  CCVector3d Pglobal = vertices->toGlobal3d<PointCoordinateType>(*P);
117  stream << "v " << Pglobal.x << " " << Pglobal.y << " " << Pglobal.z
118  << QtCompat::endl;
119  if (file.error() != QFile::NoError) return CC_FERR_WRITING;
120  if (pDlg &&
121  !nprogress
122  .oneStep()) // update progress bar, check cancel requested
123  return CC_FERR_CANCELED_BY_USER;
124  }
125 
126  // normals
127  bool withTriNormals = mesh->hasTriNormals();
128  bool withVertNormals = vertices->hasNormals();
129  bool withNormals = withTriNormals || withVertNormals;
130  if (withNormals) {
131  // per-triangle normals
132  if (withTriNormals) {
133  NormsIndexesTableType* normsTable = mesh->getTriNormsTable();
134  if (normsTable) {
135  // reset save dialog
136  unsigned numTriangleNormals = normsTable->currentSize();
137  if (pDlg)
138  pDlg->setInfo(QObject::tr("Writing $1 triangle normals")
139  .arg(numTriangleNormals));
140  nprogress.scale(numTriangleNormals);
141  nprogress.reset();
142 
143  for (unsigned i = 0; i < numTriangleNormals; ++i) {
144  const CCVector3& normalVec =
145  ccNormalVectors::GetNormal(normsTable->getValue(i));
146  stream << "vn " << normalVec.x << " " << normalVec.y << " "
147  << normalVec.z << QtCompat::endl;
148  if (file.error() != QFile::NoError) return CC_FERR_WRITING;
149 
150  // increment progress bar
151  if (pDlg && !nprogress.oneStep()) // cancel requested
152  return CC_FERR_CANCELED_BY_USER;
153  }
154  } else {
155  assert(false);
156  withTriNormals = false;
157  }
158  }
159  // per-vertices normals
160  else // if (withVertNormals)
161  {
162  // reset save dialog
163  if (pDlg)
164  pDlg->setInfo(
165  QObject::tr("Writing %1 vertex normals").arg(nbPoints));
166  nprogress.scale(nbPoints);
167  nprogress.reset();
168 
169  for (unsigned i = 0; i < nbPoints; ++i) {
170  const CCVector3& normalVec = vertices->getPointNormal(i);
171  stream << "vn " << normalVec.x << " " << normalVec.y << " "
172  << normalVec.z << QtCompat::endl;
173  if (file.error() != QFile::NoError) return CC_FERR_WRITING;
174 
175  // increment progress bar
176  if (pDlg && !nprogress.oneStep()) // cancel requested
177  {
178  return CC_FERR_CANCELED_BY_USER;
179  }
180  }
181  }
182  }
183 
184  // materials
185  const ccMaterialSet* materials = mesh->getMaterialSet();
186  bool withMaterials = (materials && mesh->hasMaterials());
187  if (withMaterials) {
188  // reset save dialog
189  if (pDlg)
190  pDlg->setInfo(
191  QObject::tr("Writing %1 materials").arg(materials->size()));
192  nprogress.scale(1);
193  nprogress.reset();
194 
195  // save mtl file
196  QStringList errors;
197  QString baseName = QFileInfo(filename).baseName();
198  if (materials->saveAsMTL(QFileInfo(filename).absolutePath(), baseName,
199  errors)) {
200  stream << "mtllib " << baseName << ".mtl" << QtCompat::endl;
201  if (file.error() != QFile::NoError) return CC_FERR_WRITING;
202  } else {
203  materials = nullptr;
204  withMaterials = false;
205  }
206 
207  // display potential 'errors'
208  for (int i = 0; i < errors.size(); ++i) {
209  CVLog::Warning(QString("[OBJ][Material file writer] ") + errors[i]);
210  }
211 
212  // increment progress bar
213  if (pDlg && !nprogress.oneStep()) // cancel requested
214  return CC_FERR_CANCELED_BY_USER;
215  }
216 
217  // save texture coordinates
218  bool withTexCoordinates =
219  withMaterials && mesh->hasPerTriangleTexCoordIndexes();
220  if (withTexCoordinates) {
221  TextureCoordsContainer* texCoords = mesh->getTexCoordinatesTable();
222  if (texCoords) {
223  // reset save dialog
224  unsigned numTexCoords = texCoords->currentSize();
225  if (pDlg) {
226  pDlg->setInfo(QObject::tr("Writing %1 texture coordinates")
227  .arg(numTexCoords));
228  }
229  nprogress.scale(numTexCoords);
230  nprogress.reset();
231 
232  for (unsigned i = 0; i < texCoords->currentSize(); ++i) {
233  const TexCoords2D& tc = texCoords->getValue(i);
234  stream << "vt " << tc.tx << " " << tc.ty << QtCompat::endl;
235  if (file.error() != QFile::NoError) return CC_FERR_WRITING;
236 
237  // increment progress bar
238  if (pDlg && !nprogress.oneStep()) // cancel requested
239  {
240  return CC_FERR_CANCELED_BY_USER;
241  }
242  }
243  } else {
244  withTexCoordinates = false;
245  }
246  }
247 
248  ccHObject::Container subMeshes;
249  // look for sub-meshes
250  mesh->filterChildren(subMeshes, false, CV_TYPES::SUB_MESH);
251  // check that the number of facets is the same as the full mesh!
252  {
253  unsigned faceCount = 0;
254  for (ccHObject::Container::const_iterator it = subMeshes.begin();
255  it != subMeshes.end(); ++it)
256  faceCount += static_cast<ccSubMesh*>(*it)->size();
257 
258  // if there's no face (i.e. no sub-mesh) or less face than the total
259  // mesh, we save the full mesh!
260  if (faceCount < mesh->size()) {
261  subMeshes.clear();
262  subMeshes.push_back(mesh);
263  }
264  }
265 
266  // reset save dialog for triangles
267  unsigned numTriangles = mesh->size();
268  if (pDlg) {
269  pDlg->setInfo(QObject::tr("Writing %1 triangles").arg(numTriangles));
270  pDlg->setAutoClose(
271  true); //(re-enable) close dialogue when progress bar is full
272  }
273  nprogress.scale(numTriangles);
274  nprogress.reset();
275 
276  // mesh or sub-meshes
277  unsigned indexShift = 0;
278  for (ccHObject::Container::const_iterator it = subMeshes.begin();
279  it != subMeshes.end(); ++it) {
280  ccGenericMesh* st = static_cast<ccGenericMesh*>(*it);
281 
282  stream << "g " << (st->getName().isNull() ? "mesh" : st->getName())
283  << QtCompat::endl;
284  if (file.error() != QFile::NoError) return CC_FERR_WRITING;
285 
286  unsigned triNum = st->size();
287  st->placeIteratorAtBeginning();
288 
289  int lastMtlIndex = -1;
290  int t1 = -1, t2 = -1, t3 = -1;
291 
292  for (unsigned i = 0; i < triNum; ++i) {
293  if (withMaterials) {
294  int mtlIndex = mesh->getTriangleMtlIndex(indexShift + i);
295  if (mtlIndex != lastMtlIndex) {
296  if (mtlIndex >= 0 &&
297  mtlIndex < static_cast<int>(materials->size())) {
298  ccMaterial::CShared mat = materials->at(mtlIndex);
299  stream << "usemtl " << mat->getName() << QtCompat::endl;
300  } else {
301  stream << "usemtl " << QtCompat::endl;
302  }
303  if (file.error() != QFile::NoError) return CC_FERR_WRITING;
304  lastMtlIndex = mtlIndex;
305  }
306 
307  if (withTexCoordinates) {
308  mesh->getTriangleTexCoordinatesIndexes(indexShift + i, t1,
309  t2, t3);
310  if (t1 >= 0) ++t1;
311  if (t2 >= 0) ++t2;
312  if (t3 >= 0) ++t3;
313  }
314  }
315 
316  const cloudViewer::VerticesIndexes* tsi =
317  st->getNextTriangleVertIndexes();
318  // for per-triangle normals
319  unsigned i1 = tsi->i1 + 1;
320  unsigned i2 = tsi->i2 + 1;
321  unsigned i3 = tsi->i3 + 1;
322 
323  stream << "f";
324  if (withNormals) {
325  int n1 = static_cast<int>(i1);
326  int n2 = static_cast<int>(i2);
327  int n3 = static_cast<int>(i3);
328  if (withTriNormals) {
329  st->getTriangleNormalIndexes(i, n1, n2, n3);
330  if (n1 >= 0) ++n1;
331  if (n2 >= 0) ++n2;
332  if (n3 >= 0) ++n3;
333  }
334 
335  if (withTexCoordinates) {
336  stream << " " << i1 << "/" << t1 << "/" << n1;
337  stream << " " << i2 << "/" << t2 << "/" << n2;
338  stream << " " << i3 << "/" << t3 << "/" << n3;
339  } else {
340  stream << " " << i1 << "//" << n1;
341  stream << " " << i2 << "//" << n2;
342  stream << " " << i3 << "//" << n3;
343  }
344  } else {
345  if (withTexCoordinates) {
346  stream << " " << i1 << "/" << t1;
347  stream << " " << i2 << "/" << t2;
348  stream << " " << i3 << "/" << t3;
349  } else {
350  stream << " " << i1;
351  stream << " " << i2;
352  stream << " " << i3;
353  }
354  }
355  stream << QtCompat::endl;
356 
357  if (file.error() != QFile::NoError) {
358  return CC_FERR_WRITING;
359  }
360 
361  if (pDlg && !nprogress.oneStep()) // cancel requested
362  {
363  return CC_FERR_CANCELED_BY_USER;
364  }
365  }
366 
367  stream << "#" << triNum << " faces" << QtCompat::endl;
368  if (file.error() != QFile::NoError) {
369  return CC_FERR_WRITING;
370  }
371 
372  indexShift += triNum;
373  }
374 
375  return CC_FERR_NO_ERROR;
376 }

422 CC_FILE_ERROR ObjFilter::loadFile(const QString& filename,
423  ccHObject& container,
424  LoadParameters& parameters) {
425  CVLog::Print(QString("[OBJ] ") + filename);
426 
427  // open file
428  QFile file(filename);
429  if (!file.open(QFile::ReadOnly)) return CC_FERR_READING;
430  QTextStream stream(&file);
431 
432  // current vertex shift
433  CCVector3d Pshift(0, 0, 0);
434 
435  // vertices
436  ccPointCloud* vertices = new ccPointCloud("vertices");
437  int pointsRead = 0;
438 
439  // facets
440  unsigned int facesRead = 0;
441  unsigned int totalFacesRead = 0;
442  int maxVertexIndex = -1;
443 
444  // base mesh
445  ccMesh* baseMesh = new ccMesh(vertices);
446  baseMesh->setName(
447  QFileInfo(filename).fileName()); // Use full filename with
448  // extension (ParaView-style)
449  // we need some space already reserved!
450  if (!baseMesh->reserve(128)) {
451  CVLog::Error("Not engouh memory!");
452  return CC_FERR_NOT_ENOUGH_MEMORY;
453  }
454 
455  // groups (starting index + name)
456  std::vector<std::pair<unsigned, QString>> groups;
457 
458  // materials
459  ccMaterialSet* materials = nullptr;
460  bool hasMaterial = false;
461  int currentMaterial = -1;
462  bool currentMaterialDefined = false;
463  bool materialsLoadFailed = true;
464 
465  // texture coordinates
466  TextureCoordsContainer* texCoords = nullptr;
467  bool hasTexCoords = false;
468  int texCoordsRead = 0;
469  int maxTexCoordIndex = -1;
470 
471  // normals
472  NormsIndexesTableType* normals = nullptr;
473  int normsRead = 0;
474  bool normalsPerFacet = false;
475  int maxTriNormIndex = -1;
476 
477  // progress dialog
478  QScopedPointer<ecvProgressDialog> pDlg(nullptr);
479  if (parameters.parentWidget) {
480  pDlg.reset(new ecvProgressDialog(true, parameters.parentWidget));
481  pDlg->setMethodTitle(QObject::tr("OBJ file"));
482  pDlg->setInfo(QObject::tr("Loading in progress..."));
483  pDlg->setRange(0, static_cast<int>(file.size()));
484  pDlg->show();
485  QApplication::processEvents();
486  }
487 
488  // common warnings that can appear multiple time (we avoid to send too many
489  // messages to the console!)
490  enum OBJ_WARNINGS {
491  INVALID_NORMALS = 0,
492  INVALID_INDEX = 1,
493  NOT_ENOUGH_MEMORY = 2,
494  INVALID_LINE = 3,
495  CANCELLED_BY_USER = 4,
496  };
497  bool objWarnings[5] = {false, false, false, false, false};
498  bool error = false;
499 
500  try {
501  unsigned lineCount = 0;
502  unsigned polyCount = 0;
503  QString currentLine = stream.readLine();
504  while (!currentLine.isNull()) {
505  ++lineCount;
506  if (pDlg && ((lineCount % 2048) == 0)) {
507  if (pDlg->wasCanceled()) {
508  error = true;
509  objWarnings[CANCELLED_BY_USER] = true;
510  break;
511  }
512  pDlg->setValue(static_cast<int>(file.pos()));
513  QApplication::processEvents();
514  }
515 
516  // specific case for weird files
517  while (currentLine.endsWith('\\')) {
518  currentLine.resize(currentLine.length() - 1);
519  currentLine += stream.readLine();
520  ++lineCount;
521  if (pDlg && ((lineCount % 2048) == 0)) {
522  if (pDlg->wasCanceled()) {
523  error = true;
524  objWarnings[CANCELLED_BY_USER] = true;
525  break;
526  }
527  pDlg->setValue(static_cast<int>(file.pos()));
528  QApplication::processEvents();
529  }
530  }
531 
532  QStringList tokens = qtCompatSplitRegex(currentLine, "\\s+",
533  QtCompat::SkipEmptyParts);
534 
535  // skip comments & empty lines
536  if (tokens.empty() ||
537  tokens.front().startsWith('/', Qt::CaseInsensitive) ||
538  tokens.front().startsWith('#', Qt::CaseInsensitive)) {
539  currentLine = stream.readLine();
540  continue;
541  }
542 
543  /*** new vertex ***/
544  if (tokens.front() == "v") {
545  // reserve more memory if necessary
546  if (vertices->size() == vertices->capacity()) {
547  if (!vertices->reserve(vertices->capacity() +
548  ccChunk::SIZE)) {
549  objWarnings[NOT_ENOUGH_MEMORY] = true;
550  error = true;
551  break;
552  }
553  }
554 
555  // malformed line?
556  if (tokens.size() < 4) {
557  objWarnings[INVALID_LINE] = true;
558  error = true;
559  break;
560  }
561 
562  CCVector3d Pd(tokens[1].toDouble(), tokens[2].toDouble(),
563  tokens[3].toDouble());
564 
565  // first point: check for 'big' coordinates
566  if (pointsRead == 0) {
567  bool preserveCoordinateShift = true;
568  if (HandleGlobalShift(Pd, Pshift, preserveCoordinateShift,
569  parameters)) {
570  if (preserveCoordinateShift) {
571  vertices->setGlobalShift(Pshift);
572  }
573  CVLog::Warning(
574  "[OBJ] Cloud has been recentered! Translation: "
575  "(%.2f ; %.2f ; %.2f)",
576  Pshift.x, Pshift.y, Pshift.z);
577  }
578  }
579 
580  // shifted point
581  CCVector3 P = CCVector3::fromArray((Pd + Pshift).u);
582  vertices->addPoint(P);
583  ++pointsRead;
584  }
585  /*** new vertex texture coordinates ***/
586  else if (tokens.front() == "vt") {
587  // create and reserve memory for tex. coords container if
588  // necessary
589  if (!texCoords) {
590  texCoords = new TextureCoordsContainer();
591  texCoords->link();
592  }
593  if (texCoords->currentSize() == texCoords->capacity()) {
594  if (!texCoords->reserveSafe(texCoords->capacity() +
595  ccChunk::SIZE)) {
596  objWarnings[NOT_ENOUGH_MEMORY] = true;
597  error = true;
598  break;
599  }
600  }
601 
602  // malformed line?
603  if (tokens.size() < 2) {
604  objWarnings[INVALID_LINE] = true;
605  error = true;
606  break;
607  }
608 
609  TexCoords2D T(tokens[1].toFloat(), 0);
610 
611  if (tokens.size() >
612  2) // OBJ specification allows for only one value!!!
613  {
614  T.ty = tokens[2].toFloat();
615  }
616 
617  texCoords->addElement(T);
618  ++texCoordsRead;
619  }
620  /*** new vertex normal ***/
621  else if (tokens.front() ==
622  "vn") //--> in fact it can also be a facet normal!!!
623  {
624  // create and reserve memory for normals container if necessary
625  if (!normals) {
626  normals = new NormsIndexesTableType;
627  normals->link();
628  }
629  if (normals->currentSize() == normals->capacity()) {
630  if (!normals->reserveSafe(normals->capacity() +
631  ccChunk::SIZE)) {
632  objWarnings[NOT_ENOUGH_MEMORY] = true;
633  error = true;
634  break;
635  }
636  }
637 
638  // malformed line?
639  if (tokens.size() < 4) {
640  objWarnings[INVALID_LINE] = true;
641  error = true;
642  break;
643  }
644 
645  CCVector3 N(
646  static_cast<PointCoordinateType>(tokens[1].toDouble()),
647  static_cast<PointCoordinateType>(tokens[2].toDouble()),
648  static_cast<PointCoordinateType>(tokens[3].toDouble()));
649 
650  if (fabs(N.norm2() - 1.0) > 0.005) {
651  objWarnings[INVALID_NORMALS] = true;
652  N.normalize();
653  }
654  CompressedNormType nIndex = ccNormalVectors::GetNormIndex(N.u);
655 
656  normals->addElement(
657  nIndex); // we don't know yet if it's per-vertex or
658  // per-triangle normal...
659  ++normsRead;
660  }
661  /*** new group ***/
662  else if (tokens.front() == "g" || tokens.front() == "o") {
663  // update new group index
664  facesRead = 0;
665  // get the group name
666  QString groupName =
667  (tokens.size() > 1 && !tokens[1].isEmpty() ? tokens[1]
668  : "default");
669  for (int i = 2; i < tokens.size(); ++i) // multiple parts?
670  groupName.append(QString(" ") + tokens[i]);
671  // push previous group descriptor (if none was pushed)
672  if (groups.empty() && totalFacesRead > 0)
673  groups.emplace_back(0, "default");
674  // push new group descriptor
675  if (!groups.empty() && groups.back().first == totalFacesRead)
676  groups.back().second =
677  groupName; // simply replace the group name if the
678  // previous group was empty!
679  else
680  groups.emplace_back(totalFacesRead, groupName);
681  polyCount = 0; // restart polyline count at 0!
682  }
683  /*** new face ***/
684  else if (tokens.front().startsWith('f')) {
685  // malformed line?
686  if (tokens.size() < 4) {
687  objWarnings[INVALID_LINE] = true;
688  currentLine = stream.readLine();
689  continue;
690  // error = true;
691  // break;
