30 std::shared_ptr<PointCloudForColoredICP> InitializePointCloudForColoredICP(
31  const ccPointCloud &target,
32  const geometry::KDTreeSearchParamHybrid &search_param) {
33  cloudViewer::utility::LogDebug("InitializePointCloudForColoredICP");
34 
35  geometry::KDTreeFlann tree;
36  tree.SetGeometry(target);
37 
38  auto output = std::make_shared<PointCloudForColoredICP>();
39  output->reserveThePointsTable(target.size());
40  if (target.hasColors()) {
41  output->reserveTheRGBTable();
42  }
43  if (target.hasNormals()) {
44  output->reserveTheNormsTable();
45  }
46 
47  for (unsigned int i = 0; i < target.size(); ++i) {
48  output->addPoint(*target.getPoint(i));
49 
50  if (target.hasColors()) {
51  output->addRGBColor(target.getPointColor(i));
52  }
53 
54  if (target.hasNormals()) {
55  output->addNorm(target.getPointNormal(i));
56  }
57  }
58 
59  size_t n_points = output->size();
60  output->color_gradient_.resize(n_points, Eigen::Vector3d::Zero());
61 
62  Eigen::Vector3d colors;
63  for (size_t k = 0; k < n_points; k++) {
64  const Eigen::Vector3d &vt = output->getEigenPoint(k);
65  const Eigen::Vector3d &nt = output->getEigenNormal(k);
66 
67  const ecvColor::Rgb &col =
68  output->getPointColor(static_cast<unsigned int>(k));
69  colors = ecvColor::Rgb::ToEigen(col);
70  double it = (colors(0) + colors(1) + colors(2)) / 3.0;
71 
72  std::vector<int> point_idx;
73  std::vector<double> point_squared_distance;
74 
75  if (tree.SearchHybrid(vt, search_param.radius_, search_param.max_nn_,
76  point_idx, point_squared_distance) >= 4) {
77  // approximate image gradient of vt's tangential plane
78  size_t nn = point_idx.size();
79  Eigen::MatrixXd A(nn, 3);
80  Eigen::MatrixXd b(nn, 1);
81  A.setZero();
82  b.setZero();
83  for (size_t i = 1; i < nn; i++) {
84  int P_adj_idx = point_idx[i];
85  Eigen::Vector3d vt_adj = output->getEigenPoint(P_adj_idx);
86  Eigen::Vector3d vt_proj = vt_adj - (vt_adj - vt).dot(nt) * nt;
87 
88  const ecvColor::Rgb &col = output->getPointColor(
89  static_cast<unsigned int>(P_adj_idx));
90  colors = ecvColor::Rgb::ToEigen(col);
91  double it_adj = (colors(0) + colors(1) + colors(2)) / 3.0;
92  A(i - 1, 0) = (vt_proj(0) - vt(0));
93  A(i - 1, 1) = (vt_proj(1) - vt(1));
94  A(i - 1, 2) = (vt_proj(2) - vt(2));
95  b(i - 1, 0) = (it_adj - it);
96  }
97  // adds orthogonal constraint
98  A(nn - 1, 0) = (nn - 1) * nt(0);
99  A(nn - 1, 1) = (nn - 1) * nt(1);
100  A(nn - 1, 2) = (nn - 1) * nt(2);
101  b(nn - 1, 0) = 0;
102  // solving linear equation
103  bool is_success = false;
104  Eigen::MatrixXd x;
105  std::tie(is_success, x) = utility::SolveLinearSystemPSD(
106  A.transpose() * A, A.transpose() * b);
107  if (is_success) {
108  output->color_gradient_[k] = x;
109  }
110  }
111  }
112  return output;
113 }

117 Eigen::Matrix4d TransformationEstimationForColoredICP::ComputeTransformation(
118  const ccPointCloud &source,
119  const ccPointCloud &target,
120  const CorrespondenceSet &corres) const {
121  if (corres.empty()) {
122  utility::LogError(
123  "No correspondences found between source and target "
124  "pointcloud.");
125  }
126  if (!target.hasNormals()) {
127  utility::LogError(
128  "ColoredICP requires target pointcloud to have normals.");
129  }
130  if (!target.hasColors()) {
131  utility::LogError(
132  "ColoredICP requires target pointcloud to have colors.");
133  }
134  if (!source.hasColors()) {
135  utility::LogError(
136  "ColoredICP requires source pointcloud to have colors.");
137  }
138 
139  double sqrt_lambda_geometric = sqrt(lambda_geometric_);
140  double lambda_photometric = 1.0 - lambda_geometric_;
141  double sqrt_lambda_photometric = sqrt(lambda_photometric);
142 
143  const auto &target_c = (const PointCloudForColoredICP &)target;
144 
145  auto compute_jacobian_and_residual =
146  [&](int i,
147  std::vector<Eigen::Vector6d, utility::Vector6d_allocator> &J_r,
148  std::vector<double> &r, std::vector<double> &w) {
149  size_t cs = corres[i][0];
150  size_t ct = corres[i][1];
151  const Eigen::Vector3d &vs = source.getEigenPoint(cs);
152  const Eigen::Vector3d &vt = target.getEigenPoint(ct);
153  const Eigen::Vector3d &nt = target.getEigenNormal(ct);
154 
155  J_r.resize(2);
156  r.resize(2);
157  w.resize(2);
158 
159  J_r[0].block<3, 1>(0, 0) = sqrt_lambda_geometric * vs.cross(nt);
160  J_r[0].block<3, 1>(3, 0) = sqrt_lambda_geometric * nt;
161  r[0] = sqrt_lambda_geometric * (vs - vt).dot(nt);
162  w[0] = kernel_->Weight(r[0]);
163 
164  // project vs into vt's tangential plane
165  Eigen::Vector3d vs_proj = vs - (vs - vt).dot(nt) * nt;
166  const ecvColor::Rgb &col_source =
167  source.getPointColor(static_cast<unsigned int>(cs));
168  Eigen::Vector3d colors = ecvColor::Rgb::ToEigen(col_source);
169  double is = (colors(0) + colors(1) + colors(2)) / 3.0;
170 
171  const ecvColor::Rgb &col_target =
172  target.getPointColor(static_cast<unsigned int>(ct));
173  colors = ecvColor::Rgb::ToEigen(col_target);
174  double it = (colors(0) + colors(1) + colors(2)) / 3.0;
175 
176  const Eigen::Vector3d &dit = target_c.color_gradient_[ct];
177  double is0_proj = (dit.dot(vs_proj - vt)) + it;
178 
179  const Eigen::Matrix3d M =
180  (Eigen::Matrix3d() << 1.0 - nt(0) * nt(0),
181  -nt(0) * nt(1), -nt(0) * nt(2), -nt(0) * nt(1),
182  1.0 - nt(1) * nt(1), -nt(1) * nt(2), -nt(0) * nt(2),
183  -nt(1) * nt(2), 1.0 - nt(2) * nt(2))
184  .finished();
185 
186  const Eigen::Vector3d &ditM = -dit.transpose() * M;
187  J_r[1].block<3, 1>(0, 0) =
188  sqrt_lambda_photometric * vs.cross(ditM);
189  J_r[1].block<3, 1>(3, 0) = sqrt_lambda_photometric * ditM;
190  r[1] = sqrt_lambda_photometric * (is - is0_proj);
191  w[1] = kernel_->Weight(r[1]);
192  };
193 
194  Eigen::Matrix6d JTJ;
195  Eigen::Vector6d JTr;
196  double r2;
197  std::tie(JTJ, JTr, r2) =
198  utility::ComputeJTJandJTr<Eigen::Matrix6d, Eigen::Vector6d>(
199  compute_jacobian_and_residual, (int)corres.size());
200 
201  bool is_success;
202  Eigen::Matrix4d extrinsic;
203  std::tie(is_success, extrinsic) =
204  utility::SolveJacobianSystemAndObtainExtrinsicMatrix(JTJ, JTr);
205 
206  return is_success ? extrinsic : Eigen::Matrix4d::Identity();
207 }

209 double TransformationEstimationForColoredICP::ComputeRMSE(
210  const ccPointCloud &source,
211  const ccPointCloud &target,
212  const CorrespondenceSet &corres) const {
213  double sqrt_lambda_geometric = sqrt(lambda_geometric_);
214  double lambda_photometric = 1.0 - lambda_geometric_;
215  double sqrt_lambda_photometric = sqrt(lambda_photometric);
216  const auto &target_c = (const PointCloudForColoredICP &)target;
217 
218  double residual = 0.0;
219  for (size_t i = 0; i < corres.size(); i++) {
220  size_t cs = corres[i][0];
221  size_t ct = corres[i][1];
222  const Eigen::Vector3d &vs = source.getEigenPoint(cs);
223  const Eigen::Vector3d &vt = target.getEigenPoint(ct);
224  const Eigen::Vector3d &nt = target.getEigenNormal(ct);
225  Eigen::Vector3d vs_proj = vs - (vs - vt).dot(nt) * nt;
226  const ecvColor::Rgb &col_source =
227  source.getPointColor(static_cast<unsigned int>(cs));
228  Eigen::Vector3d colors = ecvColor::Rgb::ToEigen(col_source);
229  double is = (colors(0) + colors(1) + colors(2)) / 3.0;
230 
231  const ecvColor::Rgb &col_target =
232  target.getPointColor(static_cast<unsigned int>(ct));
233  colors = ecvColor::Rgb::ToEigen(col_target);
234  double it = (colors(0) + colors(1) + colors(2)) / 3.0;
235 
236  const Eigen::Vector3d &dit = target_c.color_gradient_[ct];
237  double is0_proj = (dit.dot(vs_proj - vt)) + it;
238  double residual_geometric = sqrt_lambda_geometric * (vs - vt).dot(nt);
239  double residual_photometric = sqrt_lambda_photometric * (is - is0_proj);
240  residual += residual_geometric * residual_geometric +
241  residual_photometric * residual_photometric;
242  }
243  return residual;
244 };
